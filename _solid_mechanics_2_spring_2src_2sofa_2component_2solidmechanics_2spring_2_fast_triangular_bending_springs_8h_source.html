114 #ifdef LOCAL_OPTIM
115  // Optimized version of addDForce
116  void addDForce( VecDeriv& df, const VecDeriv& dp, Real kfactor) const
117  {
118  if( !is_activated ) return;
119 
120  Deriv dpKfact[4];
121 
122  dpKfact[0] = dp[vid[0]] * lambda * kfactor * alpha[0];
123  dpKfact[1] = dp[vid[1]] * lambda * kfactor * alpha[1];
124  dpKfact[2] = dp[vid[2]] * lambda * kfactor * alpha[2];
125  dpKfact[3] = dp[vid[3]] * lambda * kfactor * alpha[3];
126 
127  for( unsigned j=0; j<4; ++j)
128  {
129  for( unsigned k=0; k<4; ++k)
130  {
131  df[vid[j]] -= dpKfact[k] * alpha[j];
132  }
133  }
134  }
135 #else
136  void addDForce( VecDeriv& df, const VecDeriv& dp, Real kfactor) const
137  {
138  if( !is_activated ) return;
139  for( unsigned j=0; j<4; j++ )
140  for( unsigned k=0; k<4; k++ )
141  df[vid[j]] -= dp[vid[k]] * lambda * alpha[j] * alpha[k] * kfactor;
142  }
143 #endif
