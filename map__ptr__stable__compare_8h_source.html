34 template <typename T>
35 class ptr_stable_id
36 {
37 public:
38  ptr_stable_id()
39  : counter(0)
40  , idMap() {}
41 
42  unsigned int operator()(T* p)
43  {
44  const typename std::map<T*,unsigned int>::const_iterator it = idMap.find(p);
45  if (it != idMap.cend())
46  {
47  return it->second;
48  }
49 
50  idMap.insert({p, counter});
51  return counter++;
52  }
53 
54  inline unsigned int id(T* p)
55  {
56  return this->operator()(p);
57  }
58 
59 protected:
60  mutable unsigned int counter;
61  mutable std::map<T*,unsigned int> idMap;
62 };

97  {
98  return std::make_pair(m_ids->id(b.first), m_ids->id(b.second)) >
99  std::make_pair(m_ids->id(a.first), m_ids->id(a.second));
100  }

103  {

107  {
108  return m_ids;
109  }
110 
111 protected:
113  mutable ptr_stable_id<T>* m_ids;
114 
115 private:
