33 class StopperConstraintResolution1Dof : public core::behavior::ConstraintResolution
34 {
35 protected:
36  double _invW, _w, _min, _max ;
37 
38 public:
39 
40  StopperConstraintResolution1Dof(const double &min, const double &max)
41  : core::behavior::ConstraintResolution(1)
42  , _min(min)
43  , _max(max)
44  {
45  }
46 
47  void init(int line, double** w, double *force) override
48  {
49  _w = w[line][line];
50  _invW = 1.0/_w;
51  force[line ] = 0.0;
52  }
53 
54  void resolution(int line, double** /*w*/, double* d, double* force, double*) override
55  {
56  double dfree = d[line] - _w * force[line];
57 
58  if (dfree > _max)
59  force[line] = (_max - dfree) * _invW;
60  else if (dfree < _min)
61  force[line] = (_min - dfree) * _invW;
62  else
63  force[line] = 0;
64  }
65 };
