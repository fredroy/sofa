93  {
94  if (!arg->getAttribute("template"))

99  arg->logError(std::string("No mechanical state with the datatype '") + DataTypes::Name() +
100  "' found in the context node.");
101  return false; // this template is not the same as the existing MechanicalState

103  }
104 
105  return BaseObject::canCreate(obj, context, arg);
