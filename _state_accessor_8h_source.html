69 inline void StateAccessor::computeBBox(const core::ExecParams* params, bool onlyVisible)
70 {
71  SOFA_UNUSED(params);
72 
73  if( !onlyVisible ) return;
74 
75  static constexpr SReal max_real = std::numeric_limits<SReal>::max();
76  static constexpr SReal min_real = std::numeric_limits<SReal>::lowest();
77  SReal maxBBox[3] { min_real, min_real, min_real };
78  SReal minBBox[3] { max_real, max_real, max_real };
79 
80  bool anyMState = false;
81 
82  for (const auto mstate : l_mechanicalStates)
83  {
84  if (mstate)
85  {
86  const auto& bbox = mstate->f_bbox.getValue();
87  for (unsigned int i = 0; i < 3; ++i)
88  {
89  maxBBox[i] = std::max(maxBBox[i], bbox.maxBBox()[i]);
90  minBBox[i] = std::min(minBBox[i], bbox.minBBox()[i]);
91  }
92  anyMState = true;
93  }
94  }
95 
96  if (anyMState)
97  {
98  this->f_bbox.setValue(sofa::type::TBoundingBox(minBBox,maxBBox));
99  }
100 }
