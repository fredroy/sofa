132  void add(core::CollisionModel* cm,sofa::core::collision::Intersection * interMehtod){
133  if((_addedCM.insert(CollModID(cm->getEnumType(),cm))).second){
134 
135  bool swap;
136  core::collision::ElementIntersector * ei = interMehtod->findIntersector(cm,cm,swap);
137  if(ei){
138  _order[cm->getEnumType()][cm->getEnumType()] = 2;
139  _intersectors[cm->getEnumType()][cm->getEnumType()] = ei;
140  }
141 
142  for(std::set<CollModID>::iterator it = _addedCM.begin() ; it != _addedCM.end() ; ++it){
143  if(it->sample->getEnumType() == cm->getEnumType())
144  continue;
145 
146  swap = false;
147  ei = interMehtod->findIntersector(cm,it->sample,swap);
148 
149  if(ei && swap){
150  _order[cm->getEnumType()][it->enum_type] = -1;
151  _order[it->enum_type][cm->getEnumType()] = 1;
152  _intersectors[it->enum_type][cm->getEnumType()] = _intersectors[cm->getEnumType()][it->enum_type] = ei;
153  }
154  else if(ei){
155  _order[cm->getEnumType()][it->enum_type] = 1;
156  _order[it->enum_type][cm->getEnumType()] = -1;
157  _intersectors[it->enum_type][cm->getEnumType()] = _intersectors[cm->getEnumType()][it->enum_type] = ei;

197  for(int j = 0 ; j < sofa::core::CollisionModel::ENUM_TYPE_SIZE ; ++j){

203  ei->intersect(it->second.elem1,it->second.elem2,output);
