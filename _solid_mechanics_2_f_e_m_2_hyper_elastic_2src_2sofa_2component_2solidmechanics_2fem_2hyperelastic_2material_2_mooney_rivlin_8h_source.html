53  virtual Real getStrainEnergy(StrainInformation<DataTypes> *sinfo, const MaterialParameters<DataTypes> &param) {
54  MatrixSym inversematrix;
55  MatrixSym C=sinfo->deformationTensor;
56  invertMatrix(inversematrix,C);
57  Real I1=sinfo->trC;
58  Real I1square=(Real)(C[0]*C[0] + C[2]*C[2]+ C[5]*C[5]+2*(C[1]*C[1] + C[3]*C[3] + C[4]*C[4]));
59  Real I2=(Real)((pow(I1,(Real)2)- I1square)/2);
60  Real c1=param.parameterArray[0];
61  Real c2=param.parameterArray[1];
62  Real k0=param.parameterArray[2];
63  return c1*(I1*pow(sinfo->J,(-2/3))-3)+c2*(I2*pow(sinfo->J,(-4/3)))+k0*log(sinfo->J)*log(sinfo->J)/2;
64 
65  }

83  virtual void applyElasticityTensor(StrainInformation<DataTypes> *sinfo, const MaterialParameters<DataTypes> &param,const MatrixSym& inputTensor, MatrixSym &outputTensor) {
84  MatrixSym inversematrix;
85  MatrixSym C=sinfo->deformationTensor;
86  invertMatrix(inversematrix,C);
87  Real I1=sinfo->trC;
88  Real I1square=(Real)(C[0]*C[0] + C[2]*C[2]+ C[5]*C[5]+2*(C[1]*C[1] + C[3]*C[3] + C[4]*C[4]));
89  Real I2=(Real)((pow(I1,(Real)2)- I1square)/2);
90  Real c1=param.parameterArray[0];
91  Real c2=param.parameterArray[1];
92  Real k0=param.parameterArray[2];
93  MatrixSym ID;
94  ID.identity();
95  // C-1:H
96  Real _trHC=inputTensor[0]*inversematrix[0]+inputTensor[2]*inversematrix[2]+inputTensor[5]*inversematrix[5]
97  +2*inputTensor[1]*inversematrix[1]+2*inputTensor[3]*inversematrix[3]+2*inputTensor[4]*inversematrix[4];
98  MatrixSym Firstmatrix;
99  //C-1HC-1 convert to sym matrix
100  Firstmatrix.Mat2Sym(inversematrix.SymMatMultiply(inputTensor.SymSymMultiply(inversematrix)),Firstmatrix);
101  //C:H
102  Real trHC=inputTensor[0]*C[0]+inputTensor[2]*C[2]+inputTensor[5]*C[5]
103  +2*inputTensor[1]*C[1]+2*inputTensor[3]*C[3]+2*inputTensor[4]*C[4];
104 
105  //trH
106  Real trH=inputTensor[0]+inputTensor[2]+inputTensor[5];
107 
108  outputTensor=((ID-inversematrix*I1/(Real)3.0)*(-_trHC)/(Real)3.0+Firstmatrix*I1/(Real)3.0-inversematrix*trH/(Real)3.0)*(Real)2.0*c1*pow(sinfo->J,(Real)(-2.0/3.0))
109  +((inversematrix*(Real)(-2.0)*I2/(Real)3.0+ID*I1-C)*(Real)(-2.0)*_trHC/(Real)3.0+Firstmatrix*(Real)2.0*I2/(Real)3.0-inversematrix*(Real)2.0*(I1*trH-trHC)/(Real)3.0+ID*trH-inputTensor)*(Real)2.0*c2*pow(sinfo->J,(Real)(-4.0/3.0))
110  +inversematrix*_trHC*k0/(Real)2.0-Firstmatrix*k0*log(sinfo->J);
111 
112  }
