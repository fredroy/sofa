36 class BilateralConstraintResolution : public ConstraintResolution
37 {
38 public:
39  BilateralConstraintResolution(double* initF=nullptr)
40  : ConstraintResolution(1)
41  , _f(initF) {}
42  void resolution(int line, double** w, double* d, double* force, double *dfree) override
43  {
44  SOFA_UNUSED(dfree);
45  force[line] -= d[line] / w[line][line];
46  }
47 
48  void init(int line, double** /*w*/, double* force) override
49  {
50  if(_f) { force[line] = *_f; }
51  }
52 
53  void initForce(int line, double* force) override
54  {
55  if(_f) { force[line] = *_f; }
56  }
57 
58  void store(int line, double* force, bool /*convergence*/) override
59  {
60  if(_f) *_f = force[line];
61  }
62 
63 protected:
64  double* _f;
65 };

67 class BilateralConstraintResolution3Dof : public ConstraintResolution
68 {
69 public:
70 
71  BilateralConstraintResolution3Dof(sofa::type::Vec3d* vec = nullptr)
72  : ConstraintResolution(3)
73  , _f(vec)
74  {
75  }
76  void init(int line, double** w, double *force) override
77  {
78  sofa::type::Mat<3,3,double> temp;
79  temp[0][0] = w[line][line];
80  temp[0][1] = w[line][line+1];
81  temp[0][2] = w[line][line+2];
82  temp[1][0] = w[line+1][line];
83  temp[1][1] = w[line+1][line+1];
84  temp[1][2] = w[line+1][line+2];
85  temp[2][0] = w[line+2][line];
86  temp[2][1] = w[line+2][line+1];
87  temp[2][2] = w[line+2][line+2];
88 
89  const bool canInvert = sofa::type::invertMatrix(invW, temp);
90  assert(canInvert);
91  SOFA_UNUSED(canInvert);
92 
93  // invW is unsused in this scope, remove the warning:
94  SOFA_UNUSED(invW);
95 
96  if(_f)
97  {
98  for(int i=0; i<3; i++)
99  force[line+i] = (*_f)[i];
100  }
101  }
102 
103  void initForce(int line, double* force) override
104  {
105  if(_f)
106  {
107  for(int i=0; i<3; i++)
108  force[line+i] = (*_f)[i];
109  }
110  }
111 
112  void resolution(int line, double** /*w*/, double* d, double* force, double * dFree) override
113  {
114  SOFA_UNUSED(dFree);
115  for(int i=0; i<3; i++)
116  {
117  for(int j=0; j<3; j++)
118  force[line+i] -= d[line+j] * invW[i][j];
119  }
120  }
121 
122  void store(int line, double* force, bool /*convergence*/) override
123  {
124  if(_f)
125  {
126  for(int i=0; i<3; i++)
127  (*_f)[i] = force[line+i];
128  }
129  }
130 
131 protected:
132  sofa::type::Mat<3,3,double> invW;
133  sofa::type::Vec3d* _f;
134 };

136 class BilateralConstraintResolutionNDof : public ConstraintResolution
137 {
138 public:
139  BilateralConstraintResolutionNDof(unsigned blockSize )
140  : ConstraintResolution(blockSize)
141  , wBlock(Eigen::MatrixXd(blockSize, blockSize))
142  {
143  }
144  void init(int line, double** w, double * /*force*/) override
145  {
146  for (auto i = 0; i < wBlock.rows(); ++i)
147  {
148  wBlock.row(i) = Eigen::VectorXd::Map(&w[line + i][line], wBlock.cols());
149  }
150  wBlockInv.compute(wBlock);
151  }
152 
153  void resolution(int line, double** /*w*/, double* displacement, double* force, double* /*dFree*/) override
154  {
155  Eigen::Map< Eigen::VectorXd > f(&force[line], wBlock.cols());
156  Eigen::Map< Eigen::VectorXd > d(&displacement[line], wBlock.cols());
157  Eigen::VectorXd f_local = wBlockInv.solve(d);
158  f -= f_local;
159  }
160 
161 protected:
162  Eigen::MatrixXd wBlock;
163  Eigen::LDLT< Eigen::MatrixXd > wBlockInv;
164 };
