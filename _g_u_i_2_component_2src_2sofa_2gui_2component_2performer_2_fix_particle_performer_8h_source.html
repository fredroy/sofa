89  template<typename TTriangleCollisionModel>
90  static void getFixationPointsTriangle(sofa::core::sptr<sofa::core::CollisionModel> model, const Index idx, type::vector<Index>& points, Coord& fixPoint)
91  {
92  auto* triangle = static_cast<TTriangleCollisionModel*>(model.get());
93 
94  sofa::component::collision::geometry::Triangle t(triangle, idx);
95  fixPoint = (t.p1() + t.p2() + t.p3()) / 3.0;
96  points.push_back(t.p1Index());
97  points.push_back(t.p2Index());
98  points.push_back(t.p3Index());
99  }
100 
101  static void getFixationPointsSphere(sofa::core::sptr<sofa::core::CollisionModel> model, const Index idx, type::vector<Index>& points, Coord& fixPoint)
102  {
103  auto* collisionState = model->getContext()->getMechanicalState();
104  fixPoint[0] = collisionState->getPX(idx);
105  fixPoint[1] = collisionState->getPY(idx);
106  fixPoint[2] = collisionState->getPZ(idx);
107 
108  points.push_back(idx);
109  }
