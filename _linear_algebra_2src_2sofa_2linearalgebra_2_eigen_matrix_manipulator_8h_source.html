50  inline friend std::ostream& operator << ( std::ostream& out, const LLineManipulator& s )
51  {
52  for (InternalData::const_iterator it = s._data.begin(); it!=s._data.end(); ++it)
53  {
54  if (it->second == 1.0)
55  out << "[" << it->first << "] ";
56  else
57  out << "[" << it->first << ", " << it->second <<"] ";
58  }
59  return out;
60  }

62  template <class Container, class Result>
63  void buildCombination(const Container& lines, Result &output) const
64  {
65  //TODO: improve estimation of non zero coeff
66  for (InternalData::const_iterator it=_data.begin(); it!=_data.end(); ++it)
67  {
68  const unsigned int indexConstraint=it->first;
69  const SReal factor=it->second;
70  output += lines[indexConstraint]*factor;
71  }
72  }
73 protected:
74  InternalData _data;
75 };
