50  template<typename Node,
51  typename T = std::decay_t<decltype(*std::begin(std::declval<Node>()))>,
52  typename = std::enable_if_t<std::is_scalar_v<T>>
53  >
54  [[nodiscard]]
55  static constexpr auto squaredLength(const Node& n0, const Node& n1)
56  {
57  constexpr Node v{};
58  constexpr auto size = std::distance(std::cbegin(v), std::cend(v));
59 
60  // specialized function is faster than the generic (using STL) one
61  if constexpr (std::is_same_v< Node, sofa::type::Vec<size, T>>)
62  {
63  return (static_cast<sofa::type::Vec<size, T>>(n1) - static_cast<sofa::type::Vec<size, T>>(n0)).norm2();
64  }
65  else
66  {
67  Node diff{};
68  std::transform(n0.cbegin(), n0.cend(), n1.cbegin(), diff.begin(), std::minus<T>());
69  return std::inner_product(std::cbegin(diff), std::cend(diff), std::cbegin(diff), static_cast<T>(0));
70  }
71  }

104  static constexpr auto pointBaryCoefs(const Node& point, const Node& n0, const Node& n1)
105  {
106  sofa::type::Vec<2, T> baryCoefs;
107  const T dist = (n1 - n0).norm();
108 
109  if (dist < EQUALITY_THRESHOLD)
110  {
111  baryCoefs[0] = 0.5;
112  baryCoefs[1] = 0.5;
113  }
114  else

116  baryCoefs[0] = (point - n1).norm() / dist;
117  baryCoefs[1] = (point - n0).norm() / dist;
118  }

133  template<typename Node,
134  typename T = std::decay_t<decltype(*std::begin(std::declval<Node>()))>,
135  typename = std::enable_if_t<std::is_scalar_v<T>>
136  >
137  [[nodiscard]]
138  static constexpr bool intersectionWithPlane(const Node& n0, const Node& n1, const sofa::type::Vec<3, T>& planeP0, const sofa::type::Vec<3, T>& normal, sofa::type::Vec<3, T>& intersection)
139  {
140  constexpr Node n{};
141  static_assert(std::distance(std::begin(n), std::end(n)) == 3, "Plane - Edge intersection can only be computed in 3 dimensions.");
142 
143  //plane equation
144  const sofa::type::Vec<3, T> planeNorm = normal.normalized();
145  const T d = planeNorm * planeP0;
146 
147  //compute intersection between line and plane equation
148  const T denominator = planeNorm * (n1 - n0);
149  if (denominator < EQUALITY_THRESHOLD)
150  {
151  return false;
152  }
153 
154  const T t = (d - planeNorm * n0) / denominator;
155  if ((t <= 1) && (t >= 0))
156  {
157  intersection = n0 + (n1 - n0) * t;
158  return true;
159  }
160  return false;
161  }
