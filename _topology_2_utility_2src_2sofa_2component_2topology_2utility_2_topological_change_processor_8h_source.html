160  void display()
161  {
162  std::stringstream tmp ;
163  tmp<< "Time to incise: " << timeToIncise << msgendl;
164  tmp<< "Triangle indices : ";
165  for (Index i = 0 ; i < triangleIndices.size() ; i++)
166  tmp<< triangleIndices[i] << " ";
167  tmp<< msgendl;
168  tmp<< "Barycentric coordinates : ";
169  for (Index i = 0 ; i < barycentricCoordinates.size() ; i++)
170  tmp<< barycentricCoordinates[i] << " | " ;
171  tmp<< msgendl;
172  tmp<< "Coordinates : ";
173  for (Index i = 0 ; i < coordinates.size() ; i++)
174  tmp<< coordinates[i] << " | " ;
175  msg_info("TriangleIncisionInformation") << tmp.str() ;
176  }
177 
178 
179  std::vector<type::Vector3> computeCoordinates(core::topology::BaseMeshTopology *topology)
180  {
181  sofa::component::topology::container::dynamic::TriangleSetGeometryAlgorithms<defaulttype::Vec3Types>* triangleGeo;
182  topology->getContext()->get(triangleGeo);
183 
184  coordinates.clear();
185 
186 
187  if (coordinates.size() != triangleIndices.size())
188  {
189  coordinates.resize(triangleIndices.size());
190  }
191 
192  for (unsigned int i = 0 ; i < coordinates.size() ; i++)
193  {
194  defaulttype::Vec3Types::Coord coord[3];
195  Index triIndex = triangleIndices[i];
196 
197  if ( triIndex >= topology->getNbTriangles())
198  {
199  msg_error("TriangleIncisionInformation") << " Bad index to access triangles " << triIndex ;
200  }
201 
202  triangleGeo->getTriangleVertexCoordinates(triIndex, coord);
203 
204  coordinates[i].clear();
205  for (unsigned k = 0 ; k < 3 ; k++)
206  {
207  coordinates[i] += coord[k] * barycentricCoordinates[i][k];
208  }
209  }
210 
211  return coordinates;
212  }
