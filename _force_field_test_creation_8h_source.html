160  void run_test( const VecCoord& x, const VecDeriv& v, const VecDeriv& ef )
161  {
162  if( !(flags & TEST_POTENTIAL_ENERGY) ) msg_warning("ForceFieldTest") << "Potential energy is not tested";
163 
164  if( deltaRange.second / errorMax <= sofa::testing::g_minDeltaErrorRatio )
165  ADD_FAILURE() << "The comparison threshold is too large for the finite difference delta";
166 
167  ASSERT_TRUE(x.size()==v.size());
168  ASSERT_TRUE(x.size()==ef.size());
169  std::size_t n = x.size();
170 
171  // copy the position and velocities to the scene graph
172  this->dof->resize(n);
173  typename DOF::WriteVecCoord xdof = this->dof->writePositions();
174  sofa::testing::copyToData( xdof, x );
175  typename DOF::WriteVecDeriv vdof = this->dof->writeVelocities();
176  sofa::testing::copyToData( vdof, v );
177 
178  // init scene and compute force
179  sofa::simulation::getSimulation()->init(this->node.get());
180  core::MechanicalParams mparams;
181  mparams.setKFactor(1.0);
182  MechanicalResetForceVisitor resetForce(&mparams, core::VecDerivId::force());
183  node->execute(resetForce);
184  MechanicalComputeForceVisitor computeForce( &mparams, core::VecDerivId::force() );
185  this->node->execute(computeForce);
186 
187  // check force
188  typename DOF::ReadVecDeriv f= this->dof->readForces();
189  if(debug){
190  std::cout << "run_test, x = " << x << std::endl;
191  std::cout << " v = " << v << std::endl;
192  std::cout << " expected f = " << ef << std::endl;
193  std::cout << " actual f = " << f.ref() << std::endl;
194  }
195  ASSERT_TRUE( this->vectorMaxDiff(f,ef)< errorMax*this->epsilon() );
196 
197  if( !checkStiffness ) return;
198 
199  // to check the stiffness, generate a change of position, to check the change of force
200 
201  // store current force
202  VecDeriv curF;
203  sofa::testing::copyFromData( curF, dof->readForces() );
204 
205  // Get potential Energy before applying a displacement to dofs
206  SReal potentialEnergyBeforeDisplacement = (flags & TEST_POTENTIAL_ENERGY) ? ((const core::behavior::BaseForceField*)force.get())->getPotentialEnergy(&mparams) : 0;
207 
208  // change position
209  VecDeriv dX(n);
210  for( unsigned i=0; i<n; i++ ){
211  dX[i] = DataTypes::randomDeriv( deltaRange.first * this->epsilon(), deltaRange.second * this->epsilon() ); // todo: better random, with negative values
212  xdof[i] += dX[i];
213  }
214 
215  // compute new force and difference between previous force
216  node->execute(resetForce);
217  node->execute(computeForce);
218  VecDeriv newF;
219  sofa::testing::copyFromData( newF, dof->readForces() );
220  VecDeriv changeOfForce(curF);
221  for( unsigned i=0; i<curF.size(); ++i){
222  changeOfForce[i] = newF[i] - curF[i];
223  }
224 
225  if( flags & TEST_POTENTIAL_ENERGY )
226  {
227  // Get potential energy after displacement of dofs
228  SReal potentialEnergyAfterDisplacement = ((const core::behavior::BaseForceField*)force.get())->getPotentialEnergy(&mparams);
229 
230  // Check getPotentialEnergy() we should have dE = -dX.F
231 
232  // Compute dE = E(x+dx)-E(x)
233  SReal differencePotentialEnergy = potentialEnergyAfterDisplacement-potentialEnergyBeforeDisplacement;
234 
235  // Compute the expected difference of potential energy: -dX.F (dot product between applied displacement and Force)
236  SReal expectedDifferencePotentialEnergy = 0;
237  for( unsigned i=0; i<n; ++i){
238  expectedDifferencePotentialEnergy = expectedDifferencePotentialEnergy - dot(dX[i],curF[i]);
239  }
240 
241  SReal absoluteErrorPotentialEnergy = std::abs(differencePotentialEnergy - expectedDifferencePotentialEnergy);
242  if( absoluteErrorPotentialEnergy> errorFactorPotentialEnergy*errorMax*this->epsilon() ){
243  ADD_FAILURE()<<"dPotentialEnergy differs from -dX.F (threshold=" << errorFactorPotentialEnergy*errorMax*this->epsilon() << ")" << std::endl
244  << "dPotentialEnergy is " << differencePotentialEnergy << std::endl
245  << "-dX.F is " << expectedDifferencePotentialEnergy << std::endl
246  << "Failed seed number = " << this->seed << std::endl;
247  }
248  }
249 
250 
251  // check computeDf: compare its result to actual change
252  node->execute(resetForce);
253  dof->vRealloc( &mparams, core::VecDerivId::dx()); // dx is not allocated by default
254  typename DOF::WriteVecDeriv wdx = dof->writeDx();
255  sofa::testing::copyToData ( wdx, dX );
256  MechanicalComputeDfVisitor computeDf( &mparams, core::VecDerivId::force() );
257  node->execute(computeDf);
258  VecDeriv dF;
259  sofa::testing::copyFromData( dF, dof->readForces() );
260 
261  if( this->vectorMaxDiff(changeOfForce,dF)> errorMax*this->epsilon() ){
262  ADD_FAILURE()<<"dF differs from change of force" << std::endl << "Failed seed number = " << this->seed << std::endl;
263  }
264 
265  // check stiffness matrix: compare its product with dx to actual force change
266  typedef sofa::linearalgebra::EigenBaseSparseMatrix<SReal> Sqmat;
267  Sqmat K( n*DataTypes::deriv_total_size, n*DataTypes::deriv_total_size );
268  sofa::core::behavior::SingleMatrixAccessor accessor( &K );
269  mparams.setKFactor(1.0);
270  force->addKToMatrix( &mparams, &accessor);
271  K.compress();
272  // cout << "stiffness: " << K << endl;
273  modeling::Vector dx;
274  sofa::testing::data_traits<DataTypes>::VecDeriv_to_Vector( dx, dX );
275 
276  modeling::Vector Kdx = K * dx;
277  if( debug ){
278  std::cout << " dX = " << dX << std::endl;
279  std::cout << " newF = " << newF << std::endl;
280  std::cout << " change of force = " << changeOfForce << std::endl;
281  std::cout << " addDforce = " << dF << std::endl;
282  std::cout << " Kdx = " << Kdx.transpose() << std::endl;
283  }
284 
285  modeling::Vector df;
286  sofa::testing::data_traits<DataTypes>::VecDeriv_to_Vector( df, changeOfForce );
287  if( this->vectorMaxDiff(Kdx,df)> errorMax*this->epsilon() )
288  ADD_FAILURE()<<"Kdx differs from change of force"<< std::endl << "Failed seed number = " << this->seed << std::endl;
289 
290  }
291
