242  constexpr inline bool operator < (const fixed_array& v ) const noexcept
243  {
244  for( size_type i=0; i<N; i++ )
245  {
246  if( elems[i]<v[i] )
247  return true; // (*this)<v
248  else if( elems[i]>v[i] )
249  return false; // (*this)>v
250  }
251  return false; // (*this)==v
252  }
