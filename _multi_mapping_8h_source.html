210  template<class T>
211  static bool canCreate(T*& obj, core::objectmodel::BaseContext* context, core::objectmodel::BaseObjectDescription* arg)
212  {
213  std::string input = arg->getAttribute("input","");
214  if (input.empty()) {
215  arg->logError("The 'input' data attribute is empty. It should contain a valid path "
216  "to one or more mechanical states of type '" + std::string(TIn::Name()) + "'.");
217  return false;
218 
219  } else if (!PathResolver::CheckPaths(context, LinkFromModels::DestType::GetClass(), input)) {
220  arg->logError("The 'input' data attribute does not contain a valid path to one or more mechanical "
221  "states of type '" + std::string(TIn::Name()) + "'.");
222  return false;
223  }
224 
225  std::string output = arg->getAttribute("output","");
226  if (output.empty()) {
227  arg->logError("The 'output' data attribute is empty. It should contain a valid path "
228  "to one or more mechanical states. of type '" + std::string(TOut::Name()) + "'.");
229  return false;
230  } else if (!PathResolver::CheckPaths(context, LinkToModels::DestType::GetClass(), output)) {
231  arg->logError("The 'output' data attribute does not contain a valid path to one or more mechanical "
232  "states of type '" + std::string(TOut::Name()) + "'.");
233  return false;
234  }
235 
236  return BaseMapping::canCreate(obj, context, arg);
237  }
