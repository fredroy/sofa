79 template<class TVector>
80 class MatrixLinearSolverInternalData
81 {
82 public:
83  typedef typename TVector::Real Real;
84  typedef sofa::linearalgebra::SparseMatrix<Real> JMatrixType;
85  typedef linearalgebra::BaseMatrix ResMatrixType;
86 
87  template<typename MReal>
88  JMatrixType * copyJmatrix(linearalgebra::SparseMatrix<MReal> * J)
89  {
90  J_local.clear();
91  J_local.resize(J->rowSize(),J->colSize());
92 
93  for (auto jit1 = J->begin(); jit1 != J->end(); jit1++)
94  {
95  auto l = jit1->first;
96  for (auto i1 = jit1->second.begin(); i1 != jit1->second.end(); i1++)
97  {
98  auto c = i1->first;
99  MReal val = i1->second;
100  J_local.set(l,c,val);
101  }
102  }
103  return &J_local;
104  }
105 
106  void projectForceInConstraintSpace(linearalgebra::BaseVector* r,const linearalgebra::BaseVector* f) {
107  for (typename linearalgebra::SparseMatrix<Real>::LineConstIterator jit = J_local.begin(), jitend = J_local.end(); jit != jitend; ++jit) {
108  auto row = jit->first;
109  auto force = f->element(row);
110  for (typename linearalgebra::SparseMatrix<Real>::LElementConstIterator i2 = jit->second.begin(), i2end = jit->second.end(); i2 != i2end; ++i2) {
111  auto col = i2->first;
112  auto val = i2->second;
113  r->add(col,val * force);
114  }
115  }
116  }
117 
118  JMatrixType * getLocalJ() {
119  return &J_local;
120  }
121 
122  JMatrixType * getLocalJ(linearalgebra::BaseMatrix * J)
123  {
124  if (JMatrixType * j = dynamic_cast<JMatrixType *>(J))
125  {
126  return j;
127  }
128  else if (linearalgebra::SparseMatrix<double> * j = dynamic_cast<linearalgebra::SparseMatrix<double> *>(J))
129  {
130  return copyJmatrix(j);
131  }
132  else if (linearalgebra::SparseMatrix<float> * j = dynamic_cast<linearalgebra::SparseMatrix<float> *>(J))
133  {
134  return copyJmatrix(j);
135  }
136  else
137  {
138  J_local.clear();
139  J_local.resize(J->rowSize(),J->colSize());
140 
141  for (typename JMatrixType::Index j=0; j<J->rowSize(); j++)
142  {
143  for (typename JMatrixType::Index i=0; i<J->colSize(); i++)
144  {
145  J_local.set(j,i,J->element(j,i));
146  }
147  }
148 
149  return &J_local;
150  }
151  }
152 
153  ResMatrixType * getLocalRes(linearalgebra::BaseMatrix * R)
154  {
155  return R;
156  }
157 
158 
159  void addLocalRes(linearalgebra::BaseMatrix * /*R*/)
160  {}
161 
162 private :
163  JMatrixType J_local;
164 };
