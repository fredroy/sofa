141  template<typename Matrix1, typename Matrix2>
142  static Real matrixMaxDiff( const Matrix1& m1, const Matrix2& m2 )
143  {
144  if (m1.rows() != m2.rows() || m2.cols() != m1.cols())
145  {
146  ADD_FAILURE() << "Comparison between matrices of different sizes";
147  return infinity();
148  }
149 
150  Real result = 0;
151  for(typename Matrix1::Index i=0; i<m1.rows(); ++i)
152  {
153  for(typename Matrix1::Index j=0; j<m1.cols(); ++j)
154  {
155  const Real diff = std::fabs(m1(i,j) - m2(i,j));
156  if(diff > result)
157  result = diff;
158  }
159  }
160  return result;
161  }

164  template<Size M, Size N, typename Real, typename Matrix2>

166  {
167  Real result = 0;

169  ADD_FAILURE() << "Comparison between matrices of different sizes";

174  Real diff = (Real)fabs(m1[i][j] - m2.element(i,j));
175  if(diff > result)

243  {
244  vec.resize( vcoord.size() * coord_total_size );
245  for( Index i=0; i<vcoord.size(); i++ ){
246  for( Index j=0; j<coord_total_size; j++ )
247  vec[j+coord_total_size*i] = vcoord[i][j];
