86  struct Key

98  struct HashFunction
99  {
100  size_t operator()(const Key &key) const
101  {
102  // We use the large prime numbers proposed in paper:
103  // M.Teschner et al "Optimized Spatial Hashing for Collision Detection of Deformable Objects" (2003)
104  int h = (73856093*key.xId^19349663*key.yId^83492791*key.zId);
105  return size_t(h);
106  }
107  };
108 
109  struct HashEqual
110  {
111  bool operator()(const Key &key1, const Key &key2) const
112  {
113  return ((key1.xId==key2.xId) && (key1.yId==key2.yId) && (key1.zId==key2.zId));
114  }
115  };
