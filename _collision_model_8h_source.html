288  template<class DerivedModel>
289  DerivedModel* createPrevious()
290  {
291  CollisionModel::SPtr prev = previous.get();
292  typename DerivedModel::SPtr pmodel = sofa::core::objectmodel::SPtr_dynamic_cast<DerivedModel>(prev);
293  if (pmodel.get() == nullptr)
294  {
295  int level = 0;
296  CollisionModel *cm = getNext();
297  CollisionModel* root = this;
298  while (cm)
299  {
300  root = cm;
301  cm = cm->getNext();
302  ++level;
303  }
304 
305  pmodel = sofa::core::objectmodel::New<DerivedModel>();
306  pmodel->setName("BVLevel", level);
307  root->addSlave(pmodel);
308  pmodel->setMoving(isMoving());
309  pmodel->setSimulated(isSimulated());
310  pmodel->proximity.setParent(&proximity);
311 
312  pmodel->group.beginEdit()->insert(group.getValue().begin(), group.getValue().end());
313  pmodel->group.endEdit();
314  pmodel->f_listening.setParent(&f_listening);
315  pmodel->f_printLog.setParent(&f_printLog);
316 
317  setPrevious(pmodel);
318  }
319  return pmodel.get();
320  }
