310 template <class T>
311 bool Data<T>::read(const std::string& s)
312 {
313  if (s.empty())
314  {
315  bool resized = getValueTypeInfo()->setSize( BaseData::beginEditVoidPtr(), 0 );
316  BaseData::endEditVoidPtr();
317  return resized;
318  }
319  std::istringstream istr( s.c_str() );
320 
321  // capture std::cerr output (if any)
322  std::stringstream cerrbuffer;
323  std::streambuf* old = std::cerr.rdbuf(cerrbuffer.rdbuf());
324 
325  istr >> *beginEdit();
326  endEdit();
327 
328  // restore the previous cerr
329  std::cerr.rdbuf(old);
330 
331  if( istr.fail() )
332  {
333  // transcript the std::cerr buffer into the Messaging system
334  msg_warning(this->getName()) << cerrbuffer.str();
335 
336  return false;
337  }
338 
339  return true;
340 }
