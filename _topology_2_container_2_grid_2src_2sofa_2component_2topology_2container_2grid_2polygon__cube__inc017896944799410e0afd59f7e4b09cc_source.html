34 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *\
35  * *
36  * POLYGON-CUBE INTERSECTION *
37  * by Don Hatch & Daniel Green *
38  * January 1994 *
39  * *
40  * CONTENTS: *
41  * polygon_intersects_cube() *
42  * fast_polygon_intersects_cube() *
43  * trivial_vertex_tests() *
44  * segment_intersects_cube() *
45  * polygon_contains_point_3d() *
46  * *
47  * *
48  * This module contains routines that test points, segments and polygons *
49  * for intersections with the unit cube defined as the axially aligned *
50  * cube of edge length 1 centered at the origin. Polygons may be convex, *
51  * concave or self-intersecting. Also contained is a routine that tests *
52  * whether a point is within a polygon. All routines are intended to be *
53  * fast and robust. Note that the cube and polygons are defined to include *
54  * their boundaries. *
55  * *
56  * The fast_polygon_intersects_cube routine is meant to replace the *
57  * triangle-cube intersection routine in Graphics Gems III by Douglas *
58  * Voorhies. While that original algorithm is still sound, it is *
59  * specialized for triangles and the implementation contained several *
60  * bugs and inefficiencies. The trivial_vertex_tests routine defined here *
61  * is almost an exact copy of the trivial point-plane tests from the *
62  * beginning of Voorhies' algorithm but broken out into a separate routine *
63  * which is called by fast_polygon_intersects_cube. The segment-cube and *
64  * polygon-cube intersection algorithms have been completely rewritten. *
65  * *
66  * Notice that trivial_vertex_tests can be used to quickly test an entire *
67  * set of vertices for trivial reject or accept. This can be useful for *
68  * testing polyhedra or entire polygon meshes. When used to test *
69  * polyhedra, remember that these routines only test points, edges and *
70  * surfaces, not volumes. If no such intersection is reported, the caller *
71  * should be aware that the volume of the polyhedra could still contain *
72  * the entire unit box which would then need to be checked for with an *
73  * additional point-within-polyhedron test. The origin would be a natural *
74  * point to check in such a test. *
75  * *
76 \* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
77 
78 /*
79  * POLYGON INTERSECTS CUBE
80  *
81  * Tells how the given polygon intersects the cube of edge length 1 centered
82  * at the origin.
83  * If any vertex or edge of the polygon intersects the cube,
84  * a value of 1 will be returned.
85  * Otherwise the value returned will be the multiplicity of containment
86  * of the cross-section of the cube in the polygon; this may
87  * be interpreted as a boolean value in any of the standard
88  * ways; e.g. the even-odd rule (it's inside the polygon iff the
89  * result is odd) or the winding rule (it's inside the polygon iff
90  * the result is nonzero).
91  *
92  * The "polynormal" argument is a vector perpendicular to the polygon. It
93  * need not be of unit length. It is suggested that Newell's method be used
94  * to calculate polygon normals (See Graphics Gems III). Zero-lengthed normals
95  * are quite acceptable for degenerate polygons but are not acceptable
96  * otherwise. In particular, beware of zero-length normals which Newell's
97  * method can return for certain self-intersecting polygons (for example
98  * a bow-tie quadrilateral).
99  *
100  * The already_know_verts_are_outside_cube flag is unused by this routine
101  * but may be useful for alternate implementations.
102  *
103  * The already_know_edges_are_outside_cube flag is useful when testing polygon
104  * meshes with shared edges in order to not test the same edge more than once.
105  *
106  * Note: usually users of this module would not want to call this routine
107  * directly unless they have previously tested the vertices with the trivial
108  * vertex test below. Normally one would call the fast_polygon_intersects_cube
109  * utility instead which combines both of these tests.
110  */
111 extern SOFA_COMPONENT_TOPOLOGY_CONTAINER_GRID_API int
112 polygon_intersects_cube(int nverts, const float verts[/* nverts */][3],
113  const float polynormal[3],
114  int already_know_verts_are_outside_cube,
115  int already_know_edges_are_outside_cube);

121  * This is a version of the same polygon-cube intersection that first calls

134 /*
135  * TRIVIAL VERTEX TESTS

137  * Returns 1 if any of the vertices are inside the cube of edge length 1

156 /*
157  * POLYGON CONTAINS POINT 3D
158  *
159  * Tells whether a given polygon with nonzero area contains a point which is
160  * assumed to lie in the plane of the polygon.
161  * Actually returns the multiplicity of containment. This will always be 1
162  * or 0 for non-self-intersecting planar polygons with the normal in the
163  * standard direction (towards the eye when looking at the polygon so that
164  * it's CCW).
165  */
166 extern SOFA_COMPONENT_TOPOLOGY_CONTAINER_GRID_API int
167 polygon_contains_point_3d(int nverts, const float verts[/* nverts */][3],
168  const float polynormal[3],
169  float point[3]);

171 /*
172  * Calculate a vector perpendicular to a planar polygon.
173  * If the polygon is non-planar, a "best fit" plane will be used.
174  * The polygon may be concave or even self-intersecting,
175  * but it should have nonzero area or the result will be a zero vector
176  * (e.g. the "bowtie" quad).
177  * The length of vector will be twice the area of the polygon.
178  * NOTE: This algorithm gives the same answer as Newell's method
179  * (see Graphics Gems III) but is slightly more efficient than Newell's
180  * for triangles and quads (slightly less efficient for higher polygons).
181  */
182 SOFA_COMPONENT_TOPOLOGY_CONTAINER_GRID_API float *
183 get_polygon_normal(float normal[3],
184  int nverts, const float verts[/* nverts */][3]);
