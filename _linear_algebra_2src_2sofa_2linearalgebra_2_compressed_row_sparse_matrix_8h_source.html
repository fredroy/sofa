119  static bool sortedFind(const VecIndex& v, Range in, Index val, Index& result)
120  {
121  if (in.empty()) return false;
122  Index candidate = (result >= in.begin() && result < in.end()) ? result : ((in.begin() + in.end()) >> 1);
123  for(;;)
124  {
125  Index i = v[candidate];
126  if (i == val) { result = candidate; return true; }
127  if (i < val) in.setBegin(candidate+1);
128  else in.setEnd(candidate);
129  if (in.empty()) break;
130  candidate = (in.begin() + in.end()) >> 1;
131  }
132  return false;
133  }

223  void compress() override
224  {
225  if (compressed && btemp.empty()) return;
226  if (!btemp.empty())
227  {
228  dmsg_info_when(COMPRESSEDROWSPARSEMATRIX_VERBOSE)
229  << "(" << rowSize() << "," << colSize() << "): sort " << btemp.size() << " temp blocks." ;
230  std::sort(btemp.begin(),btemp.end());
231  dmsg_info_when(COMPRESSEDROWSPARSEMATRIX_VERBOSE)
232  << "(" << rowSize() << "," << colSize() << "): blocks sorted." ;
233  }
234  oldRowIndex.swap(rowIndex);
235  oldRowBegin.swap(rowBegin);
236  oldColsIndex.swap(colsIndex);
237  oldColsValue.swap(colsValue);
238  rowIndex.clear();
239  rowBegin.clear();
240  colsIndex.clear();
241  colsValue.clear();
242  rowIndex.reserve(oldRowIndex.empty() ? nBlockRow : oldRowIndex.size());
243  rowBegin.reserve((oldRowIndex.empty() ? nBlockRow : oldRowIndex.size())+1);
244  colsIndex.reserve(oldColsIndex.size() + btemp.size());
245  colsValue.reserve(oldColsIndex.size() + btemp.size());
246  const Index oldNRow = (Index)oldRowIndex.size();
247  const Index EndRow = nBlockRow;
248  const Index EndCol = nBlockCol;
249  //const Index EndVal = oldColsIndex.size();
250  Index inRowId = 0;
251  Index inRowIndex = (inRowId < oldNRow ) ? oldRowIndex[inRowId] : EndRow;
252  typename VecIndexedBlock::const_iterator itbtemp = btemp.begin(), endbtemp = btemp.end();
253  Index bRowIndex = (itbtemp != endbtemp) ? itbtemp->l : EndRow;
254  Index outValId = 0;
255  while (inRowIndex < EndRow || bRowIndex < EndRow)
256  {
257  dmsg_info_when(COMPRESSEDROWSPARSEMATRIX_VERBOSE)
258  << "(" << rowSize() << "," << colSize() << "): inRowIndex = " << inRowIndex << " , bRowIndex = " << bRowIndex << "" ;
259  if (inRowIndex < bRowIndex)
260  {
261  // this row contains values only from old*
262  rowIndex.push_back(inRowIndex);
263  rowBegin.push_back(outValId);
264  Range inRow( oldRowBegin[inRowId], oldRowBegin[inRowId+1] );
265  while (!inRow.empty())
266  {
267  if (!traits::empty(oldColsValue[inRow.begin()]))
268  {
269  colsIndex.push_back(oldColsIndex[inRow.begin()]);
270  colsValue.push_back(oldColsValue[inRow.begin()]);
271  ++outValId;
272  }
273  ++inRow;
274  }
275  ++inRowId;
276  inRowIndex = (inRowId < oldNRow ) ? oldRowIndex[inRowId] : EndRow;
277  }
278  else if (inRowIndex > bRowIndex)
279  {
280  // this row contains values only from btemp
281  rowIndex.push_back(bRowIndex);
282  rowBegin.push_back(outValId);
283  while (itbtemp != endbtemp && itbtemp->l == bRowIndex)
284  {
285  Index bColIndex = itbtemp->c;
286  colsIndex.push_back(bColIndex);
287  colsValue.push_back(itbtemp->value);
288  ++itbtemp;
289  Block& value = colsValue.back();
290  while (itbtemp != endbtemp && itbtemp->c == bColIndex && itbtemp->l == bRowIndex)
291  {
292  value += itbtemp->value;
293  ++itbtemp;
294  }
295  ++outValId;
296  }
297  bRowIndex = (itbtemp != endbtemp) ? itbtemp->l : EndRow;
298  }
299  else
300  {
301  // this row mixes values from old* and btemp
302  rowIndex.push_back(inRowIndex);
303  rowBegin.push_back(outValId);
304  Range inRow( oldRowBegin[inRowId], oldRowBegin[inRowId+1] );
305  Index inColIndex = (!inRow.empty()) ? oldColsIndex[inRow.begin()] : EndCol;
306  Index bColIndex = (itbtemp != endbtemp && itbtemp->l == inRowIndex) ? itbtemp->c : EndCol;
307  while (inColIndex < EndCol || bColIndex < EndCol)
308  {
309  if (inColIndex < bColIndex)
310  {
311  if (!traits::empty(oldColsValue[inRow.begin()]))
312  {
313  colsIndex.push_back(inColIndex);
314  colsValue.push_back(oldColsValue[inRow.begin()]);
315  ++outValId;
316  }
317  ++inRow;
318  inColIndex = (!inRow.empty()) ? oldColsIndex[inRow.begin()] : EndCol;
319  }
320  else if (inColIndex > bColIndex)
321  {
322  colsIndex.push_back(bColIndex);
323  colsValue.push_back(itbtemp->value);
324  ++itbtemp;
325  Block& value = colsValue.back();
326  while (itbtemp != endbtemp && itbtemp->c == bColIndex && itbtemp->l == bRowIndex)
327  {
328  value += itbtemp->value;
329  ++itbtemp;
330  }
331  bColIndex = (itbtemp != endbtemp && itbtemp->l == bRowIndex) ? itbtemp->c : EndCol;
332  ++outValId;
333  }
334  else
335  {
336  colsIndex.push_back(inColIndex);
337  colsValue.push_back(oldColsValue[inRow.begin()]);
338  ++inRow;
339  inColIndex = (!inRow.empty()) ? oldColsIndex[inRow.begin()] : EndCol;
340  Block& value = colsValue.back();
341  while (itbtemp != endbtemp && itbtemp->c == bColIndex && itbtemp->l == bRowIndex)
342  {
343  value += itbtemp->value;
344  ++itbtemp;
345  }
346  bColIndex = (itbtemp != endbtemp && itbtemp->l == bRowIndex) ? itbtemp->c : EndCol;
347  ++outValId;
348  }
349  }
350  ++inRowId;
351  inRowIndex = (inRowId < oldNRow ) ? oldRowIndex[inRowId] : EndRow;
352  bRowIndex = (itbtemp != endbtemp) ? itbtemp->l : EndRow;
353  }
354  }
355  rowBegin.push_back(outValId);
356  btemp.clear();
357  compressed = true;
358  }

402  void fullDiagonal()
403  {
404  compress();
405  Index ndiag = 0;
406  for (Index r=0; r<(decltype(r))rowIndex.size(); ++r)
407  {
408  Index i = rowIndex[r];
409  Index b = rowBegin[r];
410  Index e = rowBegin[r+1];
411  Index t = b;
412  while (b < e && (decltype(i))colsIndex[t] != i)
413  {
414  if ((decltype(i))colsIndex[t] < i)
415  b = t+1;
416  else
417  e = t;
418  t = (b+e)>>1;
419  }
420  if (b<e) ++ndiag;
421  }
422  if (ndiag == nRow) return;
423 
424  oldRowIndex.swap(rowIndex);
425  oldRowBegin.swap(rowBegin);
426  oldColsIndex.swap(colsIndex);
427  oldColsValue.swap(colsValue);
428  rowIndex.resize(nRow);
429  rowBegin.resize(nRow+1);
430  colsIndex.resize(oldColsIndex.size()+nRow-ndiag);
431  colsValue.resize(oldColsValue.size()+nRow-ndiag);
432  Index nv = 0;
433  for (Index i=0; i<nRow; ++i) rowIndex[i] = i;
434  Index j = 0;
435  for (Index i=0; i<(decltype(i))oldRowIndex.size(); ++i)
436  {
437  for (; j<(decltype(j))oldRowIndex[i]; ++j)
438  {
439  rowBegin[j] = nv;
440  colsIndex[nv] = j;
441  traits::clear(colsValue[nv]);
442  ++nv;
443  }
444  rowBegin[j] = nv;
445  Index b = oldRowBegin[i];
446  Index e = oldRowBegin[i+1];
447  for (; b<e && (decltype(j))oldColsIndex[b] < j; ++b)
448  {
449  colsIndex[nv] = oldColsIndex[b];
450  colsValue[nv] = oldColsValue[b];
451  ++nv;
452  }
453  if (b>=e || (decltype(j))oldColsIndex[b] > j)
454  {
455  colsIndex[nv] = j;
456  traits::clear(colsValue[nv]);
457  ++nv;
458  }
459  for (; b<e; ++b)
460  {
461  colsIndex[nv] = oldColsIndex[b];
462  colsValue[nv] = oldColsValue[b];
463  ++nv;
464  }
465  ++j;
466  }
467  for (; j<nRow; ++j)
468  {
469  rowBegin[j] = nv;
470  colsIndex[nv] = j;
471  traits::clear(colsValue[nv]);
472  ++nv;
473  }
474  rowBegin[j] = nv;

492  // filtering-out part of a matrix
493  typedef bool filter_fn (Index i , Index j , Block& val, const Block& ref );
494  static bool nonzeros(Index /*i*/, Index /*j*/, Block& val, const Block& /*ref*/) { return (!traits::empty(val)); }
495  static bool nonsmall(Index /*i*/, Index /*j*/, Block& val, const Block& ref )
496  {
497  for (Index bi = 0; bi < NL; ++bi)
498  for (Index bj = 0; bj < NC; ++bj)
499  if (helper::rabs(traits::v(val, bi, bj)) >= traits::v(ref, bi, bj)) return true;
500  return false;
501  }
502  static bool upper (Index i , Index j , Block& val, const Block& /*ref*/)
503  {
504  if (NL>1 && i*NL == j*NC)
505  {
506  for (Index bi = 1; bi < NL; ++bi)
507  for (Index bj = 0; bj < bi; ++bj)
508  traits::v(val, bi, bj) = 0;
509  }
510  return i*NL <= j*NC;
511  }
512  static bool lower (Index i , Index j , Block& val, const Block& /*ref*/)
513  {
514  if (NL>1 && i*NL == j*NC)
515  {
516  for (Index bi = 0; bi < NL-1; ++bi)
517  for (Index bj = bi+1; bj < NC; ++bj)
518  traits::v(val, bi, bj) = 0;
519  }
520  return i*NL >= j*NC;
521  }
522  static bool upper_nonzeros(Index i , Index j , Block& val, const Block& ref ) { return upper(i,j,val,ref) && nonzeros(i,j,val,ref); }
523  static bool lower_nonzeros(Index i , Index j , Block& val, const Block& ref ) { return lower(i,j,val,ref) && nonzeros(i,j,val,ref); }
524  static bool upper_nonsmall(Index i , Index j , Block& val, const Block& ref ) { return upper(i,j,val,ref) && nonsmall(i,j,val,ref); }
525  static bool lower_nonsmall(Index i , Index j , Block& val, const Block& ref ) { return lower(i,j,val,ref) && nonsmall(i,j,val,ref); }

527  template<class TMatrix>
528  void filterValues(TMatrix& M, filter_fn* filter = &nonzeros, const Block& ref = Block())
529  {
530  M.compress();
531  nRow = M.rowSize();
532  nCol = M.colSize();
533  nBlockRow = M.rowBSize();
534  nBlockCol = M.colBSize();
535  rowIndex.clear();
536  rowBegin.clear();
537  colsIndex.clear();
538  colsValue.clear();
539  compressed = true;
540  btemp.clear();
541  rowIndex.reserve(M.rowIndex.size());
542  rowBegin.reserve(M.rowBegin.size());
543  colsIndex.reserve(M.colsIndex.size());
544  colsValue.reserve(M.colsValue.size());
545 
546  Index vid = 0;
547  for (Index rowId = 0; rowId < (Index)M.rowIndex.size(); ++rowId)
548  {
549  Index i = M.rowIndex[rowId];
550  rowIndex.push_back(i);
551  rowBegin.push_back(vid);
552  Range rowRange(M.rowBegin[rowId], M.rowBegin[rowId+1]);
553  for (Index xj = rowRange.begin(); xj < rowRange.end(); ++xj)
554  {
555  Index j = M.colsIndex[xj];
556  Block& b = M.colsValue[xj];
557  if ((*filter)(i,j,b,ref))
558  {
559  colsIndex.push_back(j);
560  colsValue.push_back(b);
561  ++vid;
562  }
563  }
564  if ((decltype(vid))rowBegin.back() == vid) // row was empty
565  {
566  rowIndex.pop_back();
567  rowBegin.pop_back();
568  }
569  }
570  rowBegin.push_back(vid); // end of last row
571  }
