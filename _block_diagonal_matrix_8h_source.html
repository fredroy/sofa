203  template<class Real2>
204  void mul(FullVector<Real2>& res, const FullVector<Real2>& v) const
205  {
206  res.resize(cSize);
207  Index nblocs = cSize;
208  Index szlast = 0;
209  traits::split_row_index(nblocs, szlast);
210  for (sofa::Index b=0; b<(sofa::Size) nblocs; b++)
211  {
212  sofa::Index i = b*LC;
213  for (sofa::Index bj=0; bj<Index(LC); bj++)
214  {
215  Real2 r = 0;
216  for (sofa::Index bi=0; bi<Index(LC); bi++)
217  {
218  r += (Real2)(traits::v(data[b],bi,bj) * v[i+bi]);
219  }
220  res[i+bj] = r;
221  }
222  }
223  if (szlast)
224  {
225  sofa::Size b = nblocs;
226  sofa::Index i = b*LC;
227  for (sofa::Index bj=0; bj<(sofa::Size) szlast; bj++)
228  {
229  Real2 r = 0;
230  for (sofa::Index bi=0; bi<(sofa::Size) szlast; bi++)
231  {
232  r += (Real2)(traits::v(data[b],bi,bj) * v[i+bi]);
233  }
234  res[i+bj] = r;
235  }
236  }
237  }
