73  template<Size N, class T>
74  struct Impl
75  {
76  static void set( Coord& c, T x, T y, T z )
77  {
78  if constexpr (N > 2)
79  {
80  c[0] = Real(x);
81  c[1] = Real(y);
82  c[2] = Real(z);
83  }
84  if constexpr (N == 2)
85  {
86  SOFA_UNUSED(z);
87  c[0] = Real(x);
88  c[1] = Real(y);
89  }
90  if constexpr (N == 1)
91  {
92  SOFA_UNUSED(y);
93  SOFA_UNUSED(z);
94  c[0] = Real(x);
95  }
96  }
97 
98  static void get( T& x, T& y, T& z, const Coord& c )
99  {
100  if constexpr(N > 2)
101  {
102  x = T(c[0]);
103  y = T(c[1]);
104  z = T(c[2]);
105  }
106  if constexpr (N == 2)
107  {
108  SOFA_UNUSED(z);
109  x = T(c[0]);
110  y = T(c[1]);
111  z = T(0);
112  }
113  if constexpr (N == 1)
114  {
115  SOFA_UNUSED(y);
116  SOFA_UNUSED(z);
117  x = T(c[0]);
118  y = T(0);
119  z = T(0);
120  }
121  }
122 
123  static void add( Coord& c, T x, T y, T z )
124  {
125  if constexpr (N > 2)
126  {
127  c[0] += Real(x);
128  c[1] += Real(y);
129  c[2] += Real(z);
130  }
131  if constexpr (N == 2)
132  {
133  SOFA_UNUSED(z);
134  c[0] += Real(x);
135  c[1] += Real(y);
136  }
137  if constexpr (N == 1)
138  {
139  SOFA_UNUSED(y);
140  SOFA_UNUSED(z);
141  c[0] += Real(x);
142  }
143  }
144  };

185  static Coord interpolate(const type::vector< Coord > &ancestors, const type::vector< Real > &coefs)
186  {
187  assert(ancestors.size() == coefs.size());
188 
189  Coord c;
190 
191  for (std::size_t i = 0; i < ancestors.size(); i++)
192  {
193  c += ancestors[i] * coefs[i];
194  }
