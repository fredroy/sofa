573 public:
595  QFont scaledFont(const QFont &font) const {
596  if (tileRegion_ == NULL)
597  return font;
598  else {
599  QFont f(font);
600  if (f.pixelSize() == -1)
601  f.setPointSizeF(f.pointSizeF() * tileRegion_->textScale);
602  else
603  f.setPixelSize(int(f.pixelSize() * tileRegion_->textScale));
604  return f;
605  }
606  }
