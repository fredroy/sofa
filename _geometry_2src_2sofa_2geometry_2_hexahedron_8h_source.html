66  template<typename Node,
67  typename T = std::decay_t<decltype(*std::begin(std::declval<Node>()))>,
68  typename = std::enable_if_t<std::is_scalar_v<T>>
69  >
70  [[nodiscard]]
71  static constexpr auto center(const Node& n0, const Node& n1, const Node& n2, const Node& n3,
72  const Node& n4, const Node& n5, const Node& n6, const Node& n7)
73  {
74  constexpr auto dimensions = sizeof(Node) / sizeof(T);
75  auto centerRes = n0;
76  for (size_t i = 0; i < dimensions; i++)
77  {
78  centerRes[i] += n1[i] + n2[i] + n3[i] + n4[i] + n5[i] + n6[i] + n7[i];
79  centerRes[i] /= static_cast<T>(NumberOfNodes);
80  }
81 
82  return centerRes;
83  }

95  typename T = std::decay_t<decltype(*std::begin(std::declval<Node>()))>,
96  typename = std::enable_if_t<std::is_scalar_v<T>>
97  >
98  [[nodiscard]]
99  static constexpr auto barycentricCoefficients(const Node& n0, const Node& n1, const Node& n2, const Node& n3,
100  const Node& n4, const Node& n5, const Node& n6, const Node& n7,
101  const Node& pos)
102  {
103  SOFA_UNUSED(n2);
104  SOFA_UNUSED(n5);
105  SOFA_UNUSED(n6);
106  SOFA_UNUSED(n7);
107  constexpr sofa::Size dimensions = sizeof(Node) / sizeof(T);
108  constexpr auto max_spatial_dimensions = std::min(3u, dimensions);
109 
110  sofa::type::Vec<3, T> origin, p1, p3, p4, pnt;
111 
112  for (unsigned int w = 0; w < max_spatial_dimensions; ++w)
113  {
114  origin[w] = n0[w];
115  p1[w] = n1[w];
116  p3[w] = n3[w];
117  p4[w] = n4[w];
118  pnt[w] = pos[w];
119  }
120 
121  sofa::type::Mat<3,3,T> m, mt, base;
122  m[0] = p1 - origin;
123  m[1] = p3 - origin;
124  m[2] = p4 - origin;
125  mt.transpose(m);
126  const bool canInvert = base.invert(mt);
127  assert(canInvert);
128  SOFA_UNUSED(canInvert);
129  const auto tmpResult = base * (pnt - origin);
130 
131  return tmpResult;
132  }

143  template<typename Node,
144  typename T = std::decay_t<decltype(*std::begin(std::declval<Node>()))>,
145  typename = std::enable_if_t<std::is_scalar_v<T>>
146  >
147  [[nodiscard]]
148  static constexpr auto squaredDistanceTo(const Node& n0, const Node& n1, const Node& n2, const Node& n3,
149  const Node& n4, const Node& n5, const Node& n6, const Node& n7,
150  const Node& pos)
151  {
152  const auto& v = barycentricCoefficients(n0,n1,n2,n3,n4,n5,n6,n7, pos);
153 
154  T d = std::max(std::max(-v[0], -v[1]), std::max(std::max(-v[2], v[0] - 1), std::max(v[1] - static_cast<T>(1), v[2] - static_cast<T>(1))));
155 
156  if (d > 0)
157  d = (pos - center(n0, n1, n2, n3, n4, n5, n6, n7)).norm2();
158  else
159  d = static_cast<T>(0);
160 
161  return d;
162  }

173  template<typename Node,
174  typename T = std::decay_t<decltype(*std::begin(std::declval<Node>()))>,
175  typename = std::enable_if_t<std::is_scalar_v<T>>

180  {

182  const auto fy = baryC[1];
183  const auto fz = baryC[2];
184 
185  const auto pos = n0 * ((1 - fx) * (1 - fy) * (1 - fz))
186  + n1 * ((fx) * (1 - fy) * (1 - fz))
187  + n3 * ((1 - fx) * (fy) * (1 - fz))
188  + n2 * ((fx) * (fy) * (1 - fz))
189  + n4 * ((1 - fx) * (1 - fy) * (fz))
190  + n5 * ((fx) * (1 - fy) * (fz))
191  + n7 * ((1 - fx) * (fy) * (fz))
192  + n6 * ((fx) * (fy) * (fz));
193
