117  friend std::istream& operator >> ( std::istream& in, JointSpring<DataTypes>& s )
118  {
119  //default joint is a free rotation joint --> translation is bloqued, rotation is free
120  s.freeMovements = sofa::type::Vec<6,bool>(false, false, false, true, true, true);
121  s.initTrans = Vector(0,0,0);
122  s.initRot = type::Quat<SReal>(0,0,0,1);
123  s.blocStiffnessRot = 0.0;
124  //by default no angle limitation is set (bi values for initialisation)
125  s.limitAngles = sofa::type::Vec<6,Real>(-100000., 100000., -100000., 100000., -100000., 100000.);
126  bool initTransFound=false;
127 
128  std::string str;
129  in>>str;
130  if(str == "BEGIN_SPRING")
131  {
132  in>>s.m1>>s.m2; //read references
133  in>>str;
134  while(str != "END_SPRING")
135  {
136  if(str == "FREE_AXIS")
137  in>>s.freeMovements;
138  else if(str == "KS_T")
139  in>>s.softStiffnessTrans>>s.hardStiffnessTrans;
140  else if(str == "KS_R")
141  in>>s.softStiffnessRot>>s.hardStiffnessRot;
142  else if(str == "KS_B")
143  in>>s.blocStiffnessRot;
144  else if(str == "KD")
145  in>>s.kd;
146  else if(str == "R_LIM_X")
147  in>>s.limitAngles[0]>>s.limitAngles[1];
148  else if(str == "R_LIM_Y")
149  in>>s.limitAngles[2]>>s.limitAngles[3];
150  else if(str == "R_LIM_Z")
151  in>>s.limitAngles[4]>>s.limitAngles[5];
152  else if(str == "REST_T")
153  {
154  in>>s.initTrans;
155  initTransFound=true;
156  }
157  else if(str == "REST_R")
158  {
159  in>>s.initRot;
160  }
161  else
162  {
163  msg_error("JointSprintForceField")<<"Unknown Attribute while parsing '"<<str<<"'" ;
164  return in;
165  }
166 
167  in>>str;
168  }
169  }
170 
171  s.needToInitializeTrans = initTransFound;
172  s.needToInitializeRot = initTransFound;
173 
174  if(s.blocStiffnessRot == 0.0)
175  s.blocStiffnessRot = s.hardStiffnessRot/100;
176 
177  for (unsigned int i=0; i<3; i++)
178  {
179  if(s.limitAngles[2*i]==s.limitAngles[2*i+1])
180  s.freeMovements[3+i] = false;
181  }
182 
183  return in;
184  }

197  if (s.kd != 0.0)
198  out<<"KD "<<s.kd<<" ";
199  if (s.limitAngles[0]!=-100000 || s.limitAngles[1] != 100000)
200  out<<"R_LIM_X "<<s.limitAngles[0]<<" "<<s.limitAngles[1]<<" ";

205  if (s.initTrans!= Vector(0,0,0))
