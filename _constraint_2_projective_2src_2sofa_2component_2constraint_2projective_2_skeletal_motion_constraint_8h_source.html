150  inline friend std::ostream& operator << (std::ostream& out, const SkeletonJoint& skeletonJoint)
151  {
152  out << "Parent" << " " << skeletonJoint.mParentIndex << " ";
153  out << "Channels" << " " << skeletonJoint.mChannels.size() << " " << skeletonJoint.mChannels << " ";
154  out << "Times" << " " << skeletonJoint.mTimes.size() << " " << skeletonJoint.mTimes << " ";
155  out << "PreviousMotion" << " " << skeletonJoint.mPreviousMotion << " ";
156  out << "PreviousMotionTime" << " " << skeletonJoint.mPreviousMotionTime << " ";
157  out << "NextMotion" << " " << skeletonJoint.mNextMotion << " ";
158  out << "NextMotionTime" << " " << skeletonJoint.mNextMotionTime << " ";
159  out << "LocalRigid" << " " << skeletonJoint.mLocalRigid;
160 
161  return out;
162  }
163 
164  inline friend std::istream& operator >> (std::istream& in, SkeletonJoint& skeletonJoint)
165  {
166  std::string tmp;
167 
168  in >> tmp >> skeletonJoint.mParentIndex;
169 
170  size_t numChannel;
171  in >> tmp >> numChannel;
172  skeletonJoint.mChannels.resize(numChannel);
173  Coord channel;
174  for(Size i = 0; i < numChannel; ++i)
175  {
176  in >> channel;
177  skeletonJoint.mChannels[i] = channel;
178  }
179 
180  size_t numTime;
181  in >> tmp >> numTime;
182  skeletonJoint.mTimes.resize(numTime);
183  double time;
184  for(Size i = 0; i < numTime; ++i)
185  {
186  in >> time;
187  skeletonJoint.mTimes[i] = time;
188  }
189 
190  in >> tmp >> skeletonJoint.mPreviousMotion;
191  in >> tmp >> skeletonJoint.mNextMotion;
192  in >> tmp >> skeletonJoint.mLocalRigid;
193 
194  return in;
195  }
