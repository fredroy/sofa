148  const type::vector<const In2DataVecDeriv*>& dataVecIn2Vel)
149  {
150  //Not optimized at all...
151  type::vector<OutVecDeriv*> vecOutVel;
152  for(unsigned int i=0; i<dataVecOutVel.size(); i++)
153  vecOutVel.push_back(dataVecOutVel[i]->beginEdit());
154 
155  type::vector<const In1VecDeriv*> vecIn1Vel;
156  for(unsigned int i=0; i<dataVecIn1Vel.size(); i++)
157  vecIn1Vel.push_back(&dataVecIn1Vel[i]->getValue());
158  type::vector<const In2VecDeriv*> vecIn2Vel;
159  for(unsigned int i=0; i<dataVecIn2Vel.size(); i++)
160  vecIn2Vel.push_back(&dataVecIn2Vel[i]->getValue());
161  this->applyJ(vecOutVel, vecIn1Vel, vecIn2Vel);
162 
163  //Really Not optimized at all...
164  for(unsigned int i=0; i<dataVecOutVel.size(); i++)
165  dataVecOutVel[i]->endEdit();
166  }

223  template<class T>
224  static bool canCreate(T*& obj, core::objectmodel::BaseContext* context, core::objectmodel::BaseObjectDescription* arg)
225  {
226  std::string input1 = arg->getAttribute("input1","");
227  std::string input2 = arg->getAttribute("input2","");
228  std::string output = arg->getAttribute("output","");
229  if (!input1.empty() && !PathResolver::CheckPaths(context, LinkFromModels1::DestType::GetClass(), input1))
230  return false;
231  if (!input2.empty() && !PathResolver::CheckPaths(context, LinkFromModels2::DestType::GetClass(), input2))
232  return false;
233  if (output.empty() || !PathResolver::CheckPaths(context, LinkToModels::DestType::GetClass(), output))
234  return false;
235 
236  return BaseMapping::canCreate(obj, context, arg);
237  }
