232  void parse(sofa::core::objectmodel::BaseObjectDescription* arg) override
233  {
234  Inherited::parse(arg);
235 
236  if (arg->getAttribute("template"))
237  {
238  auto splitTemplates = sofa::helper::split(std::string(arg->getAttribute("template")), ',');
239  if (splitTemplates.size() > 1)
240  {
241  // check if the given 2nd template is the deprecated MassType one
242  if (splitTemplates[1] == "float" || splitTemplates[1] == "double" || splitTemplates[1].find("RigidMass") != std::string::npos)
243  {
244  msg_warning() << "MassType is not required anymore and the template is deprecated, please delete it from your scene." << msgendl
245  << "As your mass is templated on " << DataTypes::Name() << ", MassType has been defined as " << sofa::helper::NameDecoder::getTypeName<MassType>() << " .";
246  msg_warning() << "If you want to set the template, you must write now \"template='" << DataTypes::Name() << "'\" .";
247  }
248  }
