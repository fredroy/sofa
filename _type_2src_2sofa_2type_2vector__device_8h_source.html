274  void reserve(Size s, Size WARP_SIZE = MemoryManager::BSIZE)
275  {
276  s = ((s + WARP_SIZE - 1) / WARP_SIZE) * WARP_SIZE;
277 #ifdef SOFA_VECTOR_DEVICE_CUSTOM_SIZE
278  if (s > deviceReserveSize)
279  {
280  // we double the reserved size except if the requested size is bigger or if we would allocate more memory than the configured step size
281  if (s <= 2 * deviceReserveSize && 2 * deviceReserveSize <= s + SOFA_VECTOR_HOST_STEP_SIZE)
282  deviceReserveSize *= 2;
283  else
284  deviceReserveSize = s;
285  // always allocate multiples of WARP_SIZE values
286  deviceReserveSize = ((deviceReserveSize + WARP_SIZE - 1) / WARP_SIZE) * WARP_SIZE;
287  }
288 #endif
289  if (s <= allocSize) return;
290  DEBUG_OUT_V(SPACEP << "reserve " << vectorSize << "->" << s << " (alloc=" << allocSize << ")" << std
291  ::endl);
292  // we double the reserved size except if the requested size is bigger or if we would allocate more memory than the configured step size
293  if (s <= 2 * allocSize && 2 * allocSize <= s + SOFA_VECTOR_HOST_STEP_SIZE)
294  allocSize *= 2;
295  else
296  allocSize = s + SOFA_VECTOR_HOST_STEP_SIZE;
297  // always allocate multiples of WARP_SIZE values
298  allocSize = ((allocSize + WARP_SIZE - 1) / WARP_SIZE) * WARP_SIZE;
299 
300  if (MemoryManager::SUPPORT_GL_BUFFER && bufferObject)
301  {
302  DEBUG_OUT_V(SPACEN << "BUFFEROBJ " << std::endl);
303 
304  hostRead(); // make sure the host copy is valid
305  unregisterBuffer();
306 
307  MemoryManager::bufferAlloc(&bufferObject, allocSize * sizeof(T), false);
308 
309  if (vectorSize > 0) deviceIsValid = 0;
310  }
311 
312  T* prevHostPointer = hostPointer;
313  void* newHostPointer = nullptr;
314  DEBUG_OUT_V(SPACEN << "MemoryManager::hostAlloc " << allocSize << std::endl);
315  MemoryManager::hostAlloc(&newHostPointer, allocSize * sizeof(T));
316  hostPointer = (T*)newHostPointer;
317  if (vectorSize != 0 && hostIsValid) std::copy(prevHostPointer, prevHostPointer + vectorSize, hostPointer);
318  if (prevHostPointer != nullptr) MemoryManager::hostFree(prevHostPointer);
319  DEBUG_OUT_V(SPACEM << "reserve " << " (alloc=" << allocSize << ")" << std::endl);
320  }

323  void fastResize(Size s, Size WARP_SIZE = MemoryManager::BSIZE)
324  {
325  if (s == vectorSize) return;
326  DEBUG_OUT_V(SPACEP << "fastresize " << vectorSize << "->" << s << " (alloc=" << allocSize << ")" << std::endl);
327  reserve(s, WARP_SIZE);
328  vectorSize = s;
329  if (clearSize > vectorSize) clearSize = vectorSize;
330  for (int d = 0; d < MemoryManager::numDevices(); d++)
331  if (deviceVectorSize[d] > vectorSize)
332  deviceVectorSize[d] = vectorSize;
333  if (!vectorSize)
334  {
335  // special case when the vector is now empty -> host and device are valid
336  deviceIsValid = ALL_DEVICE_VALID;
337  hostIsValid = true;
338  }
339  DEBUG_OUT_V(SPACEM << "fastresize " << std::endl);
340  }

392  void resize(Size s, Size WARP_SIZE = MemoryManager::BSIZE)
393  {
394  reserve(s, WARP_SIZE);
395  if (s == vectorSize) return;
396  DEBUG_OUT_V(SPACEP << "resize " << vectorSize << "->" << s << " (alloc=" << allocSize << ")" << std::endl);
397  if (s > vectorSize)
398  {
399  if (datatypeinfo_manager::ZeroConstructor) // can use memset instead of constructors
400  {
401  if (hostIsValid)
402  {
403  DEBUG_OUT_V(SPACEN << "MemoryManager::memsetHost (new data) " << (s - vectorSize) << std::endl);
404  MemoryManager::memsetHost(hostPointer + vectorSize, 0, (s - vectorSize) * sizeof(T));
405  }
406  clearSize = s;
407 #ifndef SOFA_VECTOR_DEVICE_CUSTOM_SIZE
408  size_t deviceReserveSize = allocSize;
409 #endif
410  for (int d = 0; d < MemoryManager::numDevices(); d++)
411  {
412  if (isDeviceValid(d))
413  {
414  if (deviceAllocSize[d] >= deviceReserveSize)
415  {
416  MemoryManager::memsetDevice(d, MemoryManager::deviceOffset(devicePointer[d], deviceVectorSize[d]), 0, (s - deviceVectorSize[d]) * sizeof(T));
417  deviceVectorSize[d] = s;
418  }
419  // if the memory allocated on a device is not sufficient, we do not reallocate and memset the new data until it is requested on the device
420  // but the valid flag is kept set because we will be able to provide the data on the device without transfers
421  //else deviceIsValid &= ~(1<<d);
422  }
423  }
424  }
425  else // must use class constructors -> resize can only happen on the host
426  {
427  DEBUG_OUT_V(SPACEN << "ZEROCONST " << std::endl);
428  copyToHost();
429  DEBUG_OUT_V(SPACEN << "MemoryManager::memsetHost (new data) " << (s - vectorSize) << std::endl);
430  MemoryManager::memsetHost(hostPointer + vectorSize, 0, (s - vectorSize) * sizeof(T));
431  // Call the constructor for the new elements
432  for (Size i = vectorSize; i < s; i++) ::new (hostPointer + i) T;
433 
434  if (vectorSize == 0) // wait until the transfer is really necessary, as other modifications might follow
435  {
436  deviceIsValid = 0;
437  }
438  else
439  {
440  if (MemoryManager::SUPPORT_GL_BUFFER && bufferObject) mapBuffer();
441 
442  for (int d = 0; d < MemoryManager::numDevices(); d++)
443  {
444  if (isDeviceValid(d))
445  {
446  if (deviceVectorSize[d] == 0) // no data is currently on the device -> we simply invalidate it and the transfer will happen once the data is actually requested
447  deviceIsValid &= ~(1 << d);
448  else
449  {
450  allocate(d);
451  DEBUG_OUT_V(SPACEN << "MemoryManager::memcpyHostToDevice " << vectorSize << "->" << (s - vectorSize) << std::endl);
452  MemoryManager::memcpyHostToDevice(d, MemoryManager::deviceOffset(devicePointer[d], deviceVectorSize[d]), hostPointer + deviceVectorSize[d], (s - deviceVectorSize[d]) * sizeof(T));
453  deviceVectorSize[d] = s;
454  }
455  }
456  }
457  }
458  }
459  }
460  else if (s < vectorSize && !(datatypeinfo_manager::SimpleCopy)) // need to call destructors
461  {
462  DEBUG_OUT_V(SPACEN << "SIMPLECOPY " << std::endl);
463  copyToHost();
464  // Call the destructor for the deleted elements
465  for (Size i = s; i < vectorSize; i++)
466  {
467  hostPointer[i].~T();
468  }
469  }
470  vectorSize = s;
471  if (clearSize > vectorSize) clearSize = vectorSize;
472  for (int d = 0; d < MemoryManager::numDevices(); d++)
473  if (deviceVectorSize[d] > vectorSize)
474  deviceVectorSize[d] = vectorSize;
475 
476  if (!vectorSize) // special case when the vector is now empty -> host and device are valid
477  {
478  deviceIsValid = ALL_DEVICE_VALID;
479  hostIsValid = true;
480  }
481 
482  DEBUG_OUT_V(SPACEM << "resize " << std::endl);
483  }

708  {
709 #ifdef SOFA_VECTOR_DEVICE_CUSTOM_SIZE
710  size_t alloc = deviceReserveSize;
711 #else
712  size_t alloc = allocSize;
713 #endif
714  if (deviceAllocSize[d] < alloc)

717  device_pointer prevDevicePointer = devicePointer[d];
718 
719  MemoryManager::deviceAlloc(d, &devicePointer[d], alloc * sizeof(T));
720  deviceAllocSize[d] = alloc;
721  if (isDeviceValid(d))
722  {
723  if (deviceVectorSize[d] > 0)
724  {
725  DEBUG_OUT_V(SPACEN << "MemoryManager::memcpyDeviceToDevice(copy valid vector) " << 0 << "->" << deviceVectorSize[d] << std::endl);
726  MemoryManager::memcpyDeviceToDevice(d, devicePointer[d], prevDevicePointer, deviceVectorSize[d] * sizeof(T));
727  }
728  }
729  if (!MemoryManager::isNull(prevDevicePointer)) MemoryManager::deviceFree(d, prevDevicePointer);
730  deviceAllocSize[d] = alloc;
731  DEBUG_OUT_V(SPACEM << "allocate " << std::endl);
732  }

734  {

737  DEBUG_OUT_V(SPACEN << "MemoryManager::memsetDevice (clear new data) " << deviceVectorSize[d] << "->" << clearSize - deviceVectorSize[d] << std::endl);
738  MemoryManager::memsetDevice(d, MemoryManager::deviceOffset(devicePointer[d], deviceVectorSize[d]), 0, (clearSize - deviceVectorSize[d]) * sizeof(T));
739  }

741  }

753  {

765  }
766  hostIsValid = true;
767  break;
768  }

775  MemoryManager::memsetHost(hostPointer, 0, clearSize * sizeof(T));
