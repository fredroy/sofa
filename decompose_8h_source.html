76  template<Size spatial_dimension, Size material_dimension>
77  static void QRDecompositionGradient_dQ( const type::Mat<spatial_dimension,material_dimension,Real>&Q,
78  const type::Mat<material_dimension,material_dimension,Real>&invR,
79  const type::Mat<spatial_dimension,material_dimension,Real>& dM,
80  type::Mat<spatial_dimension,material_dimension,Real>& dQ )
81  {
82  // dQ = Q ( lower(QT*dM*R−1) − lower(QT*dM*R−1)^T )
83  // dR = ( upper(QT*dM*R−1) + lower(QT*dM*R−1)^T ) R
84  // lower -> strictly lower
85 
86  // tmp = QT*dM*R^−1
87  type::Mat<material_dimension,material_dimension,Real> tmp = Q.multTranspose(dM * invR);
88 
89  // L = lower(tmp) - (lower(tmp))^T
90  type::Mat<material_dimension,material_dimension,Real> L;
91 
92  for(Size i = 0; i < material_dimension; ++i)
93  {
94  for(Size j = 0; j < i; ++j) // strictly lower
95  L[i][j] = tmp[i][j];
96  for(Size j = i + 1; j < material_dimension; ++j) // strictly lower transposed
97  L[i][j] = -tmp[j][i];
98  }
99 
100  dQ = Q * L;
101  }
