92  inline friend std::istream& operator >> ( std::istream& in, GearSpring<DataTypes>& s )
93  {
94  //default Gear is a Gear around x
95  s.freeAxis = sofa::type::Vec<2,unsigned int>(0,0);
96 
97  std::string str;
98  in>>str;
99  if(str == "BEGIN_SPRING")
100  {
101  in>>s.p1>>s.m1>>s.p2>>s.m2; //read references
102  in>>str;
103  while(str != "END_SPRING")
104  {
105  if(str == "AXIS")
106  in>>s.freeAxis;
107  else if(str == "KS_T")
108  in>>s.hardStiffnessTrans;
109  else if(str == "KS_R")
110  in>>s.softStiffnessRot>>s.hardStiffnessRot;
111  else if(str == "KD")
112  in>>s.kd;
113  else if(str == "RATIO")
114  in>>s.Ratio;
115  else
116  {
117  msg_error("GearSpring")<<"parsing Spring : Unknown Attribute "<<str;
118  return in;
119  }
120 
121  in>>str;
122  }
123  }
124 
125  return in;
126  }

249  inline typename GearSpringForceField<DataTypes>::Real getAngleAroundAxis(Coord p1,Coord p2,unsigned int axis)
250  {
251  Mat M1, M2;
252  p1.writeRotationMatrix(M1);
253  p2.writeRotationMatrix(M2);
254  Vector u,v;
255  // compute angle as average angles between other axis
256  Real angle = 0.0;
257  unsigned int count = 0;
258  Vector W,w;
259  for(unsigned int i=0; i<u.size(); ++i) W[i]=M1[i][axis]; // ref axis
260  for(unsigned int j=0; j<u.size(); ++j) if(j!=axis)
261  {
262  for(unsigned int i=0; i<u.size(); ++i) {u[i]=M1[i][j]; v[i]=M2[i][j];}
263  count++; getVectorAngle(u,v,w); if(dot(w,W)<0) angle -= w.norm(); else angle += w.norm();
264  }
265  angle /= (Real)count;
266  return angle;
267  }
268 
269  inline void getVectorAngle(Vector u,Vector v,Vector &w)
270  {
271  w=cross(u,v);
272  Real nw=w.norm(),dt=dot(u,v);
273  if(nw) w*=atan2(nw,dt)/nw;
274  }
275 
276 
277  inline void getVectorAngle(Coord p1,Coord p2,unsigned int axis,Vector &w)
278  {
279  Mat M1, M2;
280  p1.writeRotationMatrix(M1);
281  p2.writeRotationMatrix(M2);
282  Vector u,v;
283  for(unsigned int i=0; i<u.size(); ++i) { u[i]=M1[i][axis]; v[i]=M2[i][axis]; }
284  getVectorAngle(u,v,w);
285  }
