34 template<class MatrixRow, class VecDeriv>
35 typename VecDeriv::Real SparseMatrixVecDerivMult(const MatrixRow& row, const VecDeriv& vec)
36 {
37  typename VecDeriv::Real r = 0;
38  for (typename MatrixRow::const_iterator it = row.begin(), itend = row.end(); it != itend; ++it)
39  r += it->second * vec[it->first];
40  return r;
41 }
