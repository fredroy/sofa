129  template<class T>
130  static bool canCreate ( T*& /*obj*/, core::objectmodel::BaseContext* context, core::objectmodel::BaseObjectDescription* arg )
131  {
132  if (arg->findObject(arg->getAttribute("hexaContainerPath", "../..")) == nullptr) {
133  msg_error(context) << "Cannot create " << T::GetClass()->className << " as the hexas container is missing.";
134  }
135  if (arg->findObject(arg->getAttribute("targetPath", "..")) == nullptr) {
136  msg_error(context) << "Cannot create " << T::GetClass()->className << " as the target point set is missing.";
137  }
138  if ( dynamic_cast<sofa::component::topology::container::dynamic::DynamicSparseGridTopologyContainer*> ( arg->findObject ( arg->getAttribute ( "hexaContainerPath","../.." ) ) ) == nullptr )
139  return false;
140  if ( dynamic_cast<core::behavior::MechanicalState<DataTypes>*> ( arg->findObject ( arg->getAttribute ( "targetPath",".." ) ) ) == nullptr )
141  return false;
142  return true;
143  }
148  template<class T>
149  static typename T::SPtr create(T*, core::objectmodel::BaseContext* context, core::objectmodel::BaseObjectDescription* arg )
150  {
151  typename T::SPtr obj = sofa::core::objectmodel::New<T>(
152  ( arg?dynamic_cast<sofa::component::topology::container::dynamic::DynamicSparseGridTopologyContainer*> ( arg->findObject ( arg->getAttribute ( "hexaContainerPath","../.." ) ) ) :nullptr ),
153  ( arg?dynamic_cast<core::behavior::MechanicalState<DataTypes>*> ( arg->findObject ( arg->getAttribute ( "targetPath",".." ) ) ) :nullptr ) );
154 
155  if ( context ) context->addObject ( obj );
156 
157  if ( arg )
158  {
159  if ( arg->getAttribute ( "hexaContainerPath" ) )
160  {
161  obj->hexaContainerPath.setValue ( arg->getAttribute ( "hexaContainerPath" ) );
162  arg->removeAttribute ( "hexaContainerPath" );
163  }
164  if ( arg->getAttribute ( "targetPath" ) )
165  {
166  obj->targetPath.setValue ( arg->getAttribute ( "targetPath" ) );
167  arg->removeAttribute ( "targetPath" );
168  }
169  obj->parse ( arg );
170  }
171 
172  return obj;
173  }
