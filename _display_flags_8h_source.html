78 inline tristate fusion_tristate(const tristate &lhs, const tristate &rhs)
79 {
80  if( lhs.state == rhs.state ) return lhs;
81  return tristate(tristate::neutral_value);
82 }
83 
84 inline tristate merge_tristate(const tristate& previous, const tristate& current)
85 {
86  if(current.state == tristate::neutral_value ) return previous;
87  return current;
88 }
89 inline tristate difference_tristate(const tristate& previous, const tristate& current)
90 {
91  if( current.state == tristate::neutral_value || current.state == previous.state )
92  return tristate(tristate::neutral_value);
93  return current;
