452  constexpr ValueType lNorm( int l ) const
453  {
454  if( l==2 ) return norm(); // euclidian norm
455  else if( l<0 ) // infinite norm
456  {
457  ValueType n=0;
458  for( Size i=0; i<N; i++ )
459  {
460  const ValueType a = rabs( this->elems[i] );
461  if( a>n ) n=a;
462  }
463  return n;
464  }
465  else if( l==1 ) // Manhattan norm
466  {
467  ValueType n=0;
468  for( Size i=0; i<N; i++ )
469  {
470  n += rabs( this->elems[i] );
471  }
472  return n;
473  }
474  else if( l==0 ) // counting not null
475  {
476  ValueType n=0;
477  for( Size i=0; i<N; i++ )
478  if( this->elems[i] ) n+=1;
479  return n;
480  }
481  else // generic implementation
482  {
483  ValueType n = 0;
484  for( Size i=0; i<N; i++ )
485  n += static_cast<ValueType>((pow( rabs( this->elems[i] ), l )));
486  return static_cast<ValueType>(pow( n, static_cast<ValueType>(1.0)/ static_cast<ValueType>(l) ));
487  }
488  }
