139  template<class T>
140  static typename T::SPtr create(T*, core::objectmodel::BaseContext* context, core::objectmodel::BaseObjectDescription* arg)
141  {
142  typename T::SPtr obj;
143 
144  if( context)
145  {
146  auto* _mstate = dynamic_cast<core::behavior::MechanicalState<TDataTypes>*>(context->getMechanicalState());
147  if (_mstate)
148  obj = sofa::core::objectmodel::New<T>(_mstate);
149  else
150  obj = sofa::core::objectmodel::New<T>();
151 
152  context->addObject(obj);
153  }
154 
155  if (arg) obj->parse(arg);
156 
157  return obj;
158  }
