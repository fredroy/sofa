59  typedef sofa::type::fixed_array<PointID, 4> HighOrderEdgePosition;

62  typedef sofa::type::fixed_array<PointID, 5> HighOrderTrianglePosition;
63  /* specify for each control point lying on a Quad : the control point index, the index of the quad,
64  the 2 integers specifying the degree of the element in the x and y directions, the 2 integers specifying the position within this quad (i.e. 12 for a cubic triangle ).. */
65  typedef sofa::type::fixed_array<PointID, 6> HighOrderQuadPosition;
66  /* specify for each control point lying on a tetrahedron : the control point index, the index of the tetrahedron,
67  the 3 integers specifying the position within this tetrahedron (i.e. 1111 for a quartic tetrahedron , 1211 within a quintic tetrahedron).. */
68  typedef sofa::type::fixed_array<PointID, 6> HighOrderTetrahedronPosition;
69  /* specify for each control point lying on a Hexahedron : the control point index, the index of the Hexahedron,
70  the 3 integers specifying the degree of the element in the x, y and z directions, the 3 integers specifying the position within this hexahedron (i.e. 121 ).. */
71  typedef sofa::type::fixed_array<PointID, 8> HighOrderHexahedronPosition;
