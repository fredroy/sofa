196  template<class T>
197  static bool canCreate(T*& obj, core::objectmodel::BaseContext* context, core::objectmodel::BaseObjectDescription* arg)
198  {
199  State<In>* stin = nullptr;
200  State<Out>* stout = nullptr;
201 
202  std::string inPath, outPath;
203 
204  if (arg->getAttribute("input"))
205  inPath = arg->getAttribute("input");
206  else
207  inPath = "@../";
208 
209  context->findLinkDest(stin, inPath, nullptr);
210 
211  if (arg->getAttribute("output"))
212  outPath = arg->getAttribute("output");
213  else
214  outPath = "@./";
215 
216  context->findLinkDest(stout, outPath, nullptr);
217 
218  if (stin == nullptr)
219  {
220  arg->logError("Data attribute 'input' does not point to a mechanical state of data type '"+std::string(In::Name())+"' and none can be found in the parent node context.");
221  return false;
222  }
223 
224  if (stout == nullptr)
225  {
226  arg->logError("Data attribute 'output' does not point to a mechanical state of data type '"+std::string(Out::Name())+"' and none can be found in the parent node context.");
227  return false;
228  }
229 
230  if (dynamic_cast<BaseObject*>(stin) == dynamic_cast<BaseObject*>(stout))
231  {
232  // we should refuse to create mappings with the same input and output model, which may happen if a State object is missing in the child node
233  arg->logError("Both the input and the output point to the same mechanical state ('"+stin->getName()+"').");
234  return false;
235  }
236 
237  return BaseMapping::canCreate(obj, context, arg);
238  }

244  template<class T>
245  static typename T::SPtr create(T*, core::objectmodel::BaseContext* context, core::objectmodel::BaseObjectDescription* arg)
246  {
247  typename T::SPtr obj = sofa::core::objectmodel::New<T>();
248 
249  if (context)
250  context->addObject(obj);
251 
252  if (arg)
253  {
254  std::string inPath, outPath;
255  if (arg->getAttribute("input"))
256  inPath = arg->getAttribute("input");
257  else
258  inPath = "@../";
259 
260  if (arg->getAttribute("output"))
261  outPath = arg->getAttribute("output");
262  else
263  outPath = "@./";
264 
265  obj->fromModel.setPath( inPath );
266  obj->toModel.setPath( outPath );
267 
268  obj->parse(arg);
269  }
270 
271  return obj;
272  }
