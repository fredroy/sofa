44 template<class T>
45 class QDataSeries : public QLineSeries
46 {
47 protected:
48  const T* data0;
49 public:
50  typedef vector_data_trait<T> trait;
51  typedef typename trait::value_type value_type;
52  typedef vector_data_trait<value_type> vtrait;
53  typedef typename vtrait::value_type real_type;
54  QDataSeries() : data0(NULL) {}
55  QDataSeries(const QDataSeries<T>& c)
56  : QLineSeries()
57  , data0(c.data0)
58  {
59  parseData();
60  }
61 
62  void setData(const T* p)
63  {
64  data0 = p;
65  parseData();
66  }
67 
68  virtual QLineSeries* copy() const { return new QDataSeries<T>(*this); }
69  virtual sofa::Size size() const
70  {
71  if (data0 == NULL)
72  return 0;
73  else
74  return trait::size(*(data0));
75  }
76 
77  virtual void parseData()
78  {
79  clear();
80  m_xMin = m_yMin = 1000000000.0;
81  m_xMax = m_yMax = -1000000000.0;
82 
83  for (sofa::Index i = 0; i < this->size(); i++)
84  {
85  QPointF data = sample(i);
86  append(data);
87 
88  if (data.x() > m_xMax)
89  m_xMax = data.x();
90  if (data.x() < m_xMin)
91  m_xMin = data.x();
92 
93  if (data.y() > m_yMax)
94  m_yMax = data.y();
95  if (data.y() < m_yMin)
96  m_yMin = data.y();
97  }
98  }
99 
100  virtual QPointF sample (sofa::Index i) const
101  {
102  if (i >= size())
103  return QPointF();
104  else if (vtrait::size(*trait::get(*(data0), i)) < 2)
105  return QPointF(i, (double)(*vtrait::get(*trait::get(*(data0), i), 0)));
106  else
107  return QPointF((double)(*vtrait::get(*trait::get(*(data0), i), 0)), (double)(*vtrait::get(*trait::get(*(data0), i), 1)));
108  }
109 
110  real_type m_xMin, m_yMin;
111  real_type m_xMax, m_yMax;
112 };

185  void readFromData(const data_type& d0)
186  {
187  double minX = 1000000000.0;
188  double maxX = -1000000000.0;
189  double minY = 1000000000.0;
190  double maxY = -1000000000.0;
191  currentData=d0;
192  const data_type& d = currentData;
193  auto nbData = trait::size(d);
194 
195  for (int i = 0; i < nbData; ++i)
196  {
197  const curve_type* v = trait::get(d, i);
198  const char* name = trait::header(d, i);
199  QString sName;
200  if (name && *name)
201  sName = name;
202  else
203  sName = "Unknown_" + QString::number(m_curves.size());
204 
205  auto itM = m_curves.find(sName);
206  CurveData* cdata;
207  if (itM != m_curves.end())
208  {
209  cdata = itM->second;
210  }
211  else {
212  // new curve data to register and plot
213  cdata = new CurveData();
214  m_curves[sName] = cdata;
215 
216  cdata->setName(sName);
217  cdata->setPen(QColor::fromHsv(255 * i / nbData, 255, 255));
218  m_chart->addSeries(cdata);
219 
220  cdata->attachAxis(m_axisY);
221  cdata->attachAxis(m_axisX);
222  }
223 
224  cdata->setData(v);
225  // gather min and max over all curves
226  if(cdata->m_xMin < minX) minX = cdata->m_xMin;
227  if(cdata->m_xMax > maxX) maxX = cdata->m_xMax;
228  if(cdata->m_yMin < minY) minY = cdata->m_yMin;
229  if(cdata->m_yMax > maxY) maxY = cdata->m_yMax;
230  }
231 
232  m_axisX->setRange(minX, maxX);
233  m_axisY->setRange(minY, maxY);
234  }
235 
236  void exportGNUPlot(const std::string &baseFileName) const
237  {
238  int n = trait::size(currentData);
239  for (int i=0; i<n; ++i)
240  {
241  const curve_type& v = *(trait::get(currentData,i));
242  const std::string filename=baseFileName + std::string("_") + getCurveFilename(i) + std::string(".txt");
243  std::cerr << "Export GNUPLOT file: " + filename << std::endl;
244  std::ofstream gnuplotFile(filename.c_str());
245  for (unsigned int idx=0; idx<v.size(); ++idx)
246  gnuplotFile << idx << " " << v[idx] << "\n";
247  gnuplotFile.close();
248  }
249  }
