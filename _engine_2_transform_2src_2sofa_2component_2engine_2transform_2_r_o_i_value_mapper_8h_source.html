106  void doUpdate() override
107  {
108  const unsigned int nb = nbROIs.getValue();
109  f_indices.resize(nb);
110  f_value.resize(nb);
111  if(!nb) return;
112 
113  const Real& defaultValue = p_defaultValue.getValue();
114  helper::WriteOnlyAccessor< Data< type::vector<Real> > > outputValues = f_outputValues;
115  outputValues.clear();
116 
117  for(size_t j=0; j<nb;j++)
118  {
119  helper::ReadAccessor< Data< type::vector<Index> > > indices = f_indices[j];
120  const Real& value = f_value[j]->getValue();
121 
122  for (const Index ind : indices)
123  {
124  if (ind >= outputValues.size())
125  {
126  outputValues.wref().resize(ind+1, defaultValue);
127  }
128  outputValues[ind] = value;
129  }
130  }
131  }
