51 // compare the shape of two matrix given in CSR format, return false if the matrices have the same shape and return true if their shapes are different
52 inline bool compareMatrixShape(int s_M, int * M_colptr,int * M_rowind, int s_P, int * P_colptr,int * P_rowind) {
53  if (s_M != s_P) return true;
54  if (M_colptr[s_M] != P_colptr[s_M] ) return true;
55 
56  for (int i=0;i<s_P;i++) {
57  if (M_colptr[i]!=P_colptr[i]) return true;
58  }
59 
60  for (int i=0;i<M_colptr[s_M];i++) {
61  if (M_rowind[i]!=P_rowind[i]) return true;
62  }
63 
64  return false;
65 }
