272  template<class OutType, class InType>
273  void mult_impl(OutType& result, const InType& data) const {
274 
275  if( data.empty() ) return;
276 
277  // use optimized product if possible
278  if(canCast(data)) {
279 
280 #if (SOFA_LINEARALGEBRA_HAVE_OPENMP == 1)
281  if( alias(result, data) )
282  map(result) = linearsolver::mul_EigenSparseDenseMatrix_MT( this->compressedMatrix, map(data).template cast<Real>() );
283  else
284  map(result).noalias() = linearsolver::mul_EigenSparseDenseMatrix_MT( this->compressedMatrix, map(data).template cast<Real>() );
285 #else
286  if( alias(result, data) ) {
287  this->map(result) = (this->compressedMatrix *
288  this->map(data).template cast<Real>()).template cast<OutReal>();
289  } else {
290  this->map(result).noalias() = (this->compressedMatrix *
291  this->map(data).template cast<Real>()).template cast<OutReal>();
292  }
293 #endif
294 
295  return;
296  }
297  // convert the data to Eigen type
298  VectorEigenOut aux1(this->colSize(),1), aux2(this->rowSize(),1);
299  for(size_t i = 0, n = data.size(); i < n; ++i) {
300  for(unsigned j = 0; j < Nin; ++j) {
301  aux1[Nin * i + j] = data[i][j];
302  }
303  }
304 
305  // compute the product
306 #if (SOFA_LINEARALGEBRA_HAVE_OPENMP == 1)
307  aux2.noalias() = linearsolver::mul_EigenSparseDenseMatrix_MT( this->compressedMatrix, aux1 );
308 #else
309  aux2.noalias() = this->compressedMatrix * aux1;
310 #endif
311 
312  // convert the result back to the Sofa type
313  for(size_t i = 0, n = result.size(); i < n; ++i) {
314  for(unsigned j = 0; j < Nout; ++j) {
315  result[i][j] = aux2[Nout * i + j];
316  }
317  }
318  }

323 
324  if( data.empty() ) return;
325 
326  // use optimized product if possible
327  if( canCast(data) ) {
328 
329 #if (SOFA_LINEARALGEBRA_HAVE_OPENMP == 1)

332  else
333  {
334  typename map_traits<OutType>::map_type r = map(result);
335  r.noalias() += linearsolver::mul_EigenSparseDenseMatrix_MT( this->compressedMatrix, this->map(data).template cast<Real>() * fact ).template cast<OutReal>();

485  template<class T>

506  template<class mat>
507  helper::OwnershipSPtr<mat> convertSPtr( const linearalgebra::BaseMatrix* m) {
508  assert( m );
509 
510  {
511  typedef linearalgebra::EigenBaseSparseMatrix<SReal> matrixr;
512  const matrixr* smr = dynamic_cast<const matrixr*> (m);
513  // no need to create temporary data, so the SPtr does not take this ownership
514  if ( smr ) return helper::OwnershipSPtr<mat>(&smr->compressedMatrix, false);
515  }
516 
517  msg_warning("EigenSparseMatrix")<<"convertSPtr: slow matrix conversion (scalar type conversion)";
518 
519  {
520  typedef linearalgebra::EigenBaseSparseMatrix<double> matrixd;
521  const matrixd* smd = dynamic_cast<const matrixd*> (m);
522  // the cast is creating a temporary matrix, the SPtr takes its ownership, so its deletion will be transparent
523  if ( smd ) return helper::OwnershipSPtr<mat>( new mat(smd->compressedMatrix.cast<SReal>()), true );
524  }
525 
526  {
527  typedef linearalgebra::EigenBaseSparseMatrix<float> matrixf;
528  const matrixf* smf = dynamic_cast<const matrixf*>(m);
529  // the cast is creating a temporary matrix, the SPtr takes its ownership, so its deletion will be transparent
530  if( smf ) return helper::OwnershipSPtr<mat>( new mat(smf->compressedMatrix.cast<SReal>()), true );
531  }
532 
533  msg_warning("EigenSparseMatrix")<<"convertSPtr: very slow matrix conversion (from BaseMatrix)";
534 
535  mat* res = new mat(m->rowSize(), m->colSize());
536 
537  res->reserve(res->rows() * res->cols());
538  for(unsigned i = 0, n = res->rows(); i < n; ++i) {
539  res->startVec( i );
540  for(unsigned j = 0, k = res->cols(); j < k; ++j) {
541  SReal e = m->element(i, j);
542  if( e ) res->insertBack(i, j) = e;
543  }
544  }
545 
546  // a temporary matrix is created, the SPtr takes its ownership, so its deletion will be transparent
547  return helper::OwnershipSPtr<mat>(res, true);
548  }
