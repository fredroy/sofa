845 template<sofa::Size S, class real>
846 [[nodiscard]] constexpr bool invertMatrix(Mat<S,S,real>& dest, const Mat<S,S,real>& from)
847 {
848  sofa::Size i{0}, j{0}, k{0};
849  Vec<S, sofa::Size> r, c, row, col;
850 
851  Mat<S,S,real> m1 = from;
852  Mat<S,S,real> m2;
853  m2.identity();
854 
855  for ( k = 0; k < S; k++ )
856  {
857  // Choosing the pivot
858  real pivot = 0;
859  for (i = 0; i < S; i++)
860  {
861  if (row[i])
862  continue;
863  for (j = 0; j < S; j++)
864  {
865  if (col[j])
866  continue;
867  real t = m1[i][j]; if (t<0) t=-t;
868  if ( t > pivot)
869  {
870  pivot = t;
871  r[k] = i;
872  c[k] = j;
873  }
874  }
875  }
876 
877  if (equalsZero(pivot))
878  {
879  return false;
880  }
881 
882  row[r[k]] = col[c[k]] = 1;
883  pivot = m1[r[k]][c[k]];
884 
885  // Normalization
886  m1[r[k]] /= pivot; m1[r[k]][c[k]] = 1;
887  m2[r[k]] /= pivot;
888 
889  // Reduction
890  for (i = 0; i < S; i++)
891  {
892  if (i != r[k])
893  {
894  real f = m1[i][c[k]];
895  m1[i] -= m1[r[k]]*f; m1[i][c[k]] = 0;
896  m2[i] -= m2[r[k]]*f;
897  }
898  }
899  }
900 
901  for (i = 0; i < S; i++)
902  for (j = 0; j < S; j++)
903  if (c[j] == i)
904  row[i] = r[j];
905 
906  for ( i = 0; i < S; i++ )
907  dest[i] = m2[row[i]];
908 
909  return true;
910 }
