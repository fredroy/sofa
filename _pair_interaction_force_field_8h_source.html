162  template<class T>
163  static bool canCreate(T* obj, objectmodel::BaseContext* context, objectmodel::BaseObjectDescription* arg)
164  {
165  MechanicalState<DataTypes>* mstate1 = nullptr;
166  MechanicalState<DataTypes>* mstate2 = nullptr;
167  std::string object1 = arg->getAttribute("object1","@./");
168  std::string object2 = arg->getAttribute("object2","@./");
169  if (object1.empty()) object1 = "@./";
170  if (object2.empty()) object2 = "@./";
171 
172  context->findLinkDest(mstate1, object1, nullptr);
173  context->findLinkDest(mstate2, object2, nullptr);
174 
175  if (!mstate1) {
176  arg->logError("Data attribute 'object1' does not point to a valid mechanical state of datatype '" + std::string(DataTypes::Name()) + "'.");
177  return false;
178  }
179  if (!mstate2) {
180  arg->logError("Data attribute 'object2' does not point to a valid mechanical state of datatype '" + std::string(DataTypes::Name()) + "'.");
181  return false;
182  }
183 
184  return BaseInteractionForceField::canCreate(obj, context, arg);
185  }
