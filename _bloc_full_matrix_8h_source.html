169  {
170  FullVector<Real2> res(rowSize());
171  for (Index bi=0; bi<nBRow; ++bi)
172  {
173  Index bj = 0;
174  for (Index i=0; i<BSIZE; ++i)
175  {
176  Real r = 0;
177  for (Index j=0; j<BSIZE; ++j)
178  {
179  r += bloc(bi,bj)[i][j] * v[(bi + bj - 1)*BSIZE + j];
180  }
181  res[bi*BSIZE + i] = r;
182  }
183  for (++bj; bj<nBCol; ++bj)
184  {
185  for (Index i=0; i<BSIZE; ++i)
186  {
187  Real r = 0;
188  for (Index j=0; j<BSIZE; ++j)
189  {
190  r += bloc(bi,bj)[i][j] * v[(bi + bj - 1)*BSIZE + j];
191  }
192  res[bi*BSIZE + i] += r;
193  }
194  }
195  }
196  return res;
197  }
