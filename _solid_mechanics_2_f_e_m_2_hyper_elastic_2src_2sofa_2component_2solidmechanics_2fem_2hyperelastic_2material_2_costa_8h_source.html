54  virtual Real getStrainEnergy(StrainInformation<DataTypes> *sinfo, const MaterialParameters<DataTypes> &param) {
55  Real a=param.parameterArray[0];
56  Real k0=param.parameterArray[1];
57  Real bff=param.parameterArray[2];
58  Real bfs=param.parameterArray[3];
59  Real bss=param.parameterArray[4];
60  Real bfn=param.parameterArray[5];
61  Real bsn=param.parameterArray[6];
62  Real bnn=param.parameterArray[7];
63  MatrixSym Id;
64  Id.identity();
65  MatrixSym E=(sinfo->deformationTensor-Id)/2.0;
66  Real Q=bff*E[0]*E[0]+2*bfs*E[1]*E[1]+bss*E[2]*E[2]+2*bfn*E[3]*E[3]+2*bsn*E[4]*E[4]+bnn*E[5]*E[5];
67  Real Qbar= pow(sinfo->J,(Real)(-4.0/3.0))*Q;
68  return a*(exp(Qbar)-1)/2+k0*(sinfo->J*log(sinfo->J)-sinfo->J-1);
69  }
