41 template<class DataTypes>
42 simulation::Node::SPtr createSunPlanetSystem(
43  simulation::Node::SPtr root,
44  double mSun,
45  double mPlanet,
46  double g,
47  typename DataTypes::Coord xSun,
48  typename DataTypes::Deriv vSun,
49  typename DataTypes::Coord xPlanet,
50  typename DataTypes::Deriv vPlanet)
51 {
52 
53  // Mechanical object with 2 dofs: first sun, second planet
54  MechanicalObject3::SPtr sunPlanet_dof = modeling::addNew<MechanicalObject3>(root,"sunPlanet_MO");
55 
56  // Set position and velocity
57  sunPlanet_dof->resize(2);
58  // Position
59  MechanicalObject3::VecCoord positions(2);
60  positions[0] = xSun;
61  positions[1]= xPlanet;
62  MechanicalObject3::WriteVecCoord xdof = sunPlanet_dof->writePositions();
63  sofa::testing::copyToData( xdof, positions );
64  // Velocity
65  MechanicalObject3::VecDeriv velocities(2);
66  velocities[0] = vSun;
67  velocities[1]= vPlanet;
68  MechanicalObject3::WriteVecDeriv vdof = sunPlanet_dof->writeVelocities();
69  sofa::testing::copyToData( vdof, velocities );
70 
71  // Fix sun
72  FixedConstraint3::SPtr fixed = modeling::addNew<FixedConstraint3>(root,"FixedSun");
73  fixed->addConstraint(0);
74 
75  // Uniform Mass
76  UniformMass3::SPtr massPtr = modeling::addNew<UniformMass3>(root,"mass");
77  massPtr->d_totalMass.setValue(mPlanet + mSun);
78 
79  // Lennard Jones Force Field
80  //TODO: Find another FF to add in this test. If Sympletic solver are not deprecated. Erik STC #4
81  typename component::forcefield::LennardJonesForceField<DataTypes>::SPtr ff =
82  modeling::addNew<typename component::forcefield::LennardJonesForceField<DataTypes> >(root);
83  // Set froce field parameters
84  ff->setAlpha(1);
85  ff->setBeta(-1);
86  ff->setAInit(mPlanet*mSun*g);
87 
88  return root;
89 
90 }
