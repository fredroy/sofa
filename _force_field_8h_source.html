160  template<class IndexArray, class ElementMat>
161  void addToMatrix(sofa::linearalgebra::BaseMatrix* bm, unsigned offset, const IndexArray& nodeIndex, const ElementMat& em, SReal scale )
162  {
163  constexpr auto S = DataTypes::deriv_total_size; // size of node blocks
164  for (unsigned n1=0; n1<nodeIndex.size(); n1++)
165  {
166  for(unsigned i=0; i<S; i++)
167  {
168  const unsigned ROW = offset + S*nodeIndex[n1] + i; // i-th row associated with node n1 in BaseMatrix
169  const unsigned row = S*n1+i; // i-th row associated with node n1 in the element matrix
170 
171  for (unsigned n2=0; n2<nodeIndex.size(); n2++)
172  {
173  for (unsigned j=0; j<S; j++)
174  {
175  const unsigned COLUMN = offset + S*nodeIndex[n2] +j; // j-th column associated with node n2 in BaseMatrix
176  const unsigned column = S*n2+j; // j-th column associated with node n2 in the element matrix
177  bm->add( ROW,COLUMN, em[row][column]* scale );
178  }
179  }
180  }
181  }
182  }

189  template<class T>

191  {
192  const std::string attributeName {"mstate"};
193  std::string mstateLink = arg->getAttribute(attributeName,"");
194  if (mstateLink.empty())
195  {
196  if (dynamic_cast<MechanicalState<DataTypes>*>(context->getMechanicalState()) == nullptr)
197  {
198  arg->logError("Since the attribute '" + attributeName + "' has not been specified, a mechanical state "
199  "with the datatype '" + DataTypes::Name() + "' has been searched in the current context, but not found.");
200  return false;
201  }
202  }
203  else
204  {
205  MechanicalState<DataTypes>* mstate = nullptr;
206  context->findLinkDest(mstate, mstateLink, nullptr);
207  if (!mstate)
208  {
209  arg->logError("Data attribute '" + attributeName + "' does not point to a valid mechanical state of datatype '" + std::string(DataTypes::Name()) + "'.");
210  return false;
211  }
212  }
213  return BaseObject::canCreate(obj, context, arg);
