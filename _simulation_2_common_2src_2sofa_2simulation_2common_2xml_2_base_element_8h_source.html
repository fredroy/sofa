175  template<class Node=BaseElement>
176  class child_iterator
177  {
178  protected:
179  BaseElement* parent;
180  ChildList::iterator it;
181  Node* current;
182  child_iterator(BaseElement* parent, ChildList::iterator it)
183  : parent(parent), it(it), current(nullptr)
184  {
185  checkIt();
186  }
187  void checkIt()
188  {
189  while (it != parent->children.end())
190  {
191  current=dynamic_cast<Node*>(*it);
192  if (current!=nullptr) return;
193  ++it;
194  }
195  current = nullptr;
196  }
197  public:
198  operator Node*() { return current; }
199  Node* operator->() { return current; }
200  void operator ++() { ++it; checkIt(); }
201  bool operator==(const child_iterator<Node>& i) const
202  {
203  return it == i.it;
204  }
205  friend class BaseElement;
206  };
