157  template<class T>
158  static bool canCreate ( T*& obj, core::objectmodel::BaseContext* context, core::objectmodel::BaseObjectDescription* arg )
159  {
160  In* stin = nullptr;
161  Out* stout = nullptr;
162 
163  std::string inPath, outPath;
164 
165  if (arg->getAttribute("input"))
166  inPath = arg->getAttribute("input");
167  else
168  inPath = "@../";
169 
170  context->findLinkDest(stin, inPath, nullptr);
171 
172  if (arg->getAttribute("output"))
173  outPath = arg->getAttribute("output");
174  else
175  outPath = "@./";
176 
177  context->findLinkDest(stout, outPath, nullptr);
178 
179  if (stin == nullptr)
180  {
181  arg->logError("Data attribute 'input' does not point to a valid mesh topology and none can be found in the parent node context.");
182  return false;
183  }
184 
185  if (stout == nullptr)
186  {
187  arg->logError("Data attribute 'output' does not point to a valid mesh topology and none can be found in the current node context.");
188  return false;
189  }
190 
191  if (dynamic_cast<BaseObject*>(stin) == dynamic_cast<BaseObject*>(stout))
192  {
193  // we should refuse to create mappings with the same input and output model, which may happen if a State object is missing in the child node
194  arg->logError("Both the input mesh and the output mesh points to the same mesh topology ('"+stin->getName()+"').");
195  return false;
196  }
197 
198  return BaseObject::canCreate(obj, context, arg);
199  }
