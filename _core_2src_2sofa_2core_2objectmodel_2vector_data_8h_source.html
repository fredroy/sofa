106  void resize(const unsigned int size)
107  {
108  core::DataEngine* componentAsDataEngine = m_dataEngineDataType!= DataEngineDataType::DataEngineNothing ? m_component->toDataEngine() : nullptr;
109 
110  if (size < this->size()) {
111  // removing some data if size is inferior than current size
112  for (unsigned int i=size; i<this->size(); ++i) {
113  if (componentAsDataEngine!=nullptr)
114  {
115  if(m_dataEngineDataType== DataEngineDataType::DataEngineInput) componentAsDataEngine->delInput((*this)[i]);
116  else if(m_dataEngineDataType== DataEngineDataType::DataEngineOutput) componentAsDataEngine->delOutput((*this)[i]);
117  }
118  delete (*this)[i];
119  }
120  if( size ) Inherit::resize(size);
121  else Inherit::clear();
122  }
123  else
124  {
125  for (unsigned int i=this->size(); i<size; ++i)
126  {
127  std::ostringstream oname, ohelp;
128  oname << m_name << (i+1);
129  ohelp << m_help << "(" << (i+1) << ")";
130  Data< T >* d = new Data< T >(m_defaultValue, getAStringCopy(ohelp.str().c_str()), true, false);
131  d->setName(oname.str());

146  {

149

152 
153  helper::WriteOnlyAccessor< Data<T> > out = outputData;
154  out.clear();
155  out.reserve(nbElems);
156  for( size_t i=0 ; i<nbInput ; ++i )
157  {
158  helper::ReadAccessor< Data<T> > in = vectorData[i];
159  for( size_t j=0, jend=in.size() ; j<jend ; ++j )
160  out.push_back(in[j]);
161  }
162  }
