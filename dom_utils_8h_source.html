42  static qreal qrealFromDom(const QDomElement &e, const QString &attribute,
43  qreal defValue) {
44  qreal value = defValue;
45  if (e.hasAttribute(attribute)) {
46  const QString s = e.attribute(attribute);
47  bool ok;
48  value = s.toDouble(&ok);
49  if (!ok) {
50  warning(QString("'%1' is not a valid qreal syntax for attribute \"%2\" "
51  "in initialization of \"%3\". Setting value to %4.")
52  .arg(s)
53  .arg(attribute)
54  .arg(e.tagName())
55  .arg(QString::number(defValue)));
56  value = defValue;
57  }
58  } else {
59  warning(QString("\"%1\" attribute missing in initialization of \"%2\". "
60  "Setting value to %3.")
61  .arg(attribute)
62  .arg(e.tagName())
63  .arg(QString::number(value)));
64  }
65 
66 #if defined(isnan)
67  // The "isnan" method may not be available on all platforms.
68  // Find its equivalent or simply remove these two lines
69  if (isnan(value))
70  warning(
71  QString(
72  "Warning, attribute \"%1\" initialized to Not a Number in \"%2\"")
73  .arg(attribute)
74  .arg(e.tagName()));
75 #endif
76 
77  return value;
78  }

137  static bool boolFromDom(const QDomElement &e, const QString &attribute,
138  bool defValue) {
139  bool value = defValue;
140  if (e.hasAttribute(attribute)) {
141  const QString s = e.attribute(attribute);
142  if (s.toLower() == QString("true"))
143  value = true;
144  else if (s.toLower() == QString("false"))
145  value = false;
146  else {
147  warning(
148  QString("'%1' is not a valid boolean syntax for attribute \"%2\" "
149  "in initialization of \"%3\". Setting value to %4.")
150  .arg(s)
151  .arg(attribute)
152  .arg(e.tagName())
153  .arg(defValue ? "true" : "false"));
154  }
155  } else {
156  warning(QString("\"%1\" attribute missing in initialization of \"%2\". "
157  "Setting value to %3.")
158  .arg(attribute)
159  .arg(e.tagName())
160  .arg(value ? "true" : "false"));
161  }
162 
163  return value;
164  }

171  static QDomElement QColorDomElement(const QColor &color, const QString &name,
172  QDomDocument &doc) {
173  QDomElement de = doc.createElement(name);
174  de.setAttribute("red", QString::number(color.red()));
175  de.setAttribute("green", QString::number(color.green()));
176  de.setAttribute("blue", QString::number(color.blue()));
177  return de;
178  }

180  static QColor QColorFromDom(const QDomElement &e) {
181  int color[3];
182  QStringList attribute;
183  attribute << "red"
184  << "green"
185  << "blue";
186  for (int i = 0; i < attribute.count(); ++i)
187  color[i] = DomUtils::intFromDom(e, attribute[i], 0);
188  return QColor(color[0], color[1], color[2]);
189  }
