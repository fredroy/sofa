31 template<Size TN, typename T> class bloc_index_func
32 {
33 public:
34  enum { N = TN };
35  static void split(T& index, T& modulo)
36  {
37  if constexpr (N == 1)
38  {
39  ; // nothing
40  return;
41  }
42  if constexpr (N == 2)
43  {
44  modulo = index & 1;
45  index = index >> 1;
46  return;
47  }
48  if constexpr (N == 4)
49  {
50  modulo = index & 3;
51  index = index >> 2;
52  return;
53  }
54  if constexpr (N == 8)
55  {
56  modulo = index & 7;
57  index = index >> 3;
58  return;
59  }
60  else
61  {
62  modulo = index % N;
63  index = index / N;
64  return;
65  }
66  }
67 };

69 // by default, supposing T is a type::Mat (useful for type derivated from type::Mat)
70 template<class T, typename IndexType>
71 class matrix_bloc_traits
72 {
73 public:
74  typedef T Block;
75  using Bloc SOFA_ATTRIBUTE_DEPRECATED__BLOCK_RENAMING_2404() = Block;
76  typedef typename T::Real Real;
77  enum { NL = T::nbLines };
78  enum { NC = T::nbCols };
79  static Real& v(Block& b, IndexType row, IndexType col) { return b[row][col]; }
80  static const Real& v(const Block& b, IndexType row, IndexType col) { return b[row][col]; }
81  static void clear(Block& b) { b.clear(); }
82  static bool empty(const Block& b)
83  {
84  for (IndexType i=0; i<NL; ++i)
85  for (IndexType j=0; j<NC; ++j)
86  if (b[i][j] != 0) return false;
87  return true;
88  }
89  static void invert(Block& result, const Block& b) { result.invert(b); }
90 
91  static void split_row_index(IndexType& index, IndexType& modulo) { bloc_index_func<NL, IndexType>::split(index, modulo); }
92  static void split_col_index(IndexType& index, IndexType& modulo) { bloc_index_func<NC, IndexType>::split(index, modulo); }
93 
94  static sofa::linearalgebra::BaseMatrix::ElementType getElementType() { return matrix_bloc_traits<Real, IndexType>::getElementType(); }
95 };
