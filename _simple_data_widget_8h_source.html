54 template<class T>
55 class data_widget_trait
56 {
57 public:
58  typedef T data_type;
59  typedef QLineEdit Widget;
60  static Widget* create(QWidget* parent, const data_type& /*d*/)
61  {
62  Widget* w = new Widget(parent);
63  w->setFocusPolicy(Qt::StrongFocus);
64  return w;
65  }
66  static void readFromData(Widget* w, const data_type& d)
67  {
68  std::ostringstream o;
69  o << d;
70  if (o.str() != w->text().toStdString())
71  w->setText(QString(o.str().c_str()));
72  }
73  static void writeToData(Widget* w, data_type& d)
74  {
75  std::string s = w->text().toStdString();
76  std::istringstream i(s);
77  i >> d;
78  }
79  static void setReadOnly(Widget* w, bool readOnly)
80  {
81  w->setEnabled(!readOnly);
82  w->setReadOnly(readOnly);
83  }
84  static void connectChanged(Widget* w, DataWidget* datawidget)
85  {
86  datawidget->connect(w, SIGNAL( textChanged(const QString&) ), datawidget, SLOT(setWidgetDirty()));
87  }
88 };

96 template<class T>
97 class data_widget_container
98 {
99 public:
100  typedef T data_type;
101  typedef data_widget_trait<data_type> helper;
102  typedef typename helper::Widget Widget;
103  typedef QHBoxLayout Layout;
104  Widget* w;
105  Layout* container_layout;
106  data_widget_container() : w(nullptr),container_layout(nullptr) { }
107 
108  bool createLayout(DataWidget* parent)
109  {
110  if(parent->layout() != nullptr) return false;
111  container_layout = new QHBoxLayout(parent);
112  //parent->setLayout(container_layout);
113  return true;
114  }
115 
116  bool createLayout(QLayout* layout)
117  {
118  if(container_layout) return false;
119  container_layout = new QHBoxLayout();
120  layout->addItem(container_layout);
121  return true;
122  }
123 
124  bool createWidgets(DataWidget* parent, const data_type& d, bool readOnly)
125  {
126  w = helper::create(parent,d);
127  if (w == nullptr) return false;
128 
129  helper::readFromData(w, d);
130  if (readOnly)
131  helper::setReadOnly(w, readOnly);
132  else
133  helper::connectChanged(w, parent);
134  return true;
135  }
136  void setReadOnly(bool readOnly)
137  {
138  if(w){
139  w->setEnabled(!readOnly);
140  helper::setReadOnly(w, readOnly);
141  }
142  }
143  void readFromData(const data_type& d)
144  {
145  helper::readFromData(w, d);
146  }
147  void writeToData(data_type& d)
148  {
149  helper::writeToData(w, d);
150  }
151 
152  void insertWidgets()
153  {
154  assert(w);
155  container_layout->addWidget(w);
156  }
157 };

164 protected:

177  const data_type& d = this->getData()->getValue();

183 
184  return true;
185  }

199  }

204  container.writeToData(d);
205  this->getData()->endEdit();
206  }

537  {
538  for (sofa::Size y=0; y<L; ++y)
539  for (sofa::Size x=0; x<C; ++x)
540  w[y][x].readFromData(*vhelper::get(*rhelper::get(d,y),x));
541  }
542  void writeToData(data_type& d)
543  {
544  for (sofa::Size y=0; y<L; ++y)
545  {
546  row_type r = *rhelper::get(d,y);
547  for (sofa::Size x=0; x<C; ++x)
