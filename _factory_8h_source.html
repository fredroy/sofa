59 template <typename TKey, class TObject, typename TArgument = NoArgument, typename TPtr = TObject* >
60 class Factory
61 {
62 public:
63  typedef TKey Key;

65  typedef TPtr ObjectPtr;
66  typedef TArgument Argument;
67  typedef BaseCreator<Object, Argument, ObjectPtr> Creator;
68 
69 protected:
70  std::multimap<Key, Creator*> registry;

72 public:
73  bool registerCreator(Key key, Creator* creator, bool multi=false)
74  {
75  if(!multi && this->registry.find(key) != this->registry.end())
76  return false; // key used
77  logFactoryRegister(gettypename(typeid(Object)), gettypename(creator->type()), key, multi);
78  this->registry.insert(std::pair<Key, Creator*>(key, creator));
