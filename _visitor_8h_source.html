232 template< class VisitorType, class VContext, class ObjectType>
233 void Visitor::runVisitorTask(VisitorType *visitor,
234  VContext *ctx,
235  void (VisitorType::*task)(VContext *, ObjectType *),
236  ObjectType *ptr,
237  const std::string &typeInfo)
238 {
239  if(this->testTags(ptr))
240  {
241 #ifdef SOFA_VERBOSE_TRAVERSAL
242  visitor->debug_write_state_before(ptr);
243 #endif
244  auto t = this->begin(ctx, ptr, typeInfo);
245  (visitor->*task)(ctx, ptr);
246  this->end(ctx, ptr, t);
247 #ifdef SOFA_VERBOSE_TRAVERSAL
248  visitor->debug_write_state_after(ptr);
249 #endif
250  }
251 }
