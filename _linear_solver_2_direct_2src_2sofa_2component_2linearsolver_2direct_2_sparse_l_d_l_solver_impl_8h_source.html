51 inline void CSPARSE_symbolic (int n,int * M_colptr,int * M_rowind,int * colptr,int * perm,int * invperm,int * Parent, int * Flag, int * Lnz)
52 {
53  for (int k = 0 ; k < n ; k++)
54  {
55  Parent [k] = -1 ; // parent of k is not yet known
56  Flag [k] = k ; // mark node k as visited
57  Lnz [k] = 0 ; // count of nonzeros in column k of L
58  int kk = perm[k]; // kth original, or permuted, column
59  for (int p = M_colptr[kk] ; p < M_colptr[kk+1] ; p++)
60  {
61  // A (i,k) is nonzero (original or permuted A)
62  int i = invperm[M_rowind[p]];
63  if (i < k)
64  {
65  // follow path from i to root of etree, stop at flagged node
66  for ( ; Flag [i] != k ; i = Parent [i])
67  {
68  // find parent of i if not yet determined
69  if (Parent [i] == -1) Parent [i] = k ;
70  Lnz [i]++ ; // L (k,i) is nonzero
71  Flag [i] = k ; // mark i as visited
72  }
73  }
74  }
75  }
76 
77  colptr[0] = 0 ;
78  for (int k = 0 ; k < n ; k++) colptr[k+1] = colptr[k] + Lnz[k] ;
79 }

81 template<class Real>
82 inline void CSPARSE_numeric(int n,int * M_colptr,int * M_rowind,Real * M_values,int * colptr,int * rowind,Real * values,Real * D,int * perm,int * invperm,int * Parent, int * Flag, int * Lnz, int * Pattern, Real * Y)
83 {
84  Real yi, l_ki ;
85  int i, p, kk, len, top ;
86 
87  for (int k = 0 ; k < n ; k++)
88  {
89  Y [k] = 0.0 ; // Y(0:k) is now all zero
90  top = n ; // stack for pattern is empty
91  Flag [k] = k ; // mark node k as visited
92  Lnz [k] = 0 ; // count of nonzeros in column k of L
93  kk = perm[k]; // kth original, or permuted, column
94  for (p = M_colptr[kk] ; p < M_colptr[kk+1] ; p++)
95  {
96  i = invperm[M_rowind[p]]; // get A(i,k)
97  if (i <= k)
98  {
99  Y[i] += M_values[p] ; // scatter A(i,k) into Y (sum duplicates)
100  for (len = 0 ; Flag[i] != k ; i = Parent[i])
101  {
102  Pattern [len++] = i ; // L(k,i) is nonzero
103  Flag [i] = k ; // mark i as visited
104  }
105  while (len > 0) Pattern[--top] = Pattern [--len] ;
106  }
107  }
108  // compute numerical values kth row of L (a sparse triangular solve)
109  D[k] = Y [k] ; // get D(k,k) and clear Y(k)
110  Y[k] = 0.0 ;
111  for ( ; top < n ; top++)
112  {
113  i = Pattern [top] ; // Pattern [top:n-1] is pattern of L(:,k)
114  yi = Y [i] ; // get and clear Y(i)
115  Y [i] = 0.0 ;
116  for (p = colptr[i] ; p < colptr[i] + Lnz [i] ; p++)
117  {
118  Y[rowind[p]] -= values[p] * yi ;
119  }
120  l_ki = yi / D[i] ; // the nonzero entry L(k,i)
121  D[k] -= l_ki * yi ;
122  rowind[p] = k ; // store L(k,i) in column form of L
123  values[p] = l_ki ;
124  Lnz[i]++ ; // increment count of nonzeros in col i
125  }
126 
127  if (D[k] == 0.0)
128  {
129  msg_error("SparseLDLSolver") << "Failed to factorize, D(k,k) is zero" ;
130  return;
131  }
132  }
133 }

159  template<class VecInt,class VecReal>
160  void solve_cpu(Real * x,const Real * b,SparseLDLImplInvertData<VecInt,VecReal> * data) {
161  int n = data->n;
162  const Real * invD = data->invD.data();
163  const int * perm = data->perm.data();
164  const int * L_colptr = data->L_colptr.data();
165  const int * L_rowind = data->L_rowind.data();
166  const Real * L_values = data->L_values.data();
167  const int * LT_colptr = data->LT_colptr.data();
168  const int * LT_rowind = data->LT_rowind.data();
169  const Real * LT_values = data->LT_values.data();
170 
171  Tmp.clear();
172  Tmp.fastResize(n);
173 
174  for (int j = 0 ; j < n ; j++) {
175  Real acc = b[perm[j]];
176  for (int p = LT_colptr [j] ; p < LT_colptr[j+1] ; p++) {
177  acc -= LT_values[p] * Tmp[LT_rowind[p]];
178  }
179  Tmp[j] = acc;
180  }
181 
182  for (int j = n-1 ; j >= 0 ; j--) {
183  Tmp[j] *= invD[j];
184 
185  for (int p = L_colptr[j] ; p < L_colptr[j+1] ; p++) {
186  Tmp[j] -= L_values[p] * Tmp[L_rowind[p]];
187  }
188 
189  x[perm[j]] = Tmp[j];
190  }
191  }

237  template<class VecInt,class VecReal>
238  void factorize(int n,int * M_colptr, int * M_rowind, Real * M_values, SparseLDLImplInvertData<VecInt,VecReal> * data) {
239  data->new_factorization_needed = data->P_colptr.size() == 0 || data->P_rowind.size() == 0 || compareMatrixShape(n, M_colptr, M_rowind, data->n,
240  (int *) data->P_colptr.data(),(int *) data->P_rowind.data());
241 
242  data->n = n;
243  data->P_nnz = M_colptr[data->n];
244  data->P_values.clear();data->P_values.fastResize(data->P_nnz);
245  memcpy(data->P_values.data(),M_values,data->P_nnz * sizeof(Real));
246 
247  // we test if the matrix has the same struct as previous factorized matrix
248  if (data->new_factorization_needed || !d_precomputeSymbolicDecomposition.getValue() )
249  {
250  sofa::helper::ScopedAdvancedTimer factorizationTimer("symbolic_factorization");
251  msg_info() << "Recomputing new factorization" ;
252 
253  data->perm.clear();data->perm.fastResize(data->n);
254  data->invperm.clear();data->invperm.fastResize(data->n);
255  data->invD.clear();data->invD.fastResize(data->n);
256  data->P_colptr.clear();data->P_colptr.fastResize(data->n+1);
257  data->L_colptr.clear();data->L_colptr.fastResize(data->n+1);
258  data->LT_colptr.clear();data->LT_colptr.fastResize(data->n+1);
259  data->P_rowind.clear();data->P_rowind.fastResize(data->P_nnz);
260 
261  memcpy(data->P_colptr.data(),M_colptr,(data->n+1) * sizeof(int));
262  memcpy(data->P_rowind.data(),M_rowind,data->P_nnz * sizeof(int));
263 
264  //ordering function
265  LDL_ordering( data->n , M_colptr , M_rowind , data->perm.data(), data->invperm.data() );
266 
267  data->Parent.clear();
268  data->Parent.resize(data->n);
269 
270  //symbolic factorization
271  LDL_symbolic(data->n,M_colptr,M_rowind,data->L_colptr.data(),
272  data->perm.data(),data->invperm.data(),data->Parent.data());
273 
274  data->L_nnz = data->L_colptr[data->n];
275  d_L_nnz.setValue(data->L_nnz);
276 
277  data->L_rowind.clear();data->L_rowind.fastResize(data->L_nnz);
278  data->L_values.clear();data->L_values.fastResize(data->L_nnz);
279  data->LT_rowind.clear();data->LT_rowind.fastResize(data->L_nnz);
280  data->LT_values.clear();data->LT_values.fastResize(data->L_nnz);
281  }
282 
283  Real * D = data->invD.data();
284  int * rowind = data->L_rowind.data();
285  int * colptr = data->L_colptr.data();
286  Real * values = data->L_values.data();
287  int * tran_rowind = data->LT_rowind.data();
288  int * tran_colptr = data->LT_colptr.data();
289  Real * tran_values = data->LT_values.data();
290 
291  //Numeric Factorization
292  {
293  sofa::helper::ScopedAdvancedTimer factorizationTimer("numeric_factorization");
294  LDL_numeric(data->n,M_colptr,M_rowind,M_values,colptr,rowind,values,D,
295  data->perm.data(),data->invperm.data(),data->Parent.data());
296 
297  //inverse the diagonal
298  for (int i=0;i<data->n;i++) D[i] = 1.0/D[i];
299  }
300 
301  // split the bloc diag in data->Bdiag
302 
303  if (data->new_factorization_needed || !d_precomputeSymbolicDecomposition.getValue() ) {
304  //Compute transpose in tran_colptr, tran_rowind, tran_values, tran_D
305  tran_countvec.clear();
306  tran_countvec.resize(data->n);
307 
308  //First we count the number of value on each row.
309  for (int j=0;j<data->L_nnz;j++) tran_countvec[rowind[j]]++;
310 
311  //Now we make a scan to build tran_colptr
312  tran_colptr[0] = 0;
313  for (int j=0;j<data->n;j++) tran_colptr[j+1] = tran_colptr[j] + tran_countvec[j];
314  }
315 
316  //we clear tran_countvec because we use it now to store how many values are written on each line
317  tran_countvec.clear();
318  tran_countvec.resize(data->n);
319 
320  for (int j=0;j<data->n;j++) {
321  for (int i=colptr[j];i<colptr[j+1];i++) {
322  int line = rowind[i];
323  tran_rowind[tran_colptr[line] + tran_countvec[line]] = j;
324  tran_values[tran_colptr[line] + tran_countvec[line]] = values[i];
325  tran_countvec[line]++;
326  }
327  }
328  }
