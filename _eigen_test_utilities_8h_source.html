62 class GetAssembledSizeVisitor : public sofa::simulation::Visitor
63 {
64 public:
65  GetAssembledSizeVisitor(const sofa::core::ExecParams* params = sofa::core::mechanicalparams::castToExecParams(core::mechanicalparams::defaultInstance()))
66  : sofa::simulation::Visitor(params)
67  {};
68  ~GetAssembledSizeVisitor() override {}
69 
70  Result processNodeTopDown(simulation::Node* gnode) override
71  {
72  if (gnode->mechanicalState != nullptr && (gnode->mechanicalMapping == nullptr || independentOnly == false))
73  {
74  xsize += gnode->mechanicalState->getSize() * gnode->mechanicalState->getCoordDimension();
75  vsize += gnode->mechanicalState->getMatrixSize();
76  }
77  return Visitor::RESULT_CONTINUE;
78  }
79  const char* getClassName() const override { return "GetAssembledSizeVisitor"; }
80 
81  unsigned positionSize() const { return xsize; }
82  unsigned velocitySize() const { return vsize; }
83  void setIndependentOnly(bool b) { independentOnly = b; }
84 
85 protected:
86  std::size_t xsize{ 0 };
87  std::size_t vsize{ 0 };
88  bool independentOnly{ false };
89 };

91 inline Eigen::VectorXd getVector(simulation::Node::SPtr root, core::ConstVecId id, bool indep = true )
92 {
93  GetAssembledSizeVisitor getSizeVisitor;
94  getSizeVisitor.setIndependentOnly(indep);
95  root->execute(getSizeVisitor);
96  unsigned size;
97  if (id.type == sofa::core::V_COORD)
98  size = getSizeVisitor.positionSize();
99  else
100  size = getSizeVisitor.velocitySize();
101  linearalgebra::FullVector<SReal> v(size);
102  GetVectorVisitor getVec( sofa::core::mechanicalparams::castToExecParams(core::mechanicalparams::defaultInstance()), &v, id);
103  getVec.setIndependentOnly(indep);
104  root->execute(getVec);
105 
106  Eigen::VectorXd ve(size);
107  for(size_t i=0; i<size; i++)
108  ve(i)=v[i];
109  return ve;
110 }
