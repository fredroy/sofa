91 inline Eigen::VectorXd getVector(simulation::Node::SPtr root, core::ConstVecId id, bool indep = true )
92 {
93  GetAssembledSizeVisitor getSizeVisitor;
94  getSizeVisitor.setIndependentOnly(indep);
95  root->execute(getSizeVisitor);
96  unsigned size;
97  if (id.type == sofa::core::V_COORD)
98  size = getSizeVisitor.positionSize();
99  else
100  size = getSizeVisitor.velocitySize();
101  linearalgebra::FullVector<SReal> v(size);
102  GetVectorVisitor getVec( sofa::core::mechanicalparams::castToExecParams(core::mechanicalparams::defaultInstance()), &v, id);
103  getVec.setIndependentOnly(indep);
104  root->execute(getVec);
105 
106  Eigen::VectorXd ve(size);
107  for(size_t i=0; i<size; i++)
108  ve(i)=v[i];
109  return ve;
