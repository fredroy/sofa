80  {

95  collModelPath2 = arg->getAttribute("collisionModel2");
96  else

99  context->findLinkDest(collModel2, collModelPath2, nullptr);
100 
101  if (collModel1 == nullptr && collModel2 == nullptr )

110  template<class T>
111  static typename T::SPtr create(T* , core::objectmodel::BaseContext* context, core::objectmodel::BaseObjectDescription* arg)
112  {

114  core::CollisionModel* collModel2 = nullptr;
115 
116  std::string collModelPath1;
117  std::string collModelPath2;
118 
119  if(arg)
120  {
121  collModelPath1 = arg->getAttribute(std::string("collisionModel1"), nullptr );
122  collModelPath2 = arg->getAttribute(std::string("collisionModel2"), nullptr );
123 
124  // now 3 cases
125  if ( strcmp( collModelPath1.c_str(),"" ) != 0 )
126  {
127  context->findLinkDest(collModel1, collModelPath1, nullptr);
128 
129  if ( strcmp( collModelPath2.c_str(),"" ) != 0 )
130  {
131  context->findLinkDest(collModel2, collModelPath2, nullptr);
132  }
133  }
134  else
135  {
136  context->findLinkDest(collModel1, collModelPath2, nullptr);
137  }
138  }
139  typename T::SPtr obj = sofa::core::objectmodel::New<T>( collModel1, collModel2 );
140 
141  if (context)
142  {
143  context->addObject(obj);
144  }
145 
146  if (arg)
147  {
148  obj->parse(arg);
149  }
150 
151  return obj;
152  }
