106  void propagateConnexionNumberToNeighbors( int connexionNumber, const type::vector<Connexion*>& allFineConnexions )
107  {
108  if (_tmp!=-1) return; // already in an existing connexion number
109 
110  _tmp = connexionNumber;
111  for(int i=0; i<NUM_CONNECTED_NODES; ++i)
112  for(std::set<Connexion*>::iterator n = _neighbors[i].begin(); n!=_neighbors[i].end(); ++n)
113  if( find( allFineConnexions.begin(),allFineConnexions.end(),*n ) != allFineConnexions.end() ) // the neighbors is in the good child of the coarse hexa
114  (*n)->propagateConnexionNumberToNeighbors(connexionNumber,allFineConnexions);
115  }
116  };
