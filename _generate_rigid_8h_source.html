37 template<typename Rigid3MassType>
38 void generateRigid(Rigid3MassType& mass, type::Vec3& center, const helper::io::Mesh* mesh )
39 {
40  using namespace sofa::helper;
41  // Geometric Tools, Inc.
42  // http://www.geometrictools.com
43  // Copyright (c) 1998-2006. All Rights Reserved
44  //
45  // The Wild Magic Library (WM3) source code is supplied under the terms of
46  // the license agreement
47  // http://www.geometrictools.com/License/WildMagic3License.pdf
48  // and may not be copied or disclosed except in accordance with the terms
49  // of that agreement.
50 
51  // order: 1, x, y, z, x^2, y^2, z^2, xy, yz, zx
52  SReal afIntegral[10] = { (SReal)0.0, (SReal)0.0, (SReal)0.0, (SReal)0.0,
53  (SReal)0.0, (SReal)0.0, (SReal)0.0, (SReal)0.0, (SReal)0.0, (SReal)0.0
54  };
55 
56  const auto& points = mesh->getVertices();
57  const auto& facets = mesh->getFacets();
58  for (unsigned int i = 0; i < facets.size(); i++)
59  {
60  const auto& v = facets[i][0];
61  for (unsigned int j = 2; j < v.size(); j++)
62  {
63  // get vertices of current triangle
64  const auto kV0 = points[v[0 ]];
65  const auto kV1 = points[v[j-1]];
66  const auto kV2 = points[v[j ]];
67 
68  // get cross product of edges
69  auto kV1mV0 = kV1 - kV0;
70  auto kV2mV0 = kV2 - kV0;
71  auto kN = cross(kV1mV0,kV2mV0);
72 
73  // compute integral terms
74  SReal fTmp0, fTmp1, fTmp2;
75  SReal fF1x, fF2x, fF3x, fG0x, fG1x, fG2x;
76  fTmp0 = kV0[0] + kV1[0];
77  fF1x = fTmp0 + kV2[0];
78  fTmp1 = kV0[0]*kV0[0];
79  fTmp2 = fTmp1 + kV1[0]*fTmp0;
80  fF2x = fTmp2 + kV2[0]*fF1x;
81  fF3x = kV0[0]*fTmp1 + kV1[0]*fTmp2 + kV2[0]*fF2x;
82  fG0x = fF2x + kV0[0]*(fF1x + kV0[0]);
83  fG1x = fF2x + kV1[0]*(fF1x + kV1[0]);
84  fG2x = fF2x + kV2[0]*(fF1x + kV2[0]);
85 
86  SReal fF1y, fF2y, fF3y, fG0y, fG1y, fG2y;
87  fTmp0 = kV0[1] + kV1[1];
88  fF1y = fTmp0 + kV2[1];
89  fTmp1 = kV0[1]*kV0[1];
90  fTmp2 = fTmp1 + kV1[1]*fTmp0;
91  fF2y = fTmp2 + kV2[1]*fF1y;
92  fF3y = kV0[1]*fTmp1 + kV1[1]*fTmp2 + kV2[1]*fF2y;
93  fG0y = fF2y + kV0[1]*(fF1y + kV0[1]);
94  fG1y = fF2y + kV1[1]*(fF1y + kV1[1]);
95  fG2y = fF2y + kV2[1]*(fF1y + kV2[1]);
96 
97  SReal fF1z, fF2z, fF3z, fG0z, fG1z, fG2z;
98  fTmp0 = kV0[2] + kV1[2];
99  fF1z = fTmp0 + kV2[2];
100  fTmp1 = kV0[2]*kV0[2];
101  fTmp2 = fTmp1 + kV1[2]*fTmp0;
102  fF2z = fTmp2 + kV2[2]*fF1z;
103  fF3z = kV0[2]*fTmp1 + kV1[2]*fTmp2 + kV2[2]*fF2z;
104  fG0z = fF2z + kV0[2]*(fF1z + kV0[2]);
105  fG1z = fF2z + kV1[2]*(fF1z + kV1[2]);
106  fG2z = fF2z + kV2[2]*(fF1z + kV2[2]);
107 
108  // update integrals
109  afIntegral[0] += kN[0]*fF1x;
110  afIntegral[1] += kN[0]*fF2x;
111  afIntegral[2] += kN[1]*fF2y;
112  afIntegral[3] += kN[2]*fF2z;
113  afIntegral[4] += kN[0]*fF3x;
114  afIntegral[5] += kN[1]*fF3y;
115  afIntegral[6] += kN[2]*fF3z;
116  afIntegral[7] += kN[0]*(kV0[1]*fG0x + kV1[1]*fG1x + kV2[1]*fG2x);
117  afIntegral[8] += kN[1]*(kV0[2]*fG0y + kV1[2]*fG1y + kV2[2]*fG2y);
118  afIntegral[9] += kN[2]*(kV0[0]*fG0z + kV1[0]*fG1z + kV2[0]*fG2z);
119  }
120  }
121 
122  afIntegral[0] /= (SReal)6.0;
123  afIntegral[1] /= (SReal)24.0;
124  afIntegral[2] /= (SReal)24.0;
125  afIntegral[3] /= (SReal)24.0;
126  afIntegral[4] /= (SReal)60.0;
127  afIntegral[5] /= (SReal)60.0;
128  afIntegral[6] /= (SReal)60.0;
129  afIntegral[7] /= (SReal)120.0;
130  afIntegral[8] /= (SReal)120.0;
131  afIntegral[9] /= (SReal)120.0;
132 
133  // mass
134  mass.volume = afIntegral[0];
135  mass.mass = mass.volume;
136 
137  // center of mass
138  center = {afIntegral[1]/afIntegral[0],afIntegral[2]/afIntegral[0],afIntegral[3]/afIntegral[0]};
139 
140  // inertia relative to world origin
141  mass.inertiaMatrix[0][0] = afIntegral[5] + afIntegral[6];
142  mass.inertiaMatrix[0][1] = -afIntegral[7];
143  mass.inertiaMatrix[0][2] = -afIntegral[9];
144  mass.inertiaMatrix[1][0] = mass.inertiaMatrix[0][1];
145  mass.inertiaMatrix[1][1] = afIntegral[4] + afIntegral[6];
146  mass.inertiaMatrix[1][2] = -afIntegral[8];
147  mass.inertiaMatrix[2][0] = mass.inertiaMatrix[0][2];
148  mass.inertiaMatrix[2][1] = mass.inertiaMatrix[1][2];
149  mass.inertiaMatrix[2][2] = afIntegral[4] + afIntegral[5];
150 
151  // inertia relative to center of mass
152  mass.inertiaMatrix[0][0] -= mass.mass*(center[1]*center[1] + center[2]*center[2]);
153  mass.inertiaMatrix[0][1] += mass.mass*center[0]*center[1];
154  mass.inertiaMatrix[0][2] += mass.mass*center[2]*center[0];
155  mass.inertiaMatrix[1][0] = mass.inertiaMatrix[0][1];
156  mass.inertiaMatrix[1][1] -= mass.mass*(center[2]*center[2] + center[0]*center[0]);
157  mass.inertiaMatrix[1][2] += mass.mass*center[1]*center[2];
158  mass.inertiaMatrix[2][0] = mass.inertiaMatrix[0][2];
159  mass.inertiaMatrix[2][1] = mass.inertiaMatrix[1][2];
160  mass.inertiaMatrix[2][2] -= mass.mass*(center[0]*center[0] + center[1]*center[1]);
161 
162  mass.inertiaMatrix /= mass.mass;
163 }

166 template<typename Rigid3MassType>
167 void generateRigid(Rigid3MassType& mass, type::Vec3& center, io::Mesh *mesh
168  , SReal density
169  , const type::Vec3& scale = type::Vec3(1,1,1)
170  , const type::Vec3& rotation /*Euler angles*/ = type::Vec3(0,0,0)
171  )
172 {
173  if( scale != type::Vec3(1, 1, 1) ) {
174  for(size_t i = 0, n = mesh->getVertices().size(); i < n; ++i) {
175  mesh->getVertices()[i] = mesh->getVertices()[i].linearProduct(scale);
176  }
177  }
178 
179  if( rotation != type::Vec3(0,0,0) ) {
180 
181  type::Quat q = type::Quat<SReal>::createQuaterFromEuler( rotation*M_PI/180.0 );
182 
183  for(size_t i = 0, n = mesh->getVertices().size(); i < n; ++i) {
184  mesh->getVertices()[i] = q.rotate( mesh->getVertices()[i] );
185  }
186  }
187 
188  generateRigid( mass, center, mesh );
189 
190  mass.recalc();
191 
192  mass.mass *= density;
193 
194 }

228 template<typename Rigid3MassType>
229 void generateRigid( GenerateRigidInfo& res
230  , io::Mesh *mesh
231  , std::string const& meshName
232  , SReal density
233  , const type::Vec3& scale = type::Vec3(1,1,1)
234  , const type::Vec3& rotation /*Euler angles*/ = type::Vec3(0,0,0)
235  )
236 
237 {
238  Rigid3MassType rigidMass;
239 
240  generateRigid( rigidMass, res.com, mesh, density, scale, rotation );
241 
242  if( rigidMass.mass < 0 )
243  {
244  msg_warning("generatedRigid")<<"are normals inverted? "<<meshName;
245  rigidMass.mass = -rigidMass.mass;
246  rigidMass.inertiaMatrix = -rigidMass.inertiaMatrix;
247  }
248 
249  res.mass = rigidMass.mass;
250  res.inertia = res.mass * rigidMass.inertiaMatrix;
251 
252  // a threshol to test if inertia is diagonal in function of diagonal values
253  SReal threshold = type::trace( res.inertia ) * 1e-6;
254 
255  // if not diagonal, extracting principal axes basis to get the corresponding rotation with a diagonal inertia
256  if( res.inertia[0][1]>threshold || res.inertia[0][2]>threshold || res.inertia[1][2]>threshold )
257  {
258  type::Matrix3 U;
259  Decompose<SReal>::eigenDecomposition_iterative( res.inertia, U, res.inertia_diagonal );
260 
261  // det should be 1->rotation or -1->reflexion
262  if( type::determinant( U ) < 0 ) // reflexion
263  {
264  // made it a rotation by negating a column
265  U[0][0] = -U[0][0];
266  U[1][0] = -U[1][0];
267  U[2][0] = -U[2][0];
268  }
269  res.inertia_rotation.fromMatrix( U );
270  }
271  else
272  {
273  res.inertia_diagonal[0] = res.inertia[0][0];
274  res.inertia_diagonal[1] = res.inertia[1][1];
275  res.inertia_diagonal[2] = res.inertia[2][2];
276  res.inertia_rotation.clear();
277  }
278 }
