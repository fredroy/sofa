46  template<typename Node,
47  typename T = std::decay_t<decltype(*std::begin(std::declval<Node>()))>,
48  typename = std::enable_if_t<std::is_scalar_v<T>>
49  >
50  [[nodiscard]]
51  static constexpr auto area(const Node& n0, const Node& n1, const Node& n2)
52  {
53  if constexpr (std::is_same_v < Node, sofa::type::Vec<3, T> >)
54  {
55  // half the area of a quadrilateral
56  const auto a = n1 - n0;
57  const auto b = n2 - n0;
58  return static_cast<T>(0.5) * sofa::type::cross(a, b).norm();
59  }
60  else if constexpr (std::is_same_v< Node, sofa::type::Vec<2, T> >)
61  {
62  // shoelace formula
63  return static_cast<T>(0.5) * (n0[0] * n1[1] + n1[0] * n2[1] + n2[0] * n0[1] - n1[0] * n0[1] - n2[0] * n1[1] - n0[0] * n2[1]);
64  }
65  else // generic without cross or diff
66  {
67  const auto a = sofa::geometry::Edge::length(n0, n1);
68  const auto b = sofa::geometry::Edge::length(n0, n2);
69  const auto c = sofa::geometry::Edge::length(n1, n2);
70  return static_cast<T>(0.25) * std::sqrt((a + b + c) * (-a + b + c) * (a - b + c) * (a + b - c));
71  }
72  }

122  [[nodiscard]]
123  static constexpr bool rayIntersection(const sofa::type::Vec<3, TReal>& n0, const sofa::type::Vec<3, TReal>& n1, const sofa::type::Vec<3, TReal>& n2, const sofa::type::Vec<3, TReal>& origin, const sofa::type::Vec<3, TReal>& direction, TReal& t, TReal& u, TReal& v)
124  {
125 
126  constexpr TReal epsilon = std::numeric_limits<TReal>::epsilon();
127  constexpr TReal zero = static_cast<TReal>(0);
128  constexpr TReal one = static_cast<TReal>(1);
129 
130  t = 0; u = 0; v = 0;
131 
132  const auto e0 = n1 - n0;
133  const auto e1 = n2 - n0;
134 
135  sofa::type::Vec<3, TReal> tvec(type::NOINIT);
136  sofa::type::Vec<3, TReal> pvec(type::NOINIT);
137  sofa::type::Vec<3, TReal> qvec(type::NOINIT);
138  TReal det;
139  TReal inv_det;
140 
141  pvec = sofa::type::cross(direction, e1);
142 
143  det = sofa::type::dot(e0, pvec);
144 
145  if (std::fabs(det) <= epsilon)
146  {
147  return false;
148  }
149 
150  inv_det = one / det;
151 
152  tvec = origin - n0;
153 
154  u = sofa::type::dot(tvec, pvec) * inv_det;
155  if (u < zero - epsilon || u > one + epsilon)
156  return false;
157 
158  qvec = sofa::type::cross(tvec, e0);
159 
160  v = sofa::type::dot(direction, qvec) * inv_det;
161  if (v < zero - epsilon || (u + v) > one + epsilon)
162  return false;
163 
164  t = sofa::type::dot(e1, qvec) * inv_det;
165 
166  if (t < epsilon || t != t || v != v || u != u)
167  return false;
168 
169  return true;
170  }
