105  template<class T>
106  static bool canCreate(T*& obj, objectmodel::BaseContext* context, objectmodel::BaseObjectDescription* arg)
107  {
108  MechanicalState<DataTypes>* mstate1 = nullptr;
109  MechanicalState<DataTypes>* mstate2 = nullptr;
110  std::string object1 = arg->getAttribute("object1","@./");
111  std::string object2 = arg->getAttribute("object2","@./");
112  if (object1.empty()) object1 = "@./";
113  if (object2.empty()) object2 = "@./";
114  context->findLinkDest(mstate1, object1, nullptr);
115  context->findLinkDest(mstate2, object2, nullptr);
116 
117  if (!mstate1) {
118  arg->logError("Data attribute 'object1' does not point to a valid mechanical state of datatype '" + std::string(DataTypes::Name()) + "'.");
119  return false;
120  }
121  if (!mstate2) {
122  arg->logError("Data attribute 'object2' does not point to a valid mechanical state of datatype '" + std::string(DataTypes::Name()) + "'.");
123  return false;
124  }
125 
126  return BaseInteractionConstraint::canCreate(obj, context, arg);
127  }
