22 /*
23  * vec.h -- Vector macros for 2,3, and 4 dimensions,
24  * for any combination of C scalar types.
25  *
26  * Author: Don Hatch (hatch@sgi.com)
27  * Last modified: Fri Sep 30 03:23:02 PDT 1994
28  *
29  * General description:
30  *
31  * The macro name describes its arguments; e.g.
32  * MXS3 is "matrix times scalar in 3 dimensions";
33  * VMV2 is "vector minus vector in 2 dimensions".
34  *
35  * If the result of an operation is a scalar, then the macro "returns"
36  * the value; e.g.
37  * result = DOT3(v,w);
38  * result = DET4(m);
39  *
40  * If the result of an operation is a vector or matrix, then
41  * the first argument is the destination; e.g.
42  * SET2(tovec, fromvec);
43  * MXM3(result, m1, m2);
44  *
45  * WARNING: For the operations that are not done "componentwise"
46  * (e.g. vector cross products and matrix multiplies)
47  * the destination should not be either of the arguments,
48  * for obvious reasons. For example, the following is wrong:
49  * VXM2(v,v,m);
50  * For such "unsafe" macros, there are safe versions provided,
51  * but you have to specify a type for the temporary
52  * result vector or matrix. For example, the safe versions
53  * of VXM2 are:
54  * VXM2d(v,v,m) if v's scalar type is double or float
55  * VXM2i(v,v,m) if v's scalar type is int or char
56  * VXM2l(v,v,m) if v's scalar type is long
57  * VXM2r(v,v,m) if v's scalar type is real
58  * VXM2safe(type,v,v,m) for other scalar types.
59  * These "safe" macros do not evaluate to C expressions
60  * (so, for example, they can't be used inside the parentheses of
61  * a for(...)).
62  *
63  * Specific descriptions:
64  *
65  * The "?"'s in the following can be 2, 3, or 4.
66  *
67  * SET?(to,from) to = from
68  * SETMAT?(to,from) to = from
69  * ROUNDVEC?(to,from) to = from with entries rounded
70  * to nearest integer
71  * ROUNDMAT?(to,from) to = from with entries rounded
72  * to nearest integer
73  * FILLVEC?(v,s) set each entry of vector v to be s
74  * FILLMAT?(m,s) set each entry of matrix m to be s
75  * ZEROVEC?(v) v = 0
76  * ISZEROVEC?(v) v == 0
77  * EQVEC?(v,w) v == w
78  * EQMAT?(m1,m2) m1 == m2
79  * ZEROMAT?(m) m = 0
80  * IDENTMAT?(m) m = 1
81  * TRANSPOSE?(to,from) (matrix to) = (transpose of matrix from)
82  * ADJOINT?(to,from) (matrix to) = (adjoint of matrix from)
83  * i.e. its determinant times its inverse
84  *
85  * V{P,M}V?(to,v,w) to = v {+,-} w
86  * M{P,M}M?(to,m1,m2) to = m1 {+,-} m2
87  * SX{V,M}?(to,s,from) to = s * from
88  * M{V,M}?(to,from) to = -from
89  * {V,M}{X,D}S?(to,from,s) to = from {*,/} s
90  * MXM?(to,m1,m2) to = m1 * m2
91  * VXM?(to,v,m) (row vec to) = (row vec v) * m
92  * MXV?(to,m,v) (column vec to) = m * (column vec v)
93  * LERP?(to,v0,v1,t) to = v0 + t*(v1-v0)
94  *
95  * DET?(m) determinant of m
96  * TRACE?(m) trace (sum of diagonal entries) of m
97  * DOT?(v,w) dot (scalar) product of v and w
98  * NORMSQRD?(v) square of |v|
99  * DISTSQRD?(v,w) square of |v-w|
100  *
101  * XV2(to,v) to = v rotated by 90 degrees
102  * VXV3(to,v1,v2) to = cross (vector) product of v1 and v2
103  * VXVXV4(to,v1,v2,v3) to = 4-dimensional vector cross product
104  * of v1,v2,v3 (a vector orthogonal to
105  * v1,v2,v3 whose length equals the
106  * volume of the spanned parallelotope)
107  * VXV2(v0,v1) determinant of matrix with rows v0,v1
108  * VXVXV3(v0,v1,v2) determinant of matrix with rows v0,v1,v2
109  * VXVXVXV4(v0,v1,v2,v3) determinant of matrix with rows v0,..,v3
110  *
111  * The following macros mix objects from different dimensions.
112  * For example, V3XM4 would be used to apply a composite
113  * 4x4 rotation-and-translation matrix to a 3d vector.
114  *
115  * SET3from2(to,from,pad) (3d vec to) = (2d vec from) with pad
116  * SET4from3(to,from,pad) (4d vec to) = (3d vec from) with pad
117  * SETMAT3from2(to,from,pad0,pad1) (3x3 mat to) = (2x2 mat from)
118  * padded with pad0 on the sides
119  * and pad1 in the corner
120  * SETMAT4from3(to,from,pad0,pad1) (4x4 mat to) = (3x3 mat from)
121  * padded with pad0 on the sides
122  * and pad1 in the corner
123  * V2XM3(to2,v2,m3) (2d row vec to2) = (2d row vec v2) * (3x3 mat m3)
124  * V3XM4(to3,v3,m4) (3d row vec to3) = (3d row vec v2) * (4x4 mat m4)
125  * M3XV2(to2,m3,v2) (2d col vec to2) = (3x3 mat m3) * (2d col vec v2)
126  * M4XV3(to3,m4,v3) (3d col vec to3) = (4x4 mat m4) * (3d col vec v3)
127  * M2XM3(to3,m2,m3) (3x3 mat to3) = (2x2 mat m2) * (3x3 mat m3)
128  * M3XM4(to4,m3,m4) (4x4 mat to4) = (3x3 mat m3) * (4x4 mat m4)
129  * M3XM2(to3,m3,m2) (3x3 mat to3) = (3x3 mat m3) * (2x2 mat m2)
130  * M4XM3(to4,m4,m3) (4x4 mat to4) = (4x4 mat m4) * (3x3 mat m3)
131  *
132  *
133  * This file is machine-generated and can be regenerated
134  * for any number of dimensions.
135  * The program that generated it is available upon request.
136  */
