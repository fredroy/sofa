207  virtual bool runTest( const InVecCoord& parentInit,
208  const OutVecCoord& childInit,
209  const InVecCoord& parentNew,
210  const OutVecCoord& expectedChildNew)
211  {
212  if( deltaRange.second / errorMax <= sofa::testing::g_minDeltaErrorRatio )
213  ADD_FAILURE() << "The comparison threshold is too large for the finite difference delta";
214 
215  if( !(flags & TEST_getJs) ) msg_warning("MappingTest") << "getJs is not tested";
216  if( !(flags & TEST_getK) ) msg_warning("MappingTest") << "getK is not tested";
217  if( !(flags & TEST_applyJT_matrix) ) msg_warning("MappingTest") << "applyJT on matrices is not tested";
218  if( !(flags & TEST_applyDJT) ) msg_warning("MappingTest") << "applyDJT is not tested";
219 
220 
221  const Real errorThreshold = this->epsilon()*errorMax;
222 
223  typedef linearalgebra::EigenSparseMatrix<In,Out> EigenSparseMatrix;
224  core::MechanicalParams mparams;
225  mparams.setKFactor(1.0);
226  mparams.setSymmetricMatrix(false);
227  inDofs->resize(parentInit.size());
228  WriteInVecCoord xin = inDofs->writePositions();
229  sofa::testing::copyToData(xin,parentInit); // xin = parentInit
230 
231  outDofs->resize(childInit.size());
232  WriteOutVecCoord xout = outDofs->writePositions();
233  sofa::testing::copyToData(xout,childInit);
234 
236  sofa::simulation::getSimulation()->init(root.get());
237 
239  sofa::testing::copyToData(xin,parentNew);
240  mapping->apply(&mparams, core::VecCoordId::position(), core::VecCoordId::position());
241  mapping->applyJ(&mparams, core::VecDerivId::velocity(), core::VecDerivId::velocity());
242 
244  bool succeed=true;
245 
246  if (expectedChildNew.size() != xout.size()) {
247  ADD_FAILURE() << "Size of output dofs is wrong: " << xout.size() << " expected: " << expectedChildNew.size();
248  succeed = false;
249  }
250  for( unsigned i=0; i<xout.size(); i++ )
251  {
252  if( !this->isSmall( difference(xout[i],expectedChildNew[i]).norm(), errorMax ) ) {
253  ADD_FAILURE() << "Position of mapped particle " << i << " is wrong: \n" << xout[i] <<"\nexpected: \n" << expectedChildNew[i]
254  << "\ndifference should be less than " << errorThreshold << " (" << difference(xout[i],expectedChildNew[i]).norm() << ")" << std::endl;
255  succeed = false;
256  }
257  }
258 
260  size_t Np = inDofs->getSize(), Nc=outDofs->getSize();
261 
262  InVecCoord xp(Np),xp1(Np);
263  InVecDeriv vp(Np),fp(Np),dfp(Np),fp2(Np);
264  OutVecCoord xc(Nc),xc1(Nc);
265  OutVecDeriv vc(Nc),fc(Nc);
266 
267  // get position data
268  sofa::testing::copyFromData( xp, inDofs->readPositions() );
269  sofa::testing::copyFromData( xc, outDofs->readPositions() ); // positions and have already been propagated
270 
271  // set random child forces and propagate them to the parent
272  for( unsigned i=0; i<Nc; i++ ){
273  fc[i] = Out::randomDeriv( 0.1, 1.0 );
274  }
275  fp2.fill( InDeriv() );
276  WriteInVecDeriv fin = inDofs->writeForces();
277  sofa::testing::copyToData( fin, fp2 ); // reset parent forces before accumulating child forces
278 
279  WriteOutVecDeriv fout = outDofs->writeForces();
280  sofa::testing::copyToData( fout, fc );
281  mapping->applyJT( &mparams, core::VecDerivId::force(), core::VecDerivId::force() );
282  sofa::testing::copyFromData( fp, inDofs->readForces() );
283 
284  // set small parent velocities and use them to update the child
285  for( unsigned i=0; i<Np; i++ ){
286  vp[i] = In::randomDeriv(this->epsilon() * deltaRange.first, this->epsilon() * deltaRange.second );
287  }
288 
289  for( unsigned i=0; i<Np; i++ ){ // and small displacements
290  xp1[i] = xp[i] + vp[i];
291  }
292 
293  // propagate small velocity
294  WriteInVecDeriv vin = inDofs->writeVelocities();
295  sofa::testing::copyToData( vin, vp );
296  mapping->applyJ( &mparams, core::VecDerivId::velocity(), core::VecDerivId::velocity() );
297  ReadOutVecDeriv vout = outDofs->readVelocities();
298  sofa::testing::copyFromData( vc, vout);
299 
300  // apply geometric stiffness
301  inDofs->vRealloc( &mparams, core::VecDerivId::dx() ); // dx is not allocated by default
302  WriteInVecDeriv dxin = inDofs->writeDx();
303  sofa::testing::copyToData( dxin, vp );
304  dfp.fill( InDeriv() );
305  sofa::testing::copyToData( fin, dfp );
306  mapping->updateK( &mparams, core::ConstVecDerivId::force() ); // updating stiffness matrix for the current state and force
307  mapping->applyDJT( &mparams, core::VecDerivId::force(), core::VecDerivId::force() );
308  sofa::testing::copyFromData( dfp, inDofs->readForces() ); // fp + df due to geometric stiffness
309 
310  // Jacobian will be obsolete after applying new positions
311  if( flags & TEST_getJs )
312  {
313  EigenSparseMatrix* J = this->getMatrix<EigenSparseMatrix>(mapping->getJs());
314  // cout<<"J = "<< endl << *J << endl;
315  OutVecDeriv Jv(Nc);
316  J->mult(Jv,vp);
317 
318  // ================ test applyJT()
319  InVecDeriv jfc( (long)Np,InDeriv());
320  J->addMultTranspose(jfc,fc);
321  if( this->vectorMaxDiff(jfc,fp)>errorThreshold ){
322  succeed = false;
323  ADD_FAILURE() << "applyJT test failed, difference should be less than " << errorThreshold << " (" << this->vectorMaxDiff(jfc,fp) << ")" << std::endl
324  << "jfc = " << jfc << std::endl<<" fp = " << fp << std::endl;
325  }
326  // ================ test getJs()
327  // check that J.vp = vc
328  if(this->vectorMaxDiff(Jv,vc)>errorThreshold ){
329  succeed = false;
330  std::cout<<"vp = " << vp << std::endl;
331  std::cout<<"Jvp = " << Jv << std::endl;
332  std::cout<<"vc = " << vc << std::endl;
333  ADD_FAILURE() << "getJs() test failed"<<std::endl<<"vp = " << vp << std::endl<<"Jvp = " << Jv << std::endl <<"vc = " << vc << std::endl;
334  }
335  }
336 
337  if( flags & TEST_applyJT_matrix )
338  {
339  // TODO test applyJT on matrices
340  // basic idea build a out random matrix e.g. with helper::drand(100.0)
341  // call applyJT on both this matrice and on all its lines (oe cols ?) one by one
342  // then compare results
343 
344 // OutMatrixDeriv outMatrices( ); // how to build that, what size?
345 // /*WriteInMatrixDeriv min = */inDofs->write( MatrixDerivId::constraintJacobian() );
346 // WriteOutMatrixDeriv mout = outDofs->write( MatrixDerivId::constraintJacobian() );
347 // copyToData(mout,outMatrices);
348 
349 // mapping->applyJt( ConstraintParams*, MatrixDerivId::constraintJacobian(), MatrixDerivId::constraintJacobian() );
350 
351 
352  }
353 
354  // compute parent forces from pre-treated child forces (in most cases, the pre-treatment does nothing)
355  // the pre-treatement can be useful to be able to compute 2 comparable results of applyJT with a small displacement to test applyDJT
356  fp.fill( InDeriv() );
357  sofa::testing::copyToData( fin, fp ); // reset parent forces before accumulating child forces
358  sofa::testing::copyToData( fout, preTreatment(fc) );
359  mapping->applyJT( &mparams, core::VecDerivId::force(), core::VecDerivId::force() );
360  sofa::testing::copyFromData( fp, inDofs->readForces() );
361 
362 
363 
365 
366 
367  // propagate small displacement
368  WriteInVecCoord pin (inDofs->writePositions());
369  sofa::testing::copyToData( pin, xp1 );
370 
371  mapping->apply ( &mparams, core::VecCoordId::position(), core::VecCoordId::position() );
372  ReadOutVecCoord pout = outDofs->readPositions();
373  sofa::testing::copyFromData( xc1, pout );
374 
375  // ================ test applyJ: compute the difference between propagated displacements and velocities
376  OutVecDeriv dxc(Nc);
377  for(unsigned i=0; i<Nc; i++ ){
378  dxc[i] = difference( xc1[i], xc[i] );
379  }
380 
381  if(this->vectorMaxAbs(difference(dxc,vc))>errorThreshold ){
382  succeed = false;
383  ADD_FAILURE() << "applyJ test failed: the difference between child position change and child velocity (dt=1) "<< this->vectorMaxAbs(difference(dxc,vc))<<" should be less than " << errorThreshold << std::endl
384  << "position change = " << dxc << std::endl
385  << "velocity = " << vc << std::endl;
386  }
387 
388 
389 
390  // update parent force based on the same child forces
391  fp2.fill( InDeriv() );
392  sofa::testing::copyToData( fin, fp2 ); // reset parent forces before accumulating child forces
393  sofa::testing::copyToData( fout, preTreatment(fc) );
394  mapping->applyJT( &mparams, core::VecDerivId::force(), core::VecDerivId::force() );
395  sofa::testing::copyFromData( fp2, inDofs->readForces() );
396 
397  InVecDeriv fp12(Np);
398  for(unsigned i=0; i<Np; i++){
399  fp12[i] = fp2[i]-fp[i]; // fp2 - fp
400  }
401 
402 
403  // ================ test applyDJT()
404  if( flags & TEST_applyDJT )
405  {
406  if(this->vectorMaxDiff(dfp,fp12)>errorThreshold*errorFactorDJ ){
407  succeed = false;
408  ADD_FAILURE() << "applyDJT test failed" << std::endl
409  << "dfp = " << dfp << std::endl
410  << "fp2-fp = " << fp12 << std::endl;
411  }
412  }
413 
414 
415  // ================ test getK()
416  if( flags & TEST_getK )
417  {
418  InVecDeriv Kv(Np);
419 
420  const linearalgebra::BaseMatrix* bk = mapping->getK();
421 
422  // K can be null or empty for linear mappings
423  // still performing the test with a null Kv vector to check if the mapping is really linear
424 
425  if( bk != nullptr ){
426 
427  typedef linearalgebra::EigenSparseMatrix<In,In> EigenSparseKMatrix;
428  const EigenSparseKMatrix* K = dynamic_cast<const EigenSparseKMatrix*>(bk);
429  if( K == nullptr ){
430  succeed = false;
431  ADD_FAILURE() << "getK returns a matrix of non-EigenSparseMatrix type";
432  // TODO perform a slow conversion with a big warning rather than a failure?
433  }
434 
435  if( K->compressedMatrix.nonZeros() ) K->mult(Kv,vp);
436  }
437 
438  // check that K.vp = dfp
439  if(this->vectorMaxDiff(Kv,fp12)>errorThreshold*errorFactorDJ ){
440  succeed = false;
441  ADD_FAILURE() << "K test failed, difference should be less than " << errorThreshold*errorFactorDJ << std::endl
442  << "Kv = " << Kv << std::endl
443  << "dfp = " << fp12 << std::endl;
444  }
445  }
446 
447  if(!succeed)
448  { ADD_FAILURE() << "Failed Seed number = " << this->seed << std::endl;}
449  return succeed;
450  }
