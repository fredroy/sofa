131 template<class DataTypes>
132 class ContactMapper<collision::geometry::TriangleCollisionModel<sofa::defaulttype::Vec3Types>, DataTypes> : public BarycentricContactMapper<collision::geometry::TriangleCollisionModel<sofa::defaulttype::Vec3Types>, DataTypes>
133 {
134 public:
135  typedef typename DataTypes::Real Real;
136  typedef typename DataTypes::Coord Coord;
137  using Index = sofa::Index;
138 
139  Index addPoint(const Coord& P, Index index, Real&)
140  {
141  auto nbt = this->model->getCollisionTopology()->getNbTriangles();
142  if (index < nbt)
143  return this->mapper->createPointInTriangle(P, index, &this->model->getMechanicalState()->read(core::ConstVecCoordId::position())->getValue());
144  else
145  {
146  Index qindex = (index - nbt)/2;
147  auto nbq = this->model->getCollisionTopology()->getNbQuads();
148  if (qindex < nbq)
149  return this->mapper->createPointInQuad(P, qindex, &this->model->getMechanicalState()->read(core::ConstVecCoordId::position())->getValue());
150  else
151  {
152  msg_error("ContactMapper<TriangleCollisionModel<sofa::defaulttype::Vec3Types>>") << "Invalid contact element index "<<index<<" on a topology with "<<nbt<<" triangles and "<<nbq<<" quads."<<msgendl
153  << "model="<<this->model->getName()<<" size="<<this->model->getSize() ;
154  return sofa::InvalidID;
155  }
156  }
157  }
158  Index addPointB(const Coord& P, Index index, Real& /*r*/, const type::Vec3& baryP)
159  {
160 
161  auto nbt = this->model->getCollisionTopology()->getNbTriangles();
162  if (index < nbt)
163  return this->mapper->addPointInTriangle(index, baryP.ptr());
164  else
165  {
166  // TODO: barycentric coordinates usage for quads
167  Index qindex = (index - nbt)/2;
168  auto nbq = this->model->getCollisionTopology()->getNbQuads();
169  if (qindex < nbq)
170  return this->mapper->createPointInQuad(P, qindex, &this->model->getMechanicalState()->read(core::ConstVecCoordId::position())->getValue());
171  else
172  {
173  msg_error("ContactMapper<TriangleCollisionModel<sofa::defaulttype::Vec3Types>>") << "Invalid contact element index "<<index<<" on a topology with "<<nbt<<" triangles and "<<nbq<<" quads."<<msgendl
174  << "model="<<this->model->getName()<<" size="<<this->model->getSize() ;
175  return sofa::InvalidID;
176  }
177  }
178  }
179 
180  inline Index addPointB(const Coord& P, Index index, Real& r ){return addPoint(P,index,r);}
181
