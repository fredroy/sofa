87  template<class T>
88  void createInputDataVector(unsigned int nb, type::vector< Data<T>* >& vf, std::string name, std::string help)
89  {
90  vf.reserve(nb);
91  for (unsigned int i=vf.size(); i<nb; ++i)
92  {
93  std::ostringstream oname, ohelp;
94  oname << name << (i+1);
95  ohelp << help << (i+1);
96  std::string name_i = oname.str();
97  std::string help_i = ohelp.str();
98  Data<T>* d = new Data<T>(sofa::helper::getAStringCopy(help_i.c_str()), true, false);
99  d->setName(name_i);
100  vf.push_back(d);
101  this->addData(d);
102  this->addInput(d);
103  }
104  }

115  template<class T>
116  void mergeInputDataVector(unsigned int nb, Data<T>& outF, const type::vector< Data<T>* >& inVF, const type::vector< Data<VecCoord>* >& inVFPos)
117  {
118  unsigned int nelems = 0;
119  for (unsigned int i=0; i<nb; ++i)
120  nelems += inVF[i]->getValue().size();
121  helper::WriteOnlyAccessor< Data<T> > out = outF;
122  out.clear();
123  out.reserve(nelems);
124  unsigned int shift = 0;
125  for (unsigned int i=0; i<nb; ++i)
126  {
127  helper::ReadAccessor< Data<T> > in = inVF[i];
128  for (unsigned int j=0; j<in.size(); ++j)
129  {
130  typename T::value_type outT = in[j];
131  for (unsigned int k=0; k<outT.size(); ++k)
132  outT[k] += shift;
133  out.push_back(outT);
134  }
135  shift += inVFPos[i]->getValue().size();
136  }
137  }
138  void mergeInputDataVector(unsigned int nb, Data<VecCoord>& outF, const type::vector< Data<VecCoord>* >& inVF)
139  {
140  unsigned int nelems = 0;
141  for (unsigned int i=0; i<nb; ++i)
142  nelems += inVF[i]->getValue().size();
143  helper::WriteOnlyAccessor< Data<VecCoord> > out = outF;
144  out.clear();
145  out.reserve(nelems);
146  for (unsigned int i=0; i<nb; ++i)
147  {
148  helper::ReadAccessor< Data<VecCoord> > in = inVF[i];
149  for (unsigned int j=0; j<in.size(); ++j)
150  out.push_back(in[j]);
151  }
152  }
