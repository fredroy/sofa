51  template<class Real>
52  class evaluator
53  {
54  public:
55  evaluator()
56  : map(nullptr), vmin(0), vmax(0), vscale(0)
57  {}
58 
59  evaluator(const ColorMap* map, Real vmin, Real vmax)
60  : map(map), vmin(vmin), vmax(vmax), vscale((vmax == vmin) ? (Real)0 : (map->entries.size()-1)/(vmax-vmin)) {}
61 
62  Color operator()(Real r) const
63  {
64  Real e = (r-vmin)*vscale;
65  if (e<0) return map->entries.front();
66 
67  unsigned int i = (unsigned int)(e);
68  if (i>=map->entries.size()-1) return map->entries.back();
69 
70  Color c1 = map->entries[i];
71  Color c2 = map->entries[i+1];
72  return c1+(c2-c1)*(e-i);
73  }
74  protected:
75  const ColorMap* map;
76  Real vmin;
77  Real vmax;
78  Real vscale;
79  };
