102 inline size_t lfactorial(size_t n)
103 {
104  size_t retval = 1;

109 template < class Real >
110 Real binomial(const size_t p, const size_t q) {
111  size_t ival=1;
112  size_t i;
113  if (p>q) {
114  for (i=p+q;i>p;--i){
115  ival*=i;
116  }
117  return((Real)ival)/lfactorial(q);
118  } else {
119  for (i=p+q;i>q;--i){
120  ival*=i;
121  }
122  return((Real)ival)/lfactorial(p);
123  }
124 }
125 template <class Real>
126 Real multinomial(const size_t n,type::vector<unsigned char> valArray)
127 {
128  size_t i,ival,N;
129  N=valArray.size();
130  // divide n! with the largest of the multinomial coefficient
131  std::sort(valArray.begin(),valArray.end());
132  ival=1;
133  for (i=n;i>valArray[N-1];--i){
134  ival*=i;
135  }
136  Real val=1;
137  for (i=0;i<(N-1);++i)
138  val*=lfactorial(valArray[i]);
139  return((Real)ival)/(val);
140 }

151 template <size_t N, class Real>
152 Real multinomialVector(const sofa::type::vector< sofa::type::Vec<N,unsigned char> > tbiArray)
153 {
154  size_t i,j;
155  Real result=(Real)1;
156  type::vector<unsigned char> valArray;
157  size_t totalDegree;
158  for (j=0;j<N;++j) {
159  valArray.clear();
160  totalDegree=0;
161  for (i=0;i<tbiArray.size();++i)
162  {
163  valArray.push_back(tbiArray[i][j]);
164  totalDegree+=tbiArray[i][j];
165  }
166  result*=multinomial<Real>(totalDegree,valArray);
167  }
168  return(result);
169 }
