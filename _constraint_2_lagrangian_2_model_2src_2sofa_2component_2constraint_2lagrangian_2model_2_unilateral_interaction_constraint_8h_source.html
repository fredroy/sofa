54 // A little experiment on how to best save the forces for the hot start.
55 // TODO : save as a map (index of the contact <-> force)
56 class PreviousForcesContainer
57 {
58 public:
59  PreviousForcesContainer() : resetFlag(true) {}
60  double popForce()
61  {
62  resetFlag = true;
63  if(forces.empty()) return 0;
64  double f = forces.front();
65  forces.pop_front();
66  return f;
67  }
68 
69  void pushForce(double f)
70  {
71  if(resetFlag)
72  {
73  forces.clear();
74  resetFlag = false;
75  }
76 
77  forces.push_back(f);
78  }
79 
80 protected:
81  std::deque<double> forces;
82  bool resetFlag; // We delete all forces that were not read
83 };
