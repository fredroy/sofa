56 template <class T1, class T2>
57 T1 pythag(const T1 a, const T2 b)
58 {
59  T1 absa,absb;
60  absa=fabs(a);
61  absb=fabs(b);
62  if (absa > absb) return absa*sqrt(1.0+SQR(absb/absa));
63  else return (absb == 0.0 ? 0.0 : absb*sqrt(1.0+SQR(absa/absb)));
64 }

72 template< int m, int n, typename Real>
73 void svddcmp(Mat<m,n,Real> &a, Vec<n,Real> &w, Mat<n,m,Real> &v)
74 {
75  bool flag;
76  int i,its,j,jj,k,l,nm;
77  Real anorm,c,f,g,h,s,scale,x,y,z;
78 
79  Vec<n,Real> rv1;
80  g=scale=anorm=0.0;
81  for (i=0; i<n; i++)
82  {
83  l=i+2;
84  rv1[i]=scale*g;
85  g=s=scale=0.0;
86  if (i < m)
87  {
88  for (k=i; k<m; k++) scale += fabs(a[k][i]);
89  if (scale != 0.0)
90  {
91  for (k=i; k<m; k++)
92  {
93  a[k][i] /= scale;
94  s += a[k][i]*a[k][i];
95  }
96  f=a[i][i];
97  g = -S_SIGN(sqrt(s),f);
98  h=f*g-s;
99  a[i][i]=f-g;
100  for (j=l-1; j<n; j++)
101  {
102  for (s=0.0,k=i; k<m; k++) s += a[k][i]*a[k][j];
103  f=s/h;
104  for (k=i; k<m; k++) a[k][j] += f*a[k][i];
105  }
106  for (k=i; k<m; k++) a[k][i] *= scale;
107  }
108  }
109  w[i]=scale *g;
110  g=s=scale=0.0;
111  if (i+1 <= m && i != n)
112  {
113  for (k=l-1; k<n; k++) scale += fabs(a[i][k]);
114  if (scale != 0.0)
115  {
116  for (k=l-1; k<n; k++)
117  {
118  a[i][k] /= scale;
119  s += a[i][k]*a[i][k];
120  }
121  f=a[i][l-1];
122  g = -S_SIGN(sqrt(s),f);
123  h=f*g-s;
124  a[i][l-1]=f-g;
125  for (k=l-1; k<n; k++) rv1[k]=a[i][k]/h;
126  for (j=l-1; j<m; j++)
127  {
128  for (s=0.0,k=l-1; k<n; k++) s += a[j][k]*a[i][k];
129  for (k=l-1; k<n; k++) a[j][k] += s*rv1[k];
130  }
131  for (k=l-1; k<n; k++) a[i][k] *= scale;
132  }
133  }
134  anorm=S_MAX(anorm,(fabs(w[i])+fabs(rv1[i])));
135  }
136  for (i=n-1; i>=0; i--)
137  {
138  if (i < n-1)
139  {
140  if (g != 0.0)
141  {
142  for (j=l; j<n; j++)
143  v[j][i]=(a[i][j]/a[i][l])/g;
144  for (j=l; j<n; j++)
145  {
146  for (s=0.0,k=l; k<n; k++) s += a[i][k]*v[k][j];
147  for (k=l; k<n; k++) v[k][j] += s*v[k][i];
148  }
149  }
150  for (j=l; j<n; j++) v[i][j]=v[j][i]=0.0;
151  }
152  v[i][i]=1.0;
153  g=rv1[i];
154  l=i;
155  }
156  for (i=S_MIN(m,n)-1; i>=0; i--)
157  {
158  l=i+1;
159  g=w[i];
160  for (j=l; j<n; j++) a[i][j]=0.0;
161  if (g != 0.0)
162  {
163  g=1.0/g;
164  for (j=l; j<n; j++)
165  {
166  for (s=0.0,k=l; k<m; k++) s += a[k][i]*a[k][j];
167  f=(s/a[i][i])*g;
168  for (k=i; k<m; k++) a[k][j] += f*a[k][i];
169  }
170  for (j=i; j<m; j++) a[j][i] *= g;
171  }
172  else for (j=i; j<m; j++) a[j][i]=0.0;
173  ++a[i][i];
174  }
175  for (k=n-1; k>=0; k--)
176  {
177  for (its=0; its<30; its++)
178  {
179  flag=true;
180  for (l=k; l>=0; l--)
181  {
182  nm=l-1;
183  if (fabs(rv1[l])+anorm == anorm)
184  {
185  flag=false;
186  break;
187  }
188  if (fabs(w[nm])+anorm == anorm) break;
189  }
190  if (flag)
191  {
192  c=0.0;
193  s=1.0;
194  for (i=l-1; i<k+1; i++)
195  {
196  f=s*rv1[i];
197  rv1[i]=c*rv1[i];
198  if (fabs(f)+anorm == anorm) break;
199  g=w[i];
200  h=pythag(f,g);
201  w[i]=h;
202  h=1.0/h;
203  c=g*h;
204  s = -f*h;
205  for (j=0; j<m; j++)
206  {
207  y=a[j][nm];
208  z=a[j][i];
209  a[j][nm]=y*c+z*s;
210  a[j][i]=z*c-y*s;
211  }
212  }
213  }
214  z=w[k];
215  if (l == k)
216  {
217  if (z < 0.0)
218  {
219  w[k] = -z;
220  for (j=0; j<n; j++) v[j][k] = -v[j][k];
221  }
222  break;
223  }
224  if (its == 29)
225  {
226 // msg_info()<<"Warning: Mat.h :: svddcmp: no convergence in 30 svdcmp iterations"<<std::endl;
227  return;
228  }
229  x=w[l];
230  nm=k-1;
231  y=w[nm];
232  g=rv1[nm];
233  h=rv1[k];
234  f=((y-z)*(y+z)+(g-h)*(g+h))/(2.0*h*y);
235  g=pythag(f,1.0);
236  f=((x-z)*(x+z)+h*((y/(f+S_SIGN(g,f)))-h))/x;
237  c=s=1.0;
238  for (j=l; j<=nm; j++)
239  {
240  i=j+1;
241  g=rv1[i];
242  y=w[i];
243  h=s*g;
244  g=c*g;
245  z=pythag(f,h);
246  rv1[j]=z;
247  c=f/z;
248  s=h/z;
249  f=x*c+g*s;
250  g=g*c-x*s;
251  h=y*s;
252  y *= c;
253  for (jj=0; jj<n; jj++)
254  {
255  x=v[jj][j];
256  z=v[jj][i];
257  v[jj][j]=x*c+z*s;
258  v[jj][i]=z*c-x*s;
259  }
260  z=pythag(f,h);
261  w[j]=z;
262  if (z)
263  {
264  z=1.0/z;
265  c=f*z;
266  s=h*z;
267  }
268  f=c*g+s*y;
269  x=c*y-s*g;
270  for (jj=0; jj<m; jj++)
271  {
272  y=a[jj][j];
273  z=a[jj][i];
274  a[jj][j]=y*c+z*s;
275  a[jj][i]=z*c-y*s;
276  }
277  }
278  rv1[l]=0.0;
279  rv1[k]=f;
280  w[k]=x;
281  }
282  }
283 }
