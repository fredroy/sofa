436 struct ActivationFunctor
437 {
438  ActivationFunctor(bool act, GraphListenerQListView* l):active(act), listener(l)
439  {
440  pixmap_filename= std::string("textures/media-record.png");
441  if ( sofa::helper::system::DataRepository.findFile ( pixmap_filename ) )
442  pixmap_filename = sofa::helper::system::DataRepository.getFile ( pixmap_filename );
443  }
444  void operator()(core::objectmodel::BaseNode* n)
445  {
446  if (active)
447  {
448  //Find the corresponding node in the Qt Graph
449  QTreeWidgetItem *item=listener->items[n];
450  //Remove the text
451  QString desact_text = item->text(0);
452  desact_text.remove(QString("Deactivated "), Qt::CaseInsensitive);
453  item->setText(0,desact_text);
454  //Remove the icon
455  QPixmap *p = getPixmap(n, false,false, false);
456  item->setIcon(0, QIcon(*p));
457 // item->setOpen(true);
458  item->setExpanded(true);
459  }
460  else
461  {
462  //Find the corresponding node in the Qt Graph
463  QTreeWidgetItem *item=listener->items[n];
464  //Remove the text
465  item->setText(0, QString("Deactivated ") + item->text(0));
466  item->setIcon(0, QIcon(QPixmap::fromImage(QImage(pixmap_filename.c_str()))));
467 
468  item->setExpanded(false);
469  }
470  }
471 protected:
472  std::string pixmap_filename;
473  bool active;
474  GraphListenerQListView* listener;
475 };
