76 template <class DataTypes1, class DataTypes2>
77 void PairStateAccessor<DataTypes1, DataTypes2>::init()
78 {
79  Inherit1::init();
80 
81  if (!mstate1.get())
82  {
83  mstate1.set(dynamic_cast< MechanicalState<DataTypes1>* >(getContext()->getMechanicalState()));
84 
85  msg_error_when(!mstate1) << "MechanicalState #1 (" << mstate1.getName() << "): No compatible MechanicalState "
86  "found in the current context. "
87  "This may be because there is no MechanicalState in the local context, "
88  "or because the type is not compatible.";
89  }
90 
91  if (!mstate2.get())
92  {
93  mstate2.set(dynamic_cast< MechanicalState<DataTypes2>* >(getContext()->getMechanicalState()));
94 
95  msg_error_when(!mstate2) << "MechanicalState #2 (" << mstate2.getName() << "): No compatible MechanicalState "
96  "found in the current context. "
97  "This may be because there is no MechanicalState in the local context, "
98  "or because the type is not compatible.";
99  }
100 
101  l_mechanicalStates.clear();
102  l_mechanicalStates.add(mstate1);
103  l_mechanicalStates.add(mstate2);
104 }
