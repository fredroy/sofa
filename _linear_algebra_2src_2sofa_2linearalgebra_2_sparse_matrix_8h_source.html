492  {
493  for (typename SparseMatrix<R1>::LineConstIterator itl = m1.begin(), itlend=m1.end(); itl!=itlend; ++itl)

496  for (typename SparseMatrix<R1>::LElementConstIterator ite = itl->second.begin(), iteend=itl->second.end(); ite!=iteend; ++ite)
497  {
498  const int c1 = ite->first;
499  R1 v = ite->second;
500  const typename SparseMatrix<R2>::Line& m2l = m2[c1];
501  for (typename SparseMatrix<R2>::LElementConstIterator ite2 = m2l.begin(), ite2end=m2l.end(); ite2!=ite2end; ++ite2)
502  {
503  R2 v2 = ite2->second;

505  d->add(l1, c2, (v*v2));

507  }
