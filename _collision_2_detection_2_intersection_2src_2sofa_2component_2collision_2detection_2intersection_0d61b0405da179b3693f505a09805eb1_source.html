67 template <class T>
68 bool MeshMinProximityIntersection::testIntersection(collision::geometry::Triangle& e2, collision::geometry::TSphere<T>& e1)
69 {
70  const SReal alarmDist = intersection->getAlarmDistance() + e1.r() + e1.getProximity() + e2.getProximity();
71 
72  const type::Vector3 x13 = e2.p1()-e2.p2();
73  const type::Vector3 x23 = e2.p1()-e2.p3();
74  const type::Vector3 x03 = e2.p1()-e1.center();
75  type::Matrix2 A;
76  type::Vector2 b;
77  A[0][0] = x13*x13;
78  A[1][1] = x23*x23;
79  A[0][1] = A[1][0] = x13*x23;
80  b[0] = x13*x03;
81  b[1] = x23*x03;
82  const SReal det = type::determinant(A);
83 
84  SReal alpha = 0.5;
85  SReal beta = 0.5;
86 
87  //if (det < -0.000001 || det > 0.000001)
88  {
89  alpha = (b[0]*A[1][1] - b[1]*A[0][1])/det;
90  beta = (b[1]*A[0][0] - b[0]*A[1][0])/det;
91  if (alpha < 0.000001 ||
92  beta < 0.000001 ||
93  alpha + beta > 0.999999)
94  return false;
95  }
96 
97  type::Vector3 P,Q,PQ;
98  P = e1.center();
99  Q = e2.p1() - x13 * alpha - x23 * beta;
100  PQ = Q-P;
101 
102  if (PQ.norm2() < alarmDist*alarmDist)
103  {
104  return true;
105  }
106  else
107  return false;
108 }

111 int MeshMinProximityIntersection::computeIntersection(collision::geometry::Triangle& e2, collision::geometry::TSphere<T>& e1, OutputVector* contacts)
112 {
113  const SReal alarmDist = intersection->getAlarmDistance() + e1.r() + e1.getProximity() + e2.getProximity();
114 
115  const type::Vector3 x13 = e2.p1()-e2.p2();
116  const type::Vector3 x23 = e2.p1()-e2.p3();
117  const type::Vector3 x03 = e2.p1()-e1.center();
118  type::Matrix2 A;
119  type::Vector2 b;
120  A[0][0] = x13*x13;
121  A[1][1] = x23*x23;
122  A[0][1] = A[1][0] = x13*x23;
123  b[0] = x13*x03;
124  b[1] = x23*x03;
125  const SReal det = type::determinant(A);
126 
127  SReal alpha = 0.5;
128  SReal beta = 0.5;
129 
130  //if (det < -0.000001 || det > 0.000001)
131  {
132  alpha = (b[0]*A[1][1] - b[1]*A[0][1])/det;
133  beta = (b[1]*A[0][0] - b[0]*A[1][0])/det;
134  if (alpha < 0.000001 ||
135  beta < 0.000001 ||
136  alpha + beta > 0.999999)
137  return 0;
138  }
139 
140  type::Vector3 P = e1.center();
141  type::Vector3 Q = e2.p1() - x13 * alpha - x23 * beta;
142  type::Vector3 QP = P-Q;
143  //Vector3 PQ = Q-P;
144 
145  if (QP.norm2() >= alarmDist*alarmDist)
146  return 0;
147 
148  const SReal contactDist = intersection->getContactDistance() + e1.r() + e1.getProximity() + e2.getProximity();
149 
150  contacts->resize(contacts->size()+1);
151  sofa::core::collision::DetectionOutput *detection = &*(contacts->end()-1);
152  detection->elem = std::pair<core::CollisionElementIterator, core::CollisionElementIterator>(e2, e1);
153  detection->id = e1.getIndex();
154  detection->normal=QP;
155  detection->value = detection->normal.norm();
156  if(detection->value>1e-15)
157  {
158  detection->normal /= detection->value;
159  }
160  else
161  {
162  msg_warning(intersection) << "Null distance between contact detected";
163  detection->normal= type::Vector3(1,0,0);
164  }
165  detection->value -= contactDist;
166  detection->point[0]=Q;
167  detection->point[1]=e1.getContactPointByNormal( detection->normal );
168  return 1;
169 }
