122  template<class T>
123  static bool canCreate(T* obj, objectmodel::BaseContext* context, objectmodel::BaseObjectDescription* arg)
124  {
125  MechanicalState<DataTypes>* mstate1 = nullptr;
126  MechanicalState<DataTypes>* mstate2 = nullptr;
127  std::string object1 = arg->getAttribute("object1","@./");
128  std::string object2 = arg->getAttribute("object2","@./");
129  if (object1.empty()) object1 = "@./";
130  if (object2.empty()) object2 = "@./";
131  context->findLinkDest(mstate1, object1, nullptr);
132  context->findLinkDest(mstate2, object2, nullptr);
133 
134  if (!mstate1) {
135  arg->logError("Data attribute 'object1' does not point to a valid mechanical state of datatype '" + std::string(DataTypes::Name()) + "'.");
136  return false;
137  }
138  if (!mstate2) {
139  arg->logError("Data attribute 'object2' does not point to a valid mechanical state of datatype '" + std::string(DataTypes::Name()) + "'.");
140  return false;
141  }
142 
143  return BaseInteractionProjectiveConstraintSet::canCreate(obj, context, arg);
144  }
