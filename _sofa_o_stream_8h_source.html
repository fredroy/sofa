31 // The methods in SofaOStream have been deprecated, but we still want it to work.
32 // Its usage in Base.h would generate warnings.
33 // That is why the deprecation is set between SOFA_BUILD_SOFA_CORE macro, to disable the warnings only when building Sofa.Core.
34 // Warning messages will appear when used outside of Sofa.Core (logically in the components)
35 
36 // There is also a check on the version of MSVC2017
37 // as there is a bug when applying [[deprecated]] on a friend method.
38 // (and does not choke when using the msvc keyword __declspec(deprecated("xxx"))
39 
40 #ifndef SOFA_BUILD_SOFA_CORE
41 #if defined(_MSC_VER) && (_MSC_VER < 1920) // if less than VS 2019
42 #define SOFA_ATTRIBUTE_DEPRECATED__SOFAOSTREAM__OVERRIDE() \
43  __declspec(deprecated(\
44  "It is still usable but has been DEPRECATED since v21.12 (PR#2292). " \
45  "You have until v22.06 to fix your code. " \
46  "Use the Messaging API instead of using SofaOStream and sout/serr/sendl."))
47 #else // _MSC_VER
48 #define SOFA_ATTRIBUTE_DEPRECATED__SOFAOSTREAM__OVERRIDE() \
49  SOFA_ATTRIBUTE_DEPRECATED__SOFAOSTREAM()
50 #endif // _MSC_VER
51 #else // SOFA_BUILD_SOFA_CORE
52 #define SOFA_ATTRIBUTE_DEPRECATED__SOFAOSTREAM__OVERRIDE() // dont do anything if we are building Sofa.Core
53 #endif // SOFA_BUILD_SOFA_CORE
