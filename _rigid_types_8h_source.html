407  void operator +=(const Deriv& dg) {
408  // R3 x SO(3) exponential integration
409  center += dg.getVCenter();
410 
411  const Vec3 omega = dg.getVOrientation() / 2;
412  const real theta = omega.norm();
413 
414  static const real epsilon = std::numeric_limits<real>::epsilon();
415 
416  if( theta < epsilon ) {
417  // fallback to gnomonic projection
418  Quat exp(omega[0], omega[1], omega[2], 1);
419  exp.normalize();
420  orientation = exp * orientation;
421  } else {
422  // expontential
423  const real sinc = std::sin(theta) / theta;
424  const Quat exp(sinc * omega[0],
425  sinc * omega[1],
426  sinc * omega[2],
427  std::cos(theta));
428  orientation = exp * orientation;
429  }
