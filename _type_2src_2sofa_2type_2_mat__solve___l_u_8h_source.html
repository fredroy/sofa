30 template< int n, typename Real>
31 void ludcmp(Mat<n,n,Real> &a, Vec<n,int> &indx)
32 {
33  const Real TINY=(Real)1.0e-20;
34  int i,imax=0,j,k;
35  Real big,dum,sum,temp;
36 
37  Vec<n,Real> vv;
38  for (i=0; i<n; i++)
39  {
40  big=0.0;
41  for (j=0; j<n; j++)
42  if ((temp=fabs(a[i][j])) > big) big=temp;
43  assert (big != 0.0);
44  vv[i]=(Real)1.0/big;
45  }
46  for (j=0; j<n; j++)
47  {
48  for (i=0; i<j; i++)
49  {
50  sum=a[i][j];
51  for (k=0; k<i; k++) sum -= a[i][k]*a[k][j];
52  a[i][j]=sum;
53  }
54  big=0.0;
55  for (i=j; i<n; i++)
56  {
57  sum=a[i][j];
58  for (k=0; k<j; k++) sum -= a[i][k]*a[k][j];
59  a[i][j]=sum;
60  if ((dum=vv[i]*fabs(sum)) >= big)
61  {
62  big=dum;
63  imax=i;
64  }
65  }
66  if (j != imax)
67  {
68  for (k=0; k<n; k++)
69  {
70  dum=a[imax][k];
71  a[imax][k]=a[j][k];
72  a[j][k]=dum;
73  }
74  vv[imax]=vv[j];
75  }
76  indx[j]=imax;
77  if (a[j][j] == 0.0) a[j][j]=TINY;
78  if (j != n-1)
79  {
80  dum=(Real)1.0/(a[j][j]);
81  for (i=j+1; i<n; i++) a[i][j] *= dum;
82  }
83  }
84 }

87 template< int n, typename Real>
88 void lubksb(const Mat<n,n,Real> &a, const Vec<n,int> &indx, Vec<n,Real> &b)
89 {
90  int i,ii=0,ip,j;
91  Real sum;
92 
93  for (i=0; i<n; i++)
94  {
95  ip=indx[i];
96  sum=b[ip];
97  b[ip]=b[i];
98  if (ii != 0)
99  for (j=ii-1; j<i; j++) sum -= a[i][j]*b[j];
100  else if (sum != 0.0)
101  ii=i+1;
102  b[i]=sum;
103  }
104  for (i=n-1; i>=0; i--)
105  {
106  sum=b[i];
107  for (j=i+1; j<n; j++) sum -= a[i][j]*b[j];
108  b[i]=sum/a[i][i];
109  }
110 }
