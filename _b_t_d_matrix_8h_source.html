194  template<class Real2>
195  FullVector<Real2> operator*(const FullVector<Real2>& v) const
196  {
197  FullVector<Real2> res(rowSize());
198  for (Index bi=0; bi<nBRow; ++bi)
199  {
200  Index b0 = (bi > 0) ? 0 : 1;
201  Index b1 = ((bi < nBRow - 1) ? 3 : 2);
202  for (Index i=0; i<BSIZE; ++i)
203  {
204  Real r = 0;
205  for (Index bj = b0; bj < b1; ++bj)
206  {
207  for (Index j=0; j<BSIZE; ++j)
208  {
209  r += data[bi*3+bj][i][j] * v[(bi + bj - 1)*BSIZE + j];
210  }
211  }
212  res[bi*BSIZE + i] = r;
213  }
214  }
215  return res;
216  }
