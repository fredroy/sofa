110  template<class Dispatcher>
111  void apply(Dispatcher& dispatch, sofa::linearalgebra::BaseMatrix *m, unsigned int offsetL, unsigned int offsetC)
112  {
113  if ((offsetL % NL) == 0 && (offsetC % NC) == 0 && m->getBlockRows() == NL && m->getBlockCols() == NC)
114  {
115  unsigned int boffsetL = offsetL / NL;
116  unsigned int boffsetC = offsetC / NC;
117  if (sofa::linearalgebra::CompressedRowSparseMatrix<type::Mat<NL,NC,double> > * mat = dynamic_cast<sofa::linearalgebra::CompressedRowSparseMatrix<type::Mat<NL,NC,double> > * >(m))
118  {
119  dispatch(BlocCRSMatrixWriter<double>(mat, boffsetL, boffsetC));
120  }
121  else if (sofa::linearalgebra::CompressedRowSparseMatrix<type::Mat<NL,NC,float> > * mat = dynamic_cast<sofa::linearalgebra::CompressedRowSparseMatrix<type::Mat<NL,NC,float> > * >(m))
122  {
123  dispatch(BlocCRSMatrixWriter<float>(mat, boffsetL, boffsetC));
124  }
125  else
126  {
127  dispatch(BlocBaseMatrixWriter(m, boffsetL, boffsetC));
128  }
129  }
130  else
131  {

133  {
134  dispatch(CRSMatrixWriter<double>(mat, offsetL, offsetC));
135  }
136  else if (sofa::linearalgebra::CompressedRowSparseMatrix<float> * mat = dynamic_cast<sofa::linearalgebra::CompressedRowSparseMatrix<float> * >(m))
137  {
138  dispatch(CRSMatrixWriter<float>(mat, offsetL, offsetC));
139  }
140  else
141  {
142  dispatch(BaseMatrixWriter(m, offsetL, offsetC));
143  }
144  }
