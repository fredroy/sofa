202 template<class BaseClass>
203 void ObjectFactory::getEntriesDerivedFrom(std::vector<ClassEntry::SPtr>& result) const
204 {
205  result.clear();
206  for (const auto& r : registry)
207  {
208  ClassEntry::SPtr entry = r.second;
209  // Push the entry only if it is not an alias
210  if (entry->className == r.first)
211  {
212  const auto creatorEntry = entry->creatorMap.begin();
213  if (creatorEntry != entry->creatorMap.end())
214  {
215  const auto* baseClass = creatorEntry->second->getClass();
216  if (baseClass && baseClass->hasParent(BaseClass::GetClass()))
217  {
218  result.push_back(entry);
219  }
220  }
221  }
222  }
223 }

225 template<class BaseClass>
226 std::string ObjectFactory::listClassesDerivedFrom(const std::string& separator) const
227 {
228  std::vector<ClassEntry::SPtr> entries;
229  getEntriesDerivedFrom<BaseClass>(entries);
230  if (entries.empty()) return std::string();
231 
232  const auto join = [&separator](std::string a, ClassEntry::SPtr b)
233  {
234  return std::move(a) + separator + b->className;
235  };
236  return std::accumulate(std::next(entries.begin()), entries.end(),
237  entries.front()->className, join);
238 }
