96  inline friend std::istream& operator >> ( std::istream& in, Spring& s )
97  {
98  //default joint is a free rotation joint --> translation is bloqued, rotation is free
99 
100  std::string str;
101  in>>str;
102  if ( str == "BEGIN_SPRING" )
103  {
104  in>>s.m1>>s.m2; //read references
105  in>>str;
106  while ( str != "END_SPRING" )
107  {
108  if ( str == "KS_T" )
109  in>>s.stiffnessTrans;
110  else if ( str == "KS_R" )
111  in>>s.stiffnessRot;
112  else if ( str == "KD" )
113  in>>s.kd;
114  else if ( str == "VEC1" )
115  in>>s.vec1;
116  else if ( str == "VEC2" )
117  in>>s.vec2;
118  else
119  {
120  msg_error("FrameSpringForceField")<<" Error parsing Spring : Unknown Attribute '"<<str << "'";
121  return in;
122  }
123 
124  in>>str;
125  }
126  }
127 
128  return in;
129  }
