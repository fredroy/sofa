71 template<class T, int N = struct_data_trait<T>::NVAR >
72 class struct_data_widget_container
73 {
74 public:
75  typedef T data_type;
76  typedef struct_data_trait<data_type> shelper;
77  typedef struct_data_trait_var<data_type,N-1> vhelper;
78  typedef typename vhelper::value_type value_type;
79  typedef data_widget_container<value_type> Container;
80  typedef struct_data_widget_container<data_type,N-1> PrevContainer;
81  typedef QVBoxLayout MasterLayout;
82  typedef QHBoxLayout Layout;
83  PrevContainer p;
84  Container w;
85  QCheckBox* check;
86  QLabel* label;
87  Layout* container_layout;
88  MasterLayout* master_layout;
89  struct_data_widget_container() : check(nullptr),label(nullptr),container_layout(nullptr),master_layout(nullptr) {}
90 
91  void setMasterLayout(MasterLayout* layout)
92  {
93  p.setMasterLayout(layout);
94  master_layout = layout;
95  }
96 
97  bool createLayout( DataWidget* parent )
98  {
99  if( parent->layout() != nullptr )
100  {
101  return false;
102  }
103  master_layout = new QVBoxLayout(parent);
104  setMasterLayout(master_layout);
105  return true;
106  }
107 
108  bool createLayout( QLayout* layout)
109  {
110  container_layout = new QHBoxLayout();
111  layout->addItem(container_layout);
112  return true;
113  }
114 
115  void insertWidgets()
116  {
117  p.insertWidgets();
118  createLayout(master_layout);
119 
120  if(check)
121  {
122  container_layout->addWidget(check);
123  }
124  if(label)
125  {
126  container_layout->addWidget(label);
127  }
128  w.createLayout(container_layout); // create the layout for the rest of the widgets
129  w.insertWidgets(); // insert them accordingly
130 
131  }
132 
133  bool createWidgets(DataWidget * parent, const data_type& d, bool readOnly )
134  {
135 
136  if (!p.createWidgets( parent, d, readOnly))
137  return false;
138 
139  const char* name = vhelper::name();
140  bool checkable = vhelper::isCheckable();
141  if (checkable)
142  {
143  check = new QCheckBox(parent);
144  if (name && *name)
145  {
146  check->setText(QString(name));
147  }
148  }
149  else
150  {
151  if (name && *name && N > 1)
152  {
153  label = new QLabel(QString(name),parent);
154  }
155  }
156  if (!w.createWidgets(parent, *vhelper::get(d), readOnly || vhelper::readOnly()))
157  return false;
158 
159  if (checkable)
160  {
161  bool isChecked = vhelper::isChecked(d);
162  check->setChecked(isChecked);
163  if (readOnly || vhelper::readOnly())
164  check->setEnabled(false);
165  else
166  {
167  if (!isChecked)
168  w.setReadOnly(true);
169  parent->connect(check, SIGNAL( toggled(bool) ),parent, SLOT( setWidgetDirty() ));
170  parent->connect(check, SIGNAL( toggled(bool) ),parent, SLOT( setReadOnly(bool) ));
171 
172  }
173  }
174  return true;
175  }
176 
177  void setReadOnly(bool readOnly)
178  {
179  p.setReadOnly(readOnly);
180  w.setReadOnly(readOnly || vhelper::readOnly() || (check && !(check->checkState() == Qt::Checked)));
181  }
182  void readFromData(const data_type& d)
183  {
184  p.readFromData(d);
185  if (check)
186  {
187  bool wasChecked = (check->checkState() == Qt::Checked);
188  bool isChecked = vhelper::isChecked(d);
189  if (isChecked != wasChecked)
190  {
191  check->setChecked(isChecked);
192  if (check->isEnabled())
193  w.setReadOnly(!isChecked);
194  }
195  }
196  w.readFromData(*vhelper::get(d));
197  }
198  void readConstantsFromData(const data_type& d)
199  {
200  p.readConstantsFromData(d);
201  if (vhelper::readOnly())
202  {
203  if (check)
204  {
205  check->setChecked(vhelper::isChecked(d));
206  }
207  w.readFromData(*vhelper::get(d));
208  }
209  }
210  void writeToData(data_type& d)
211  {
212  p.writeToData(d);
213  if (check)
214  {
215  bool isChecked = (check->checkState() == Qt::Checked);
216  vhelper::setChecked(isChecked, d);
217  }
218  value_type v = *vhelper::get(d);
219  w.writeToData(v);
220  vhelper::set(v,d);
221  if ( N == struct_data_trait<T>::NVAR )
222  {
223  shelper::set(d);
224  readConstantsFromData(d); // reread constant fields
225  }
226  }
227 };
