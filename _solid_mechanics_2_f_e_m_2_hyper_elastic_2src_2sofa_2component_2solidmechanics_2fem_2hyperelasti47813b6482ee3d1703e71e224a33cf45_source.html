54  virtual Real getStrainEnergy(StrainInformation<DataTypes> *sinfo, const MaterialParameters<DataTypes> &param) {
55  Real I1=sinfo->trC;
56  MatrixSym C=sinfo->deformationTensor;
57  Real I1square=(Real)(C[0]*C[0] + C[2]*C[2]+ C[5]*C[5]+2*(C[1]*C[1] + C[3]*C[3] + C[4]*C[4]));
58  Real I2=(Real)((pow(I1,(Real)2)- I1square)/2);
59  Real mu=param.parameterArray[0];
60  Real lambda=param.parameterArray[1];
61  return (Real)(-mu*I2/2+(mu/4+lambda/8)*pow(I1,(Real)2)-I1*(3*lambda/4+mu/2));
62 
63 
64  }
65  void deriveSPKTensor(StrainInformation<DataTypes> *sinfo, const MaterialParameters<DataTypes> &param,MatrixSym &SPKTensorGeneral){
66  MatrixSym C=sinfo->deformationTensor;
67  Real I1=sinfo->trC;
68  Real mu=param.parameterArray[0];
69  Real lambda=param.parameterArray[1];
70  MatrixSym ID;
71  ID.identity();
72  SPKTensorGeneral=(C-ID)*mu+ID*lambda/(Real)2.0*(I1-(Real)3.0);
73  }

75  void applyElasticityTensor(StrainInformation<DataTypes> *, const MaterialParameters<DataTypes> &param,const MatrixSym& inputTensor, MatrixSym &outputTensor) {
76  Real mu=param.parameterArray[0];
77  Real lambda=param.parameterArray[1];
78  MatrixSym ID;
79  ID.identity();
80  Real trH=inputTensor[0]+inputTensor[2]+inputTensor[5];
81  outputTensor=ID*trH*lambda/2.0+inputTensor*mu;
82 
83  }
84  virtual void ElasticityTensor(StrainInformation<DataTypes> *, const MaterialParameters<DataTypes> &param,Matrix6 &outputTensor) {
85  Real mu=param.parameterArray[0];
86  Real lambda=param.parameterArray[1];
87  MatrixSym ID;
88  ID.identity();
89  Matrix6 IDHID;
90  IDHID.identity();
91  Matrix6 trIDHID;
92  trIDHID[0]=ID[0]*ID;
93  trIDHID[1]=ID[1]*ID;
94  trIDHID[2]=ID[2]*ID;
95  trIDHID[3]=ID[3]*ID;
96  trIDHID[4]=ID[4]*ID;
97  trIDHID[5]=ID[5]*ID;
98  outputTensor=(IDHID*mu+trIDHID*lambda/(Real)2.0)*2.0;
99 
100  }
