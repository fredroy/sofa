106  void doUpdate() override
107  {
108  helper::WriteOnlyAccessor< Data< type::vector<Index> > > indices = d_indices;
109  indices.clear();
110 
111  unsigned int nb = d_nbLabels.getValue();
112  if(nb<2) return;
113 
114  // convert connectLabels to set for efficient look-up
115  helper::ReadAccessor<Data< type::vector<T> > > connectL(this->d_connectLabels);
116  typedef std::pair<T,T> TPair;
117  std::set<TPair> connectS;
118  for(unsigned int i=0;i<connectL.size()/2;i++)
119  {
120  connectS.insert(TPair(connectL[2*i],connectL[2*i+1]));
121  connectS.insert(TPair(connectL[2*i+1],connectL[2*i]));
122  }
123 
124  // read input label lists
125  std::vector<const VecVLabels* > labels;
126  for(unsigned int l=0; l<nb ; ++l)
127  {
128  helper::ReadAccessor< Data<VecVLabels> > rlab(d_labels[l]);
129  labels.push_back(&rlab.ref());
130  }
131 
132  size_t nbp = (*labels[0]).size();
133  for(size_t i=0; i<nbp;i++)
134  {
135  bool connected = false;
136  for(unsigned int l1=0; l1<nb && !connected;l1++)
137  for(size_t i1=0; i1<(*labels[l1])[i].size() && !connected;i1++)
138  for(unsigned int l2=0; l2<nb && !connected;l2++)
139  if(l1!=l2)
140  for(size_t i2=0; i2<(*labels[l2])[i].size() && !connected;i2++)
141  if(connectS.find(TPair((*labels[l1])[i][i1],(*labels[l2])[i][i2]))!=connectS.end())
142  connected=true;
143  if(connected)
144  indices.push_back((Index)i);
145  }
146  }
