35 /*
36  * Engine which computes a displacement with respect to an origin position, as D(t) = M(t).M(0)^-1
37  * @warning Assumes that the initial positions never change after initialization. This allows computing their inverses only once. To force recomputation, use init().
38  * @author Fran√ßois Faure, 2015;
39  *
40  * Matthieu Nesme, 2015:
41  * Generalization of DisplacementMatrixEngine (kept for backward compatibility)
42  * The output transform type is a template parameter, it can be a Mat4x4f or a Rigid::Coord (ie translation+quaternion)
43  *
44  * Ali Dicko, 2015
45  * Add of a data scale to the DisplacementMatrixEngine which is a scale matrix add to the computation of D(t). Now D(t) = M(t).M(0)^-1.S
46  * By default S is the identity matrix, but the user can set other matrices
47  */
48 template < class DataTypes, class OutputType >
49 class DisplacementTransformEngine : public sofa::core::DataEngine
50 {
51 public:
52  SOFA_CLASS( SOFA_TEMPLATE2( DisplacementTransformEngine, DataTypes, OutputType ), sofa::core::DataEngine );
53 
54  typedef sofa::core::DataEngine Inherit;
55  typedef typename DataTypes::Real Real;
56  typedef typename DataTypes::Coord Coord; // rigid
57  typedef typename DataTypes::VecCoord VecCoord;
58 
59  // inputs
60  Data< VecCoord > d_x0;
61  Data< VecCoord > d_x;
62 
63  // outputs
64  Data< type::vector< OutputType > > d_displacements;
65 
66  // methods
67  DisplacementTransformEngine();
68  void init() override; // compute the inverse matrices
69  void doUpdate() override; // compute the displacements wrt original positions
70 
71  // To simplify the template name in the xml file
72  virtual std::string getTemplateName() const override { return templateName(this); }
73  static std::string templateName(const DisplacementTransformEngine<DataTypes,OutputType>* = nullptr) { return DataTypes::Name()+std::string(",")+defaulttype::DataTypeInfo<OutputType>::name(); }
74 
75 protected:
76  type::vector<OutputType> inverses;
77 
79  void setInverse( OutputType& inv, const Coord& x0 );
80  void mult( OutputType& out, const OutputType& inv, const Coord& x );
81 
82 };
