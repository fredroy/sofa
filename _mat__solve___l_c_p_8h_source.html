31 // --- Resoud un LCP ecrit sous la forme U = q + M.F
32 // --- dim : dimension du pb
33 // --- res[0..dim-1] = U
34 // --- res[dim..2*dim-1] = F
35 // -----------------------------------------------------------------
36 template <Size dim, typename real>
37 [[nodiscard]]
38 bool solveLCP(const Vec<dim,real> &q, const Mat<dim,dim,real> &M, Vec<dim * 2, real> &res)
39 {
40  constexpr real EPSILON_ZERO = 0.00001; // epsilon pour tests = 0
41  constexpr real EPSILON_PIVOT = 0.00000000001; // epsilon pour pivot
42  constexpr Size MAX_NB_LOOP = 50; // nombre maximal de boucles de calcul
43 
44  const Size dim_mult2 = 2 * dim;
45  const Size dim_mult2_plus1 = dim_mult2 + 1;
46 
47  Index ii, jj;
48  int ligPiv; // ligne du pivot
49  int colPiv; // colonne du pivot
50  double pivot; // pivot
51  double min; // recherche du minimum pour le pivot
52  double coeff; // valeur du coefficient de la combinaison lineaire
53  Index boucles; // ii du nombre de passages dans la boucle
54  double mat[dim][dim_mult2_plus1];
55  Index base[dim]; // base des variables non nulles
56 
57  // matrix initialization
58  for (ii = 0; ii < dim; ii++)
59  {
60  // colonnes correspondantes a w
61  for (jj = 0; jj < dim; jj++)
62  {
63  if (jj == ii)
64  mat[ii][jj] = 1;
65  else
66  mat[ii][jj] = 0;
67  }
68 
69  // colonnes correspondantes a z
70  for (; jj < dim_mult2; jj++)
71  mat[ii][jj] = -(M[ii][jj - dim]);
72 
73  // colonne correspondante a q
74  mat[ii][jj] = q[ii];
75  }
76 
77  // initialisation de la base
78  for (ii = 0; ii < dim; ii++)
79  base[ii] = ii;
80 
81  // initialisation du nombre de boucles
82  boucles = 0;
83 
84  // recherche de la ligne du pivot
85  ligPiv = -1;
86  min = -EPSILON_ZERO;
87  for (ii = 0; ii < dim; ii++)
88  {
89  if (mat[ii][dim_mult2] < min)
90  {
91  ligPiv = ii;
92  min = mat[ii][dim_mult2];
93  }
94  }
95 
96  // tant que tous les q[i] ne sont pas > 0 et qu'on ne boucle pas trop
97  while ((ligPiv >= 0) && (boucles < MAX_NB_LOOP))
98  {
99  // augmentation du nombre de passages dans cette boucle
100  boucles++;
101 
102  // recherche de la colonne du pivot
103  if (base[ligPiv] < dim)
104  {
105  // c'est un wi dans la base
106  colPiv = dim + base[ligPiv];
107  }
108  else
109  {
110  // c'est un zi dans la base
111  colPiv = base[ligPiv] - dim;
112  }
113 
114  // stockage de la valeur du pivot
115  pivot = mat[ligPiv][colPiv];
116 
117  // si le pivot est nul, le LCP echoue
118  if (fabs(pivot) < EPSILON_PIVOT)
119  {
120  printf(" + No solution to LCP + \n");
121  boucles = MAX_NB_LOOP;
122  return false;
123  }
124  else
125  {
126  // division de la ligne du pivot par le pivot
127  for (ii = 0; ii < dim_mult2_plus1; ii++)
128  mat[ligPiv][ii] /= pivot;
129 
130  // combinaisons lineaires mettant la colonne du pivot a 0
131  for (ii = 0; ii < dim; ii++)
132  {
133  if (int(ii) != ligPiv)
134  {
135  coeff = mat[ii][colPiv];
136  for (jj = 0; jj < dim_mult2_plus1; jj++)
137  mat[ii][jj] -= coeff * mat[ligPiv][jj];
138  }
139  }
140 
141  // on rentre dans la base la nouvelle variable
142  base[ligPiv] = colPiv;
143 
144  // recherche de la nouvelle ligne du pivot
145  ligPiv = -1;
146  min = -EPSILON_ZERO;
147 
148  for (ii = 0; ii < dim; ii++)
149  {
150  if (mat[ii][dim_mult2] < min)
151  {
152  ligPiv = ii;
153  min = mat[ii][dim_mult2];
154  }
155  }
156  }
157  }
158 
159  // stockage du resultat
160  for (ii = 0; ii < dim_mult2; ii++)
161  res[ii] = 0;
162 
163  // si on est arrive a resoudre le pb, seules les variables en base sont non nulles
164  if (boucles < MAX_NB_LOOP)
165  {
166  for (ii = 0; ii < dim; ii++)
167  res[base[ii]] = mat[ii][dim_mult2];
168  }
169 
170  return true;
171 }
